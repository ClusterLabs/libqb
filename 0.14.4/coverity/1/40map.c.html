50   	void
51   	qb_map_foreach(struct qb_map *map, qb_map_transverse_fn func, void *user_data)
52   	{
53   		const char *key;
54   		void *value;
55   		qb_map_iter_t *i = qb_map_iter_create(map);
56   	
57   		for (key = qb_map_iter_next(i, &value);
58   		     key; key = qb_map_iter_next(i, &value)) {
59   			if (func(key, value, user_data)) {
60   				goto clean_up;
61   			}
62   		}
63   	clean_up:
64   		qb_map_iter_free(i);
65   	}

91   	int32_t
92   	qb_map_notify_add(qb_map_t * m, const char *key, qb_map_notify_fn fn,
93   			  int32_t events, void *user_data)
94   	{
95   		if (key != NULL && events & QB_MAP_NOTIFY_FREE) {
96   			return -EINVAL;
97   		}
98   		if (m->notify_add) {
99   			return m->notify_add(m, key, fn, events, user_data);
100  		} else {
101  			return -ENOSYS;
102  		}
103  	}
104  	
105  	int32_t
106  	qb_map_notify_del(qb_map_t * m, const char *key, qb_map_notify_fn fn,
107  			  int32_t events)
108  	{
109  		if (m->notify_del) {
110  			return m->notify_del(m, key, fn, events, QB_FALSE, NULL);
111  		} else {
112  			return -ENOSYS;
113  		}
114  	}
