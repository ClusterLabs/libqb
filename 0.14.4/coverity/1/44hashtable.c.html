65   	static uint32_t
66   	hash_fnv(const void *value, uint32_t valuelen, uint32_t order)
67   	{
68   		uint8_t *cd = (uint8_t *) value;
69   		uint8_t *ed = (uint8_t *) value + valuelen;
70   		uint32_t hash_result = 0x811c9dc5;
71   		int32_t res;
72   	
73   		while (cd < ed) {
74   			hash_result ^= *cd;
75   			hash_result *= FNV_32_PRIME;
76   			cd++;
77   		}
78   		res = ((hash_result >> order) ^ hash_result) & ((1 << order) - 1);
79   		return (res);
80   	}

82   	static uint32_t
83   	qb_hash_string(const void *key, uint32_t order)
84   	{
85   		char *str = (char *)key;
86   		return hash_fnv(key, strlen(str), order);
87   	}

89   	static struct hash_node *
90   	hashtable_lookup(struct hash_table *t, const char *key)
91   	{
92   		uint32_t hash_entry;
93   		struct qb_list_head *list;
94   		struct hash_node *hash_node;
95   	
96   		hash_entry = qb_hash_string(key, t->order);
97   	
98   		qb_list_for_each(list, &t->hash_buckets[hash_entry].list_head) {
99   	
100  			hash_node = qb_list_entry(list, struct hash_node, list);
101  			if (strcmp(hash_node->key, key) == 0) {
102  				return hash_node;
103  			}
104  		}
105  	
106  		return NULL;
107  	}

111  	{
112  		struct hash_table *t = (struct hash_table *)map;
113  		struct hash_node *n = hashtable_lookup(t, key);

121  	hashtable_node_destroy(struct hash_table *t, struct hash_node *hash_node)
122  	{
123  		struct qb_list_head *pos;
124  		struct qb_list_head *next;
125  		struct qb_map_notifier *tn;
126  	
127  		hashtable_notify(t, hash_node,
128  				 QB_MAP_NOTIFY_DELETED,
129  				 hash_node->key, hash_node->value, NULL);
130  	
131  		qb_list_for_each_safe(pos, next, &hash_node->notifier_head) {
132  			tn = qb_list_entry(pos, struct qb_map_notifier, list);
133  			qb_list_del(&tn->list);
134  			free(tn);
135  		}
136  	
137  		qb_list_del(&hash_node->list);
138  		free(hash_node);
139  	}

141  	static void
142  	hashtable_node_deref(struct qb_map *map, struct hash_node *hash_node)
143  	{
144  		struct hash_table *t = (struct hash_table *)map;
145  	
146  		hash_node->refcount--;
147  		if (hash_node->refcount > 0) {
148  			return;
149  		}
150  		hashtable_node_destroy(t, hash_node);
151  	}

154  	hashtable_rm_with_hash(struct qb_map *map, const char *key, uint32_t hash_entry)
155  	{
156  		struct hash_table *hash_table = (struct hash_table *)map;
157  		struct qb_list_head *list;
158  		struct qb_list_head *next;
159  		struct hash_node *hash_node;
160  	
161  		qb_list_for_each_safe(list, next,
162  		                      &hash_table->hash_buckets[hash_entry].list_head) {
163  	
164  			hash_node = qb_list_entry(list, struct hash_node, list);
165  			if (strcmp(hash_node->key, key) == 0) {
166  				hashtable_node_deref(map, hash_node);
167  				hash_table->count--;
168  				return QB_TRUE;
169  			}
170  		}
171  	
172  		return QB_FALSE;
173  	}

269  	static int32_t
270  	hashtable_notify_add(qb_map_t * m, const char *key,
271  			     qb_map_notify_fn fn, int32_t events, void *user_data)
272  	{
273  		struct hash_table *t = (struct hash_table *)m;
274  		struct qb_map_notifier *f;
275  		struct hash_node *n;
276  		struct qb_list_head *head = NULL;
277  		struct qb_list_head *list;
278  		int add_to_tail = QB_FALSE;
279  	
280  		if (key) {
281  			n = hashtable_lookup(t, key);
282  			if (n) {
283  				head = &n->notifier_head;
284  			}
285  		} else {
286  			head = &t->notifier_head;
287  		}
288  		if (head == NULL) {
289  			return -ENOENT;
290  		}
291  		if (events & QB_MAP_NOTIFY_FREE) {
292  			add_to_tail = QB_TRUE;
293  		}
294  	
295  		qb_list_for_each(list, head) {
296  			f = qb_list_entry(list, struct qb_map_notifier, list);
297  	
298  			if (events & QB_MAP_NOTIFY_FREE &&
299  			    f->events == events) {
300  				/* only one free notifier */
301  				return -EEXIST;
302  			}
303  			if (f->events == events &&
304  			    f->user_data == user_data &&
305  			    f->callback == fn) {
306  				return -EEXIST;
307  			}
308  		}
309  	
310  		f = malloc(sizeof(struct qb_map_notifier));
311  		if (f == NULL) {
312  			return -errno;
313  		}
314  		f->events = events;
315  		f->user_data = user_data;
316  		f->callback = fn;
317  		qb_list_init(&f->list);
318  	
319  		if (add_to_tail) {
320  			qb_list_add_tail(&f->list, head);
321  		} else {
322  			qb_list_add(&f->list, head);
323  		}
324  		return 0;
325  	}

327  	static int32_t
328  	hashtable_notify_del(qb_map_t * m, const char *key,
329  			     qb_map_notify_fn fn, int32_t events,
330  			     int32_t cmp_userdata, void *user_data)
331  	{
332  		struct hash_table *t = (struct hash_table *)m;
333  		struct qb_map_notifier *f;
334  		struct hash_node *n;
335  		struct qb_list_head *head = NULL;
336  		struct qb_list_head *list;
337  		struct qb_list_head *next;
338  		int32_t found = QB_FALSE;
339  	
340  		if (key) {
341  			n = hashtable_lookup(t, key);
342  			if (n) {
343  				head = &n->notifier_head;
344  			}
345  		} else {
346  			head = &t->notifier_head;
347  		}
348  		if (head == NULL) {
349  			return -ENOENT;
350  		}
351  	
352  		qb_list_for_each_safe(list, next, head) {
353  			f = qb_list_entry(list, struct qb_map_notifier, list);
354  	
355  			if (f->events == events && f->callback == fn) {
356  				if (cmp_userdata && (f->user_data == user_data)) {
357  					found = QB_TRUE;
358  					qb_list_del(&f->list);
359  					free(f);
360  				} else if (!cmp_userdata) {
361  					found = QB_TRUE;
362  					qb_list_del(&f->list);
363  					free(f);
364  				}
365  			}
366  		}
367  		if (found) {
368  			return 0;
369  		} else {
370  			return -ENOENT;
371  		}
372  	}

394  	static const char *
395  	hashtable_iter_next(qb_map_iter_t * it, void **value)
396  	{
397  		struct hashtable_iter *hi = (struct hashtable_iter *)it;
398  		struct hash_table *hash_table = (struct hash_table *)hi->i.m;
399  		struct qb_list_head *ln;
400  		struct hash_node *hash_node = NULL;
401  		int found = QB_FALSE;
402  		int cont = QB_TRUE;
403  		int b;
404  	
405  		if (hi->node == NULL) {
406  			cont = QB_FALSE;
407  		}
408  		for (b = hi->bucket; b < hash_table->hash_buckets_len && !found; b++) {
409  			if (cont) {
410  				ln = &hi->node->list;
411  				cont = QB_FALSE;
412  			} else {
413  				ln = &hash_table->hash_buckets[b].list_head;
414  			}
415  			hash_node = qb_list_first_entry(ln, struct hash_node, list);
416  			qb_list_for_each_entry_from(hash_node,
417  			                &hash_table->hash_buckets[b].list_head, list) {
418  				if (hash_node->refcount > 0) {
419  					found = QB_TRUE;
420  					hash_node->refcount++;
421  					hi->bucket = b;
422  					*value = hash_node->value;
423  					break;
424  				}
425  			}
426  		}
427  	
428  		if (hi->node) {
429  			hashtable_node_deref(hi->i.m, hi->node);
430  		}
431  		if (!found) {
432  			return NULL;
433  		}
434  		hi->node = hash_node;
435  		return hash_node->key;
436  	}

444  	static void
445  	hashtable_destroy(struct qb_map *map)
446  	{
447  		struct hash_table *hash_table = (struct hash_table *)map;
448  		struct qb_list_head *pos;
449  		struct qb_list_head *next;
450  		struct qb_map_notifier *tn;
451  		int32_t i;
452  	
453  		for (i = 0; i < hash_table->hash_buckets_len; i++) {
454  			hashtable_node_deref_under_bucket(map, i);
455  		}
456  	
457  		qb_list_for_each_safe(pos, next, &hash_table->notifier_head) {
458  			tn = qb_list_entry(pos, struct qb_map_notifier, list);
459  			qb_list_del(&tn->list);
460  			free(tn);
461  		}
462  	
463  		free(hash_table);
464  	}

467  	hashtable_node_deref_under_bucket(struct qb_map *map, int32_t hash_entry)
468  	{
469  		struct hash_table *hash_table = (struct hash_table *)map;
470  		struct hash_node *hash_node;
471  		struct qb_list_head *pos;
472  		struct qb_list_head *next;
473  	
474  		qb_list_for_each_safe(pos, next,

476  			hash_node = qb_list_entry(pos, struct hash_node, list);

478  			hash_table->count--;

482  	qb_map_t *
483  	qb_hashtable_create(size_t max_size)
484  	{
485  		int32_t i;
486  		int32_t order;
487  		int32_t n = max_size;
488  		uint64_t size;
489  		struct hash_table *ht;
490  	
491  		for (i = 0; n; i++) {
492  			n >>= 1;
493  		}
494  		order = QB_MAX(i, 3);
495  	
496  		size = sizeof(struct hash_table) +
497  		    (sizeof(struct hash_bucket) * (1 << order));
498  	
499  		ht = calloc(1, size);
500  		if (ht == NULL) {
501  			return NULL;
502  		}
503  	
504  		ht->map.put = hashtable_put;
505  		ht->map.get = hashtable_get;
506  		ht->map.rm = hashtable_rm;
507  		ht->map.count_get = hashtable_count_get;
508  		ht->map.iter_create = hashtable_iter_create;
509  		ht->map.iter_next = hashtable_iter_next;
510  		ht->map.iter_free = hashtable_iter_free;
511  		ht->map.destroy = hashtable_destroy;
512  		ht->map.notify_add = hashtable_notify_add;
513  		ht->map.notify_del = hashtable_notify_del;
514  		ht->count = 0;
515  		ht->order = order;
516  		qb_list_init(&ht->notifier_head);
517  	
518  		ht->hash_buckets_len = 1 << order;
519  		for (i = 0; i < ht->hash_buckets_len; i++) {
520  			qb_list_init(&ht->hash_buckets[i].list_head);
521  		}
522  		return (qb_map_t *) ht;
523  	}
