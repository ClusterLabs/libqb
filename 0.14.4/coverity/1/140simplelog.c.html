32   	static uint8_t _log_priority = LOG_WARNING;

34   	static void
35   	func_one(void)
36   	{
37   		FILE *fd;
38   	
39   		qb_enter();
40   		qb_logt(LOG_DEBUG, MY_TAG_TWO, "arf arf?");
41   		qb_logt(LOG_CRIT, MY_TAG_THREE, "arrrg!");
42   		qb_logt(134, MY_TAG_THREE, "big priority");
43   		qb_logt(LOG_ERR, MY_TAG_THREE, "oops, I did it again");
44   		qb_log(LOG_INFO, "are you aware ...");
45   	
46   		fd = fopen("/nothing.txt", "r+");
47   		if (fd == NULL) {
48   			qb_perror(LOG_ERR, "can't open(\"/nothing.txt\")");
49   		} else {
50   			fclose(fd);
51   		}
52   		qb_leave();
53   	}

55   	static void
56   	func_two(void)
57   	{
58   		qb_enter();
59   		qb_logt(LOG_DEBUG, 0, "arf arf?");
60   		qb_logt(LOG_CRIT, MY_TAG_ONE, "arrrg!");
61   		qb_log(LOG_ERR, "oops, I did it again");
62   		qb_logt(LOG_INFO, MY_TAG_THREE, "are you aware ...");
63   		qb_leave();
64   	}

66   	static void
67   	show_usage(const char *name)
68   	{
69   		printf("usage: \n");
70   		printf("%s <options>\n", name);
71   		printf("\n");
72   		printf("  options:\n");
73   		printf("\n");
74   		printf("  -v             verbose\n");
75   		printf("  -t             threaded logging\n");
76   		printf("  -e             log to stderr\n");
77   		printf("  -b             log to blackbox\n");
78   		printf("  -f <filename>  log to a file\n");
79   		printf("  -h             show this help text\n");
80   		printf("\n");
81   	}

83   	static int32_t do_blackbox = QB_FALSE;
84   	static int32_t do_threaded = QB_FALSE;
85   	
86   	static void
87   	sigsegv_handler(int sig)
88   	{
89   		(void)signal(SIGSEGV, SIG_DFL);
90   		if (do_blackbox) {
91   			qb_log_blackbox_write_to_file("simple-log.fdata");
92   		}
93   		qb_log_ctl(QB_LOG_BLACKBOX, QB_LOG_CONF_ENABLED, QB_FALSE);
94   		raise(SIGSEGV);
95   	}

97   	static const char *
98   	my_tags_stringify(uint32_t tags)
99   	{
100  		if (qb_bit_is_set(tags, QB_LOG_TAG_LIBQB_MSG_BIT)) {
101  			return "libqb";
102  		} else if (qb_bit_is_set(tags, 0)) {
103  			return "ONE";
104  		} else if (qb_bit_is_set(tags, 1)) {
105  			return "TWO";
106  		} else if (qb_bit_is_set(tags, 2)) {
107  			return "THREE";
108  		} else {
109  			return "MAIN";
110  		}
111  	}

113  	static void
114  	trace_logger(int32_t t,
115  		     struct qb_log_callsite *cs, time_t timestamp, const char *msg)
116  	{
117  		char output_buffer[QB_LOG_MAX_LEN];
118  		output_buffer[0] = '\0';
119  		qb_log_target_format(t, cs, timestamp, msg, output_buffer);
120  		fprintf(stderr, "%s\n", output_buffer);
121  	}
122  	
123  	static void
124  	m_filter(struct qb_log_callsite *cs)
125  	{
126  		if ((cs->priority >= LOG_ALERT &&
127  		     cs->priority <= _log_priority) &&
128  		    strcmp(cs->filename, __FILE__) == 0) {
129  			qb_bit_set(cs->targets, QB_LOG_STDERR);
130  		} else {
131  			qb_bit_clear(cs->targets, QB_LOG_STDERR);
132  		}
133  	}

148  			switch (opt) {

152  			case 't':

155  			case 'e':

158  			case 'o':

161  			case 'b':

164  			case 'f':

167  			case 'v':
168  				_log_priority++;
169  				break;
170  			case 'h':

176  		}
177

201  		if (do_stdout) {

203  					   QB_LOG_FILTER_FILE, __FILE__,
204  					   LOG_ALERT, QB_MIN(LOG_DEBUG, _log_priority));
205  			qb_log_format_set(QB_LOG_STDOUT, "[%p] %4g: %f:%l %b");
206  			qb_log_ctl(QB_LOG_STDOUT, QB_LOG_CONF_ENABLED, QB_TRUE);
207  		}
208  		if (do_blackbox) {
209  			qb_log_filter_ctl(QB_LOG_BLACKBOX, QB_LOG_FILTER_ADD,
210  					  QB_LOG_FILTER_FILE, "*", LOG_DEBUG);
211  			qb_log_ctl(QB_LOG_BLACKBOX, QB_LOG_CONF_SIZE, 4096);
212  			qb_log_ctl(QB_LOG_BLACKBOX, QB_LOG_CONF_THREADED, QB_FALSE);
213  			qb_log_ctl(QB_LOG_BLACKBOX, QB_LOG_CONF_ENABLED, QB_TRUE);
214  		}
215  		if (logfile) {
216  			log_fd = qb_log_file_open(logfile);
217  			qb_log_filter_ctl(log_fd, QB_LOG_FILTER_ADD,
218  					  QB_LOG_FILTER_FILE, __FILE__, _log_priority);
219  			qb_log_format_set(log_fd, "[%N] %t %n() [%p] %b");
220  			qb_log_ctl(log_fd, QB_LOG_CONF_THREADED, do_threaded);
221  			qb_log_ctl(log_fd, QB_LOG_CONF_ENABLED, QB_TRUE);
222  		}
