91   	static void
92   	_check_connection_state(struct qb_ipcc_connection * c, int32_t res)
93   	{
94   		if (res >= 0) return;
95   	
96   		if (qb_ipc_us_sock_error_is_disconnected(res)) {
97   			errno = -res;
98   			qb_util_perror(LOG_DEBUG,
99   				    "interpreting result %d as a disconnect",
100  				    res);
101  			c->is_connected = QB_FALSE;
102  		}
103  	}

105  	static struct qb_ipc_one_way *
106  	_event_sock_one_way_get(struct qb_ipcc_connection * c)
107  	{
108  		if (c->needs_sock_for_poll) {
109  			return &c->setup;
110  		}
111  		if (c->event.type == QB_IPC_SOCKET) {
112  			return &c->event;
113  		}
114  		return NULL;
115  	}

119  	{
120  		if (c->needs_sock_for_poll) {

123  		if (c->response.type == QB_IPC_SOCKET) {
124  			return &c->response;
125  		}
126  		return NULL;
127  	}

129  	ssize_t
130  	qb_ipcc_send(struct qb_ipcc_connection * c, const void *msg_ptr, size_t msg_len)
131  	{
132  		ssize_t res;
133  		ssize_t res2;
134  	
135  		if (c == NULL || msg_len > c->request.max_msg_size) {
136  			return -EINVAL;
137  		}
138  		if (c->funcs.fc_get) {
139  			res = c->funcs.fc_get(&c->request);
140  			if (res < 0) {
141  				return res;
142  			} else if (res > 0 && res <= c->fc_enable_max) {
143  				return -EAGAIN;
144  			} else {
145  				/*
146  				 * we can transmit
147  				 */
148  			}
149  		}
150  	
151  		res = c->funcs.send(&c->request, msg_ptr, msg_len);
152  		if (res == msg_len && c->needs_sock_for_poll) {
153  			do {
154  				res2 = qb_ipc_us_send(&c->setup, msg_ptr, 1);
155  			} while (res2 == -EAGAIN);
156  			if (res2 == -EPIPE) {
157  				res2 = -ENOTCONN;
158  			}
159  			if (res2 != 1) {
160  				res = res2;
161  			}
162  		}
163  		_check_connection_state(c, res);
164  		return res;
165  	}

222  	ssize_t
223  	qb_ipcc_recv(struct qb_ipcc_connection * c, void *msg_ptr,
224  		     size_t msg_len, int32_t ms_timeout)
225  	{
226  		int32_t res = 0;
227  		int32_t res2 = 0;
228  		int32_t poll_ms = 0;
229  		if (c == NULL) {
230  			return -EINVAL;
231  		}
232  	
233  		res = c->funcs.recv(&c->response, msg_ptr, msg_len, ms_timeout);
234  		if (res == -EAGAIN || res == -ETIMEDOUT) {
235  			struct qb_ipc_one_way *ow = _response_sock_one_way_get(c);
236  	
237  			if (ow == NULL) return res;
238  	                if (res == -EAGAIN) poll_ms = ms_timeout;
239  	
240  			res2 = qb_ipc_us_ready(ow, poll_ms, POLLIN);
241  			if (res2 < 0) {
242  				res = res2;
243  			}
244  		}
245  		_check_connection_state(c, res);
246  		return res;
247  	}

249  	ssize_t
250  	qb_ipcc_sendv_recv(qb_ipcc_connection_t * c,
251  			   const struct iovec * iov, uint32_t iov_len,
252  			   void *res_msg, size_t res_len, int32_t ms_timeout)
253  	{
254  		ssize_t res = 0;
255  		int32_t timeout_now;
256  		int32_t timeout_rem = ms_timeout;
257  	
258  		if (c == NULL) {
259  			return -EINVAL;
260  		}
261  	
262  		if (c->funcs.fc_get) {
263  			res = c->funcs.fc_get(&c->request);
264  			if (res < 0) {
265  				return res;
266  			} else if (res > 0 && res <= c->fc_enable_max) {
267  				return -EAGAIN;
268  			} else {
269  				/*
270  				 * we can transmit
271  				 */
272  			}
273  		}
274  	
275  		res = qb_ipcc_sendv(c, iov, iov_len);
276  		if (res < 0) {
277  			return res;
278  		}
279  	
280  		do {
281  			if (timeout_rem > QB_IPC_MAX_WAIT_MS || ms_timeout == -1) {
282  				timeout_now = QB_IPC_MAX_WAIT_MS;
283  			} else {
284  				timeout_now = timeout_rem;
285  			}
286  	
287  			res = qb_ipcc_recv(c, res_msg, res_len, timeout_now);
288  			if (res == -ETIMEDOUT) {
289  				if (ms_timeout < 0) {
290  					res = -EAGAIN;
291  				} else {
292  					timeout_rem -= timeout_now;
293  					if (timeout_rem > 0) {
294  						res = -EAGAIN;
295  					}
296  				}
297  			} else	if (res < 0 && res != -EAGAIN) {
298  				errno = -res;
299  				qb_util_perror(LOG_DEBUG,
300  					       "qb_ipcc_recv %d timeout:(%d/%d)",
301  					       res, timeout_now, timeout_rem);
302  			}
303  		} while (res == -EAGAIN && c->is_connected);
304  	
305  		return res;
306  	}

308  	int32_t
309  	qb_ipcc_fd_get(struct qb_ipcc_connection * c, int32_t * fd)
310  	{
311  		if (c == NULL) {
312  			return -EINVAL;
313  		}
314  		if (c->event.type == QB_IPC_SOCKET) {
315  			*fd = c->event.u.us.sock;
316  		} else {
317  			*fd = c->setup.u.us.sock;
318  		}
319  		return 0;
320  	}

322  	ssize_t
323  	qb_ipcc_event_recv(struct qb_ipcc_connection * c, void *msg_pt,
324  			   size_t msg_len, int32_t ms_timeout)
325  	{
326  		char one_byte = 1;
327  		int32_t res;
328  		ssize_t size;
329  		struct qb_ipc_one_way *ow = NULL;
330  	
331  		if (c == NULL) {
332  			return -EINVAL;
333  		}
334  		ow = _event_sock_one_way_get(c);
335  		if (ow) {
336  			res = qb_ipc_us_ready(ow, ms_timeout, POLLIN);
337  			if (res < 0) {
338  				_check_connection_state(c, res);
339  				return res;
340  			}
341  		}
342  		size = c->funcs.recv(&c->event, msg_pt, msg_len, ms_timeout);
343  		if (size < 0) {
344  			_check_connection_state(c, size);
345  			return size;
346  		}
347  		if (c->needs_sock_for_poll) {
348  			res = qb_ipc_us_recv(&c->setup, &one_byte, 1, -1);
349  			if (res < 0) {
350  				_check_connection_state(c, res);
351  				return res;
352  			}
353  		}
354  		return size;
355  	}

357  	void
358  	qb_ipcc_disconnect(struct qb_ipcc_connection *c)
359  	{
360  		struct qb_ipc_one_way *ow = NULL;
361  		int32_t res = 0;
362  	
363  		qb_util_log(LOG_DEBUG, "%s()", __func__);
364  	
365  		if (c == NULL) {
366  			return;
367  		}
368  	
369  		ow = _event_sock_one_way_get(c);
370  		if (ow) {
371  			res = qb_ipc_us_ready(ow, 0, POLLIN);
372  			_check_connection_state(c, res);
373  			qb_ipcc_us_sock_close(ow->u.us.sock);
374  		}
375  	
376  		if (c->funcs.disconnect) {
377  			c->funcs.disconnect(c);
378  		}
379  		free(c->receive_buf);
380  		free(c);
381  	}

400  	int32_t
401  	qb_ipcc_is_connected(qb_ipcc_connection_t *c)
402  	{
403  		struct qb_ipc_one_way *ow;
404  	
405  		if (c == NULL) {
406  			return QB_FALSE;
407  		}
408  	
409  		ow = _response_sock_one_way_get(c);
410  		if (ow) {
411  			_check_connection_state(c, qb_ipc_us_ready(ow, 0, POLLIN));
412  		}
413  	
414  		return c->is_connected;
