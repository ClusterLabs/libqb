28   	static void
29   	notify_fn(uint32_t event, char *key, void *old_value, void *value,
30   		  void *user_data)
31   	{
32   		if (event == QB_MAP_NOTIFY_FREE) {
33   			fprintf(stderr, "Notify[FREE] %s [%d]\n",
34   				key, *(int *)old_value);
35   			free(old_value);
36   		} else if (event == QB_MAP_NOTIFY_DELETED) {
37   			fprintf(stderr, "Notify[DELETED] %s [%d]\n",
38   				key, *(int *)old_value);
39   		} else if (event == QB_MAP_NOTIFY_REPLACED) {
40   			fprintf(stderr, "Notify[REPLACED] %s [%d] -> [%d]\n",
41   				key, *(int *)old_value, *(int *)value);
42   		} else {
43   			fprintf(stderr, "Notify[%d] %s \n", event, key);
44   			if (value != NULL) {
45   				fprintf(stderr, " value = [%d]\n", *(int *)value);
46   			}
47   			if (old_value != NULL) {
48   				fprintf(stderr, " old value = [%d]\n",
49   					*(int *)old_value);
50   			}
51   		}
52   	}
