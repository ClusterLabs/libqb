52   	static void
53   	_benchmark(qb_ipcc_connection_t *conn, int write_size)
54   	{
55   		struct iovec iov[2];
56   		unsigned int res;
57   		struct qb_ipc_request_header hdr;
58   		int write_count = 0;
59   		float secs;
60   	
61   		alarm_notice = 0;
62   		hdr.size = write_size;
63   		hdr.id = QB_IPC_MSG_USER_START + 1;
64   	
65   		iov[0].iov_base = &hdr;
66   		iov[0].iov_len = sizeof(struct qb_ipc_request_header);
67   	
68   		iov[1].iov_base = data;
69   		iov[1].iov_len = write_size - sizeof(struct qb_ipc_request_header);
70   	
71   		alarm (10);
72   	
73   		qb_util_stopwatch_start(sw);
74   		do {
75   			res = qb_ipcc_sendv(conn, iov, 2);
76   			if (res == write_size) {
77   				write_count++;
78   			}
79   		} while (alarm_notice == 0 && (res == write_size || res == -EAGAIN));
80   		if (res < 0) {
81   			perror("qb_ipcc_sendv");
82   		}
83   		qb_util_stopwatch_stop(sw);
84   		secs = qb_util_stopwatch_sec_elapsed_get(sw);
85   	
86   		printf ("%5d messages sent ", write_count);
87   		printf ("%5d bytes per write ", write_size);
88   		printf ("%7.3f Seconds runtime ", secs);
89   		printf ("%9.3f TP/s ",
90   			((float)write_count) / secs);
91   		printf ("%7.3f MB/s.\n",
92   			((float)write_count) * ((float)write_size) / secs);
93   	}
