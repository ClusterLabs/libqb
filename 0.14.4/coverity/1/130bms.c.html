102  	static int32_t s1_msg_process_fn(qb_ipcs_connection_t *c,
103  			void *data, size_t size)
104  	{
105  		struct qb_ipc_request_header *req_pt = (struct qb_ipc_request_header *)data;
106  		struct qb_ipc_response_header response;
107  		ssize_t res;
108  	
109  		qb_log(LOG_TRACE, "msg:%d, size:%d",
110  		       req_pt->id, req_pt->size);
111  		response.size = sizeof(struct qb_ipc_response_header);
112  		response.id = 13;
113  		response.error = 0;
114  		if (blocking) {
115  			res = qb_ipcs_response_send(c, &response,
116  					sizeof(response));
117  			if (res < 0) {
118  				qb_perror(LOG_ERR, "qb_ipcs_response_send");
119  			}
120  		}
121  		if (events) {
122  			res = qb_ipcs_event_send(c, &response,
123  					sizeof(response));
124  			if (res < 0) {
125  				qb_perror(LOG_ERR, "qb_ipcs_event_send");
126  			}
127  		}
128  		return 0;
129  	}

169  	static gboolean
170  	gio_read_socket (GIOChannel *gio, GIOCondition condition, gpointer data)
171  	{
172  		struct gio_to_qb_poll *adaptor = (struct gio_to_qb_poll *)data;
173  		gint fd = g_io_channel_unix_get_fd(gio);
174  	
175  		return (adaptor->fn(fd, condition, adaptor->data) == 0);
176  	}

178  	static int32_t my_g_dispatch_add(enum qb_loop_priority p, int32_t fd, int32_t evts,
179  		void *data, qb_ipcs_dispatch_fn_t fn)
180  	{
181  		struct gio_to_qb_poll *adaptor;
182  		GIOChannel *channel;
183  		int32_t res = 0;
184  	
185  		res = qb_array_grow(gio_map, fd + 1);
186  		if (res < 0) {
187  			return res;
188  		}
189  		res = qb_array_index(gio_map, fd, (void**)&adaptor);
190  		if (res < 0) {
191  			return res;
192  		}
193  		if (adaptor->is_used) {
194  			return -EEXIST;
195  		}
196  	
197  		channel = g_io_channel_unix_new(fd);
198  		if (!channel) {
199  			return -ENOMEM;
200  		}
201  	
202  		adaptor->channel = channel;
203  		adaptor->fn = fn;
204  		adaptor->events = evts;
205  		adaptor->data = data;
206  		adaptor->p = p;
207  		adaptor->is_used = TRUE;
208  	
209  		g_io_add_watch(channel, evts, gio_read_socket, adaptor);
210  		return 0;
211  	}

219  	static int32_t my_g_dispatch_del(int32_t fd)
220  	{
221  		struct gio_to_qb_poll *adaptor;
222  		if (qb_array_index(gio_map, fd, (void**)&adaptor) == 0) {
223  			g_io_channel_unref(adaptor->channel);
224  			adaptor->is_used = FALSE;
225  		}
226  		return 0;
227  	}

254  	int32_t main(int32_t argc, char *argv[])
255  	{
256  		const char *options = "nevhmpsug";
257  		int32_t opt;
258  		int32_t rc;
259  		enum qb_ipc_type ipc_type = QB_IPC_SHM;
260  		struct qb_ipcs_service_handlers sh = {
261  			.connection_accept = s1_connection_accept_fn,
262  			.connection_created = s1_connection_created_fn,
263  			.msg_process = s1_msg_process_fn,
264  			.connection_destroyed = s1_connection_destroyed_fn,
265  			.connection_closed = s1_connection_closed_fn,
266  		};
267  		struct qb_ipcs_poll_handlers ph = {
268  			.job_add = my_job_add,
269  			.dispatch_add = my_dispatch_add,
270  			.dispatch_mod = my_dispatch_mod,
271  			.dispatch_del = my_dispatch_del,
272  		};
273  	#ifdef HAVE_GLIB
274  		struct qb_ipcs_poll_handlers glib_ph = {
275  			.job_add = NULL, /* FIXME */
276  			.dispatch_add = my_g_dispatch_add,
277  			.dispatch_mod = my_g_dispatch_mod,
278  			.dispatch_del = my_g_dispatch_del,
279  		};
280  	#endif /* HAVE_GLIB */
281  	
282  		while ((opt = getopt(argc, argv, options)) != -1) {
283  			switch (opt) {
284  			case 'm':
285  				ipc_type = QB_IPC_SHM;
286  				break;
287  			case 'u':
288  				ipc_type = QB_IPC_SOCKET;
289  				break;
290  			case 'n':	/* non-blocking */
291  				blocking = QB_FALSE;
292  				break;
293  			case 'e':	/* events */
294  				events = QB_TRUE;
295  				break;
296  			case 'g':
297  				use_glib = QB_TRUE;
298  				break;
299  			case 'v':
300  				verbose++;
301  				break;
302  			case 'h':
303  			default:
304  				show_usage(argv[0]);
305  				exit(0);
306  				break;
307  			}
308  		}
309  		signal(SIGINT, sigusr1_handler);
310  		signal(SIGILL, sigusr1_handler);
311  		signal(SIGTERM, sigusr1_handler);
312  	
313  		qb_log_init("bms", LOG_USER, LOG_EMERG);
314  		qb_log_ctl(QB_LOG_SYSLOG, QB_LOG_CONF_ENABLED, QB_FALSE);
315  		qb_log_filter_ctl(QB_LOG_STDERR, QB_LOG_FILTER_ADD,
316  				  QB_LOG_FILTER_FILE, "*", LOG_INFO + verbose);
317  		qb_log_ctl(QB_LOG_STDERR, QB_LOG_CONF_ENABLED, QB_TRUE);
318  	
319  		if (!use_glib) {
320  			bms_loop = qb_loop_create();
321  			s1 = qb_ipcs_create("bm1", 0, ipc_type, &sh);
322  			if (s1 == 0) {
323  				qb_perror(LOG_ERR, "qb_ipcs_create");
324  				exit(1);
325  			}
326  			qb_ipcs_poll_handlers_set(s1, &ph);
327  			rc = qb_ipcs_run(s1);
328  			if (rc != 0) {
329  				errno = -rc;
330  				qb_perror(LOG_ERR, "qb_ipcs_run");
331  				exit(1);
332  			}
333  			qb_loop_run(bms_loop);
334  		} else {
335  	#ifdef HAVE_GLIB
336  			glib_loop = g_main_loop_new(NULL, FALSE);
337  	
338  			gio_map = qb_array_create(64, sizeof(struct gio_to_qb_poll));
339  	
340  			s1 = qb_ipcs_create("bm1", 0, ipc_type, &sh);
341  			if (s1 == 0) {
342  				qb_perror(LOG_ERR, "qb_ipcs_create");
343  				exit(1);
344  			}
345  			qb_ipcs_poll_handlers_set(s1, &glib_ph);
346  			rc = qb_ipcs_run(s1);
347  			if (rc != 0) {
348  				errno = -rc;
349  				qb_perror(LOG_ERR, "qb_ipcs_run");
350  				exit(1);
351  			}
352  	
353  			g_main_loop_run(glib_loop);
354  	#else
355  			qb_log(LOG_ERR, "You don't seem to have glib-devel installed.\n");
356  	#endif
357  		}
358  		return EXIT_SUCCESS;
359  	}
