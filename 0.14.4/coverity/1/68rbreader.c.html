30   	
31   	#define ONE_MEG 1048576
32   	static qb_ringbuffer_t *rb = NULL;
33   	static int keep_reading = QB_TRUE;
34   	int8_t buffer[ONE_MEG];
35   	
36   	
37   	static void sigterm_handler(int32_t num)
38   	{
(1) Event tainted_string_warning: Using a non string-literal such as "descriptor.format" as a format-string argument can lead to a vulnerability if a user can control its content. [details] 
39   		qb_log(LOG_INFO, "signal %d", num);
40   		keep_reading = QB_FALSE;
41   	}
42   	
43   	int32_t main(int32_t argc, char *argv[])
44   	{
45   		ssize_t num_read;
46   	
47   		signal(SIGINT, sigterm_handler);
48   	
49   		qb_log_init("rbreader", LOG_USER, LOG_EMERG);
50   		qb_log_ctl(QB_LOG_SYSLOG, QB_LOG_CONF_ENABLED, QB_FALSE);
51   		qb_log_filter_ctl(QB_LOG_STDERR, QB_LOG_FILTER_ADD,
52   				  QB_LOG_FILTER_FILE, "*", LOG_TRACE);
53   		qb_log_ctl(QB_LOG_STDERR, QB_LOG_CONF_ENABLED, QB_TRUE);
54   	
55   		rb = qb_rb_open("tester", ONE_MEG * 3,
56   				QB_RB_FLAG_SHARED_PROCESS |
57   				QB_RB_FLAG_CREATE, 0);
58   	
59   		if (rb == NULL) {
60   			qb_perror(LOG_ERR, "failed to create ringbuffer");
61   			return -1;
62   		}
63   		while (keep_reading) {
64   			num_read = qb_rb_chunk_read(rb, buffer,
65   						    ONE_MEG, 0);
66   			if (num_read == -ETIMEDOUT) {
67   				usleep(100000);
68   			} else if (num_read < 0) {
69   				errno = -num_read;
70   				qb_perror(LOG_ERR, "nothing to read");
71   			}
72   		}
73   		qb_rb_close(rb);
74   		return 0;
75   	}
