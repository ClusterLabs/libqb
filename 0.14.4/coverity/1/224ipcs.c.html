36   	qb_ipcs_service_t *
37   	qb_ipcs_create(const char *name,
38   		       int32_t service_id,
39   		       enum qb_ipc_type type, struct qb_ipcs_service_handlers *handlers)
40   	{
41   		struct qb_ipcs_service *s;
42   	
43   		s = calloc(1, sizeof(struct qb_ipcs_service));
44   		if (s == NULL) {
45   			return NULL;
46   		}
47   		if (type == QB_IPC_NATIVE) {
48   	#ifdef DISABLE_IPC_SHM
49   			s->type = QB_IPC_SOCKET;
50   	#else
51   			s->type = QB_IPC_SHM;
52   	#endif /* DISABLE_IPC_SHM */
53   		} else {
54   			s->type = type;
55   		}
56   	
57   		s->pid = getpid();
58   		s->needs_sock_for_poll = QB_FALSE;
59   		s->poll_priority = QB_LOOP_MED;
60   		s->ref_count = 1;
61   	
62   		s->service_id = service_id;
63   		(void)strlcpy(s->name, name, NAME_MAX);
64   	
65   		s->serv_fns.connection_accept = handlers->connection_accept;
66   		s->serv_fns.connection_created = handlers->connection_created;
67   		s->serv_fns.msg_process = handlers->msg_process;
68   		s->serv_fns.connection_closed = handlers->connection_closed;
69   		s->serv_fns.connection_destroyed = handlers->connection_destroyed;
70   	
71   		qb_list_init(&s->connections);
72   		qb_list_init(&s->list);
73   		qb_list_add(&s->list, &qb_ipc_services);
74   	
75   		return s;
76   	}

88   	int32_t
89   	qb_ipcs_run(struct qb_ipcs_service *s)
90   	{
91   		int32_t res = 0;
92   	
93   		if (s->poll_fns.dispatch_add == NULL ||
94   		    s->poll_fns.dispatch_mod == NULL ||
95   		    s->poll_fns.dispatch_del == NULL) {
96   			return -EINVAL;
97   		}
98   	
99   		switch (s->type) {
100  		case QB_IPC_SOCKET:
101  			qb_ipcs_us_init((struct qb_ipcs_service *)s);
102  			break;
103  		case QB_IPC_SHM:
104  	#ifdef DISABLE_IPC_SHM
105  			res = -ENOTSUP;
106  	#else
107  			qb_ipcs_shm_init((struct qb_ipcs_service *)s);
108  	#endif /* DISABLE_IPC_SHM */
109  			break;
110  		case QB_IPC_POSIX_MQ:
111  		case QB_IPC_SYSV_MQ:
112  			res = -ENOTSUP;
113  			break;
114  		default:
115  			res = -EINVAL;
116  			break;
117  		}
118  		if (res < 0) {
119  			qb_ipcs_unref(s);
120  			return res;
121  		}
122  		res = qb_ipcs_us_publish(s);
123  		if (res < 0) {
124  			(void)qb_ipcs_us_withdraw(s);
125  			qb_ipcs_unref(s);
126  			return res;
127  		}
128  	
129  		return res;
130  	}

132  	static int32_t
133  	_modify_dispatch_descriptor_(struct qb_ipcs_connection *c)
134  	{
135  		qb_ipcs_dispatch_mod_fn disp_mod = c->service->poll_fns.dispatch_mod;
136  	
137  		if (c->service->type == QB_IPC_SOCKET) {
138  			return disp_mod(c->service->poll_priority,
139  					c->event.u.us.sock,
140  					c->poll_events, c,
141  					qb_ipcs_dispatch_connection_request);
142  		} else {
143  			return disp_mod(c->service->poll_priority,
144  					c->setup.u.us.sock,
145  					c->poll_events, c,
146  					qb_ipcs_dispatch_connection_request);
147  		}
148  		return -EINVAL;
149  	}

151  	void
152  	qb_ipcs_request_rate_limit(struct qb_ipcs_service *s,
153  				   enum qb_ipcs_rate_limit rl)
154  	{
155  		struct qb_ipcs_connection *c;
156  		enum qb_loop_priority old_p = s->poll_priority;
157  		struct qb_list_head *pos;
158  		struct qb_list_head *n;
159  	
160  		switch (rl) {
161  		case QB_IPCS_RATE_FAST:
162  			s->poll_priority = QB_LOOP_HIGH;
163  			break;
164  		case QB_IPCS_RATE_SLOW:
165  		case QB_IPCS_RATE_OFF:
166  		case QB_IPCS_RATE_OFF_2:
167  			s->poll_priority = QB_LOOP_LOW;
168  			break;
169  		default:
170  		case QB_IPCS_RATE_NORMAL:
171  			s->poll_priority = QB_LOOP_MED;
172  			break;
173  		}
174  	
175  		qb_list_for_each_safe(pos, n, &s->connections) {
176  	
177  			c = qb_list_entry(pos, struct qb_ipcs_connection, list);
178  			qb_ipcs_connection_ref(c);
179  	
180  			if (rl == QB_IPCS_RATE_OFF) {
181  				qb_ipcs_flowcontrol_set(c, 1);
182  			} else if (rl == QB_IPCS_RATE_OFF_2) {
183  				qb_ipcs_flowcontrol_set(c, 2);
184  			} else {
185  				qb_ipcs_flowcontrol_set(c, QB_FALSE);
186  			}
187  			if (old_p != s->poll_priority) {
188  				(void)_modify_dispatch_descriptor_(c);
189  			}
190  			qb_ipcs_connection_unref(c);
191  		}
192  	}

200  	void
201  	qb_ipcs_unref(struct qb_ipcs_service *s)
202  	{
203  		int32_t free_it;
204  		struct qb_ipcs_connection *c = NULL;
205  		struct qb_list_head *pos;
206  		struct qb_list_head *n;
207  	
208  		assert(s->ref_count > 0);
209  		free_it = qb_atomic_int_dec_and_test(&s->ref_count);
210  		if (free_it) {
211  			qb_util_log(LOG_DEBUG, "%s() - destroying", __func__);
212  			qb_list_for_each_safe(pos, n, &s->connections) {
213  				c = qb_list_entry(pos, struct qb_ipcs_connection, list);
214  				if (c == NULL) {
215  					continue;
216  				}
217  				qb_ipcs_disconnect(c);
218  			}
219  			(void)qb_ipcs_us_withdraw(s);
220  			free(s);
221  		}
222  	}

257  	ssize_t
258  	qb_ipcs_response_send(struct qb_ipcs_connection *c, const void *data,
259  			      size_t size)
260  	{
261  		ssize_t res;
262  	
263  		if (c == NULL) {
264  			return -EINVAL;
265  		}
266  		qb_ipcs_connection_ref(c);
267  		res = c->service->funcs.send(&c->response, data, size);
268  		if (res == size) {
269  			c->stats.responses++;
270  		} else if (res == -EAGAIN || res == -ETIMEDOUT) {
271  			struct qb_ipc_one_way *ow = _response_sock_one_way_get(c);
272  			if (ow) {
273  				ssize_t res2  = qb_ipc_us_ready(ow, 0, POLLOUT);
274  				if (res2 < 0) {
275  					res = res2;
276  				}
277  			}
278  			c->stats.send_retries++;
279  		}
280  		qb_ipcs_connection_unref(c);
281  	
282  		return res;
283  	}

313  	static int32_t
314  	resend_event_notifications(struct qb_ipcs_connection *c)
315  	{
316  		ssize_t res = 0;
317  	
318  		if (c->outstanding_notifiers > 0) {
319  			res = qb_ipc_us_send(&c->setup, c->receive_buf,
320  					     c->outstanding_notifiers);
321  		}
322  		if (res > 0) {
323  			c->outstanding_notifiers -= res;
324  		}
325  		assert(c->outstanding_notifiers >= 0);
326  		if (c->outstanding_notifiers == 0) {
327  			c->poll_events = POLLIN | POLLPRI | POLLNVAL;
328  			(void)_modify_dispatch_descriptor_(c);
329  		}
330  		return res;
331  	}

333  	static int32_t
334  	new_event_notification(struct qb_ipcs_connection * c)
335  	{
336  		ssize_t res = 0;
337  	
338  		if (!c->service->needs_sock_for_poll) {
339  			return res;
340  		}
341  	
342  		assert(c->outstanding_notifiers >= 0);
343  		if (c->outstanding_notifiers > 0) {
344  			c->outstanding_notifiers++;
345  		} else {
346  			res = qb_ipc_us_send(&c->setup, &c->outstanding_notifiers, 1);
347  			if (res == -EAGAIN) {
348  				/*
349  				 * notify the client later, when we can.
350  				 */
351  				c->outstanding_notifiers++;
352  				c->poll_events = POLLOUT | POLLIN | POLLPRI | POLLNVAL;
353  				(void)_modify_dispatch_descriptor_(c);
354  			}
355  		}
356  		return res;
357  	}

359  	ssize_t
360  	qb_ipcs_event_send(struct qb_ipcs_connection * c, const void *data, size_t size)
361  	{
362  		ssize_t res;
363  		ssize_t resn;
364  	
365  		if (c == NULL) {
366  			return -EINVAL;
367  		}
368  		qb_ipcs_connection_ref(c);
369  	
370  		res = c->service->funcs.send(&c->event, data, size);
371  		if (res == size) {
372  			c->stats.events++;
373  			resn = new_event_notification(c);
374  			if (resn < 0 && resn != -EAGAIN) {
375  				errno = -resn;
376  				qb_util_perror(LOG_WARNING,
377  					       "new_event_notification (%s)",
378  					       c->description);
379  				res = resn;
380  			}
381  		} else if (res == -EAGAIN || res == -ETIMEDOUT) {
382  			struct qb_ipc_one_way *ow = _event_sock_one_way_get(c);
383  			if (ow) {
384  				resn  = qb_ipc_us_ready(ow, 0, POLLOUT);
385  				if (resn < 0) {
386  					res = resn;
387  				}
388  			}
389  			c->stats.send_retries++;
390  		}
391  	
392  		qb_ipcs_connection_unref(c);
393  		return res;
394  	}

578  				rerun_job =
579  				    (qb_loop_job_dispatch_fn) qb_ipcs_disconnect;
580  				res = c->service->poll_fns.job_add(QB_LOOP_LOW,
581  								   c,
582  								   rerun_job);
583  				if (res != 0) {
584  					/* last ditch attempt to cleanup */
585  					qb_ipcs_connection_unref(c);
586  				}
587  			}
588  		}
589  	}

592  	qb_ipcs_flowcontrol_set(struct qb_ipcs_connection *c, int32_t fc_enable)
593  	{
594  		if (c == NULL) {
595  			return;
596  		}

599  			c->fc_enabled = fc_enable;

601  			c->stats.flow_control_count++;
602  		}

605  	static int32_t
606  	_process_request_(struct qb_ipcs_connection *c, int32_t ms_timeout)
607  	{
608  		int32_t res = 0;
609  		ssize_t size;
610  		struct qb_ipc_request_header *hdr;
611  	
612  		qb_ipcs_connection_ref(c);
613  	
614  		if (c->service->funcs.peek && c->service->funcs.reclaim) {
615  			size = c->service->funcs.peek(&c->request, (void **)&hdr,
616  						      ms_timeout);
617  		} else {
618  			hdr = c->receive_buf;
619  			size = c->service->funcs.recv(&c->request,
620  						      hdr,
621  						      c->request.max_msg_size,
622  						      ms_timeout);
623  		}
624  		if (size < 0) {
625  			if (size != -EAGAIN && size != -ETIMEDOUT) {
626  				qb_util_perror(LOG_DEBUG,
627  					       "recv from client connection failed (%s)",
628  					       c->description);
629  			} else {
630  				c->stats.recv_retries++;
631  			}
632  			res = size;
633  			goto cleanup;
634  		} else if (size == 0 || hdr->id == QB_IPC_MSG_DISCONNECT) {
635  			qb_util_log(LOG_DEBUG, "client requesting a disconnect (%s)",
636  				    c->description);
637  			qb_ipcs_disconnect(c);
638  			c = NULL;
639  			res = -ESHUTDOWN;
640  		} else {
641  			c->stats.requests++;
642  			res = c->service->serv_fns.msg_process(c, hdr, hdr->size);
643  			/* 0 == good, negative == backoff */
644  			if (res < 0) {
645  				res = -ENOBUFS;
646  			} else {
647  				res = size;
648  			}
649  		}
650  	
651  		if (c && c->service->funcs.peek && c->service->funcs.reclaim) {
652  			c->service->funcs.reclaim(&c->request);
653  		}
654  	
655  	cleanup:
656  		qb_ipcs_connection_unref(c);
657  		return res;
658  	}

660  	#define IPC_REQUEST_TIMEOUT 10
661  	#define MAX_RECV_MSGS 50
662  	
663  	int32_t
664  	qb_ipcs_dispatch_service_request(int32_t fd, int32_t revents, void *data)
665  	{
666  		int32_t res = _process_request_((struct qb_ipcs_connection *)data,
667  						IPC_REQUEST_TIMEOUT);
668  		if (res > 0) {
669  			return 0;
670  		}
671  		return res;
672  	}

674  	static ssize_t
675  	_request_q_len_get(struct qb_ipcs_connection *c)
676  	{
677  		ssize_t q_len;
678  		if (c->service->funcs.q_len_get) {
679  			q_len = c->service->funcs.q_len_get(&c->request);
680  			if (q_len <= 0) {
681  				return q_len;
682  			}
683  			if (c->service->poll_priority == QB_LOOP_MED) {
684  				q_len = QB_MIN(q_len, 5);
685  			} else if (c->service->poll_priority == QB_LOOP_LOW) {
686  				q_len = 1;
687  			} else {
688  				q_len = QB_MIN(q_len, MAX_RECV_MSGS);
689  			}
690  		} else {
691  			q_len = 1;
692  		}
693  		return q_len;
694  	}
695  	
696  	int32_t
697  	qb_ipcs_dispatch_connection_request(int32_t fd, int32_t revents, void *data)
698  	{
699  		struct qb_ipcs_connection *c = (struct qb_ipcs_connection *)data;
700  		char bytes[MAX_RECV_MSGS];
701  		int32_t res;
702  		int32_t res2;
703  		int32_t recvd = 0;
704  		ssize_t avail;
705  	
706  		if (revents & POLLNVAL) {
707  			qb_util_log(LOG_DEBUG, "NVAL conn (%s)", c->description);
708  			return -EINVAL;
709  		}
710  		if (revents & POLLHUP) {
711  			qb_util_log(LOG_DEBUG, "HUP conn (%s)", c->description);
712  			qb_ipcs_disconnect(c);
713  			return -ESHUTDOWN;
714  		}
715  	
716  		if (revents & POLLOUT) {
717  			res = resend_event_notifications(c);
718  			if (res < 0 && res != -EAGAIN) {
719  				errno = -res;
720  				qb_util_perror(LOG_WARNING,
721  					       "resend_event_notifications (%s)",
722  					       c->description);
723  			}
724  			if ((revents & POLLIN) == 0) {
725  				return 0;
726  			}
727  		}
728  		if (c->fc_enabled) {
729  			return 0;
730  		}
731  		avail = _request_q_len_get(c);
732  	
733  		if (c->service->needs_sock_for_poll && avail == 0) {
734  			res2 = qb_ipc_us_recv(&c->setup, bytes, 1, 0);
735  			if (qb_ipc_us_sock_error_is_disconnected(res2)) {
736  				errno = -res2;
737  				qb_util_perror(LOG_WARNING, "conn (%s) disconnected",
738  					       c->description);
739  				qb_ipcs_disconnect(c);
740  				return -ESHUTDOWN;
741  			} else {
742  				qb_util_log(LOG_WARNING,
743  					    "conn (%s) Nothing in q but got POLLIN on fd:%d (res2:%d)",
744  					    c->description, fd, res2);
745  				return 0;
746  			}
747  		}
748  	
749  		do {
750  			res = _process_request_(c, IPC_REQUEST_TIMEOUT);
751  			if (res > 0 || res == -ENOBUFS || res == -EINVAL) {
752  				recvd++;
753  			}
754  			if (res > 0) {
755  				avail--;
756  			}
757  		} while (avail > 0 && res > 0 && !c->fc_enabled);
758  	
759  		if (c->service->needs_sock_for_poll && recvd > 0) {
760  			res2 = qb_ipc_us_recv(&c->setup, bytes, recvd, -1);
761  			if (res2 < 0) {
762  				errno = -res2;
763  				qb_util_perror(LOG_ERR,
764  					       "error receiving from setup sock (%s)",
765  					       c->description);
766  			}
767  		}
768  	
769  		res = QB_MIN(0, res);
770  		if (res == -EAGAIN || res == -ETIMEDOUT || res == -ENOBUFS) {
771  			res = 0;
772  		}
773  		if (res != 0) {
774  			if (res != -ENOTCONN) {
775  				/*
776  				 * Abnormal state (ENOTCONN is normal shutdown).
777  				 */
778  				errno = -res;
779  				qb_util_perror(LOG_ERR, "request returned error (%s)",
780  					       c->description);
781  			}
782  			qb_ipcs_connection_unref(c);
783  		}
784  	
785  		return res;
786  	}
