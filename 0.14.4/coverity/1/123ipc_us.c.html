92   	static void sigpipe_ctl(enum qb_sigpipe_ctl ctl)
93   	{
94   	#if !defined(MSG_NOSIGNAL) && !defined(SO_NOSIGPIPE)
95   		struct sigaction act;
96   		struct sigaction oact;
97   	
98   		act.sa_handler = SIG_IGN;
99   	
100  		if (ctl == QB_SIGPIPE_IGNORE) {
101  			sigaction(SIGPIPE, &act, &oact);
102  		} else {
103  			sigaction(SIGPIPE, &oact, NULL);
104  		}
105  	#endif  /* !MSG_NOSIGNAL && !defined(SO_NOSIGPIPE) */
106  	}

112  	ssize_t
113  	qb_ipc_us_send(struct qb_ipc_one_way *one_way, const void *msg, size_t len)
114  	{
115  		int32_t result;
116  		int32_t processed = 0;
117  		char *rbuf = (char *)msg;
118  	
119  		sigpipe_ctl(QB_SIGPIPE_IGNORE);
120  	
121  	retry_send:
122  		result = send(one_way->u.us.sock,
123  			      &rbuf[processed],
124  			      len - processed,
125  			      MSG_NOSIGNAL);
126  	
127  		if (result == -1) {
128  			if (errno == EAGAIN && processed > 0) {
129  				goto retry_send;
130  			} else {
131  				sigpipe_ctl(QB_SIGPIPE_DEFAULT);
132  				return -errno;
133  			}
134  		}
135  	
136  		processed += result;
137  		if (processed != len) {
138  			goto retry_send;
139  		}
140  	
141  		sigpipe_ctl(QB_SIGPIPE_DEFAULT);
142  	
143  		if (one_way->type == QB_IPC_SOCKET) {
144  			struct ipc_us_control *ctl = NULL;
145  			ctl = (struct ipc_us_control *)one_way->u.us.shared_data;
146  			if (ctl) {
147  				qb_atomic_int_inc(&ctl->sent);
148  			}
149  		}
150  		return processed;
151  	}

153  	static ssize_t
154  	qb_ipc_us_sendv(struct qb_ipc_one_way *one_way, const struct iovec *iov,
155  			size_t iov_len)
156  	{
157  		int32_t result;
158  		int32_t processed = 0;
159  		int32_t total_processed = 0;
160  		int32_t iov_p = 0;
161  		char *rbuf = (char *)iov[iov_p].iov_base;
162  	
163  		sigpipe_ctl(QB_SIGPIPE_IGNORE);
164  	
165  	retry_send:
166  		result = send(one_way->u.us.sock,
167  			      &rbuf[processed],
168  			      iov[iov_p].iov_len - processed,
169  			      MSG_NOSIGNAL);
170  	
171  		if (result == -1) {
172  			if (errno == EAGAIN &&
173  			    (processed > 0 || iov_p > 0)) {
174  				goto retry_send;
175  			} else {
176  				sigpipe_ctl(QB_SIGPIPE_DEFAULT);
177  				return -errno;
178  			}
179  		}
180  	
181  		processed += result;
182  		if (processed == iov[iov_p].iov_len) {
183  			iov_p++;
184  			total_processed += processed;
185  			if (iov_p < iov_len) {
186  				processed = 0;
187  				rbuf = (char *)iov[iov_p].iov_base;
188  				goto retry_send;
189  			}
190  		} else {
191  			goto retry_send;
192  		}
193  	
194  		sigpipe_ctl(QB_SIGPIPE_DEFAULT);
195  	
196  		if (one_way->type == QB_IPC_SOCKET) {
197  			struct ipc_us_control *ctl;
198  			ctl = (struct ipc_us_control *)one_way->u.us.shared_data;
199  			if (ctl) {
200  				qb_atomic_int_inc(&ctl->sent);
201  			}
202  		}
203  		return total_processed;
204  	}

206  	static ssize_t
207  	qb_ipc_us_recv_msghdr(int32_t s, struct msghdr *hdr, char *msg, size_t len)
208  	{
209  		int32_t result;
210  		int32_t processed = 0;
211  	
212  		sigpipe_ctl(QB_SIGPIPE_IGNORE);
213  	
214  	retry_recv:
215  		hdr->msg_iov->iov_base = &msg[processed];
216  		hdr->msg_iov->iov_len = len - processed;
217  	
218  		result = recvmsg(s, hdr, MSG_NOSIGNAL | MSG_WAITALL);
219  		if (result == -1 && errno == EAGAIN) {
220  			goto retry_recv;
221  		}
222  		if (result == -1) {
223  			sigpipe_ctl(QB_SIGPIPE_DEFAULT);
224  			return -errno;
225  		}
226  		if (result == 0) {
227  			sigpipe_ctl(QB_SIGPIPE_DEFAULT);
228  			qb_util_log(LOG_DEBUG,
229  				    "recv(fd %d) got 0 bytes assuming ENOTCONN", s);
230  			return -ENOTCONN;
231  		}
232  	
233  		processed += result;
234  		if (processed != len) {
235  			goto retry_recv;
236  		}
237  		sigpipe_ctl(QB_SIGPIPE_DEFAULT);
238  		assert(processed == len);
239  	
240  		return processed;
241  	}

243  	int32_t
244  	qb_ipc_us_sock_error_is_disconnected(int err)
245  	{
246  		if (err == -EAGAIN ||
247  		    err == -ETIMEDOUT ||
248  		    err == -EINTR ||
249  	#ifdef EWOULDBLOCK
250  		    err == -EWOULDBLOCK ||
251  	#endif
252  		    err == -EINVAL) {
253  			return QB_FALSE;
254  		}
255  		return QB_TRUE;
256  	}

258  	int32_t
259  	qb_ipc_us_ready(struct qb_ipc_one_way * one_way,
260  			int32_t ms_timeout, int32_t events)
261  	{
262  		struct pollfd ufds;
263  		int32_t poll_events;
264  	
265  		ufds.fd = one_way->u.us.sock;
266  		ufds.events = events;
267  		ufds.revents = 0;
268  	
269  		poll_events = poll(&ufds, 1, ms_timeout);
270  		if ((poll_events == -1 && errno == EINTR) || poll_events == 0) {
271  			return -EAGAIN;
272  		} else if (poll_events == -1) {
273  			return -errno;
274  		} else if (poll_events == 1 && (ufds.revents & POLLERR)) {
275  			qb_util_log(LOG_DEBUG, "poll(fd %d) got POLLERR", one_way->u.us.sock);
276  			return -ENOTCONN;
277  		} else if (poll_events == 1 && (ufds.revents & POLLHUP)) {
278  			qb_util_log(LOG_DEBUG, "poll(fd %d) got POLLHUP", one_way->u.us.sock);
279  			return -ENOTCONN;
280  		} else if (poll_events == 1 && (ufds.revents & POLLNVAL)) {
281  			qb_util_log(LOG_DEBUG, "poll(fd %d) got POLLNVAL", one_way->u.us.sock);
282  			return -ENOTCONN;
283  		}
284  		return 0;
285  	}

287  	/*
288  	 * recv an entire message - and try hard to get all of it.
289  	 */
290  	ssize_t
291  	qb_ipc_us_recv(struct qb_ipc_one_way * one_way,
292  		       void *msg, size_t len, int32_t timeout)
293  	{
294  		int32_t result;
295  		int32_t final_rc = 0;
296  		int32_t processed = 0;
297  		int32_t to_recv = len;
298  		char *data = msg;
299  	
300  		sigpipe_ctl(QB_SIGPIPE_IGNORE);
301  	
302  	retry_recv:
303  		result = recv(one_way->u.us.sock, &data[processed], to_recv,
304  			      MSG_NOSIGNAL | MSG_WAITALL);
305  	
306  		if (result == -1) {
307  			if (errno == EAGAIN &&
308  			    (processed > 0 || timeout == -1)) {
309  				result = qb_ipc_us_ready(one_way, timeout,
310  							 POLLIN);
311  				if (result == 0 || result == -EAGAIN) {
312  					goto retry_recv;
313  				}
314  				final_rc = result;
315  				goto cleanup_sigpipe;
316  			} else if (errno == ECONNRESET || errno == EPIPE) {
317  				final_rc = -ENOTCONN;
318  				goto cleanup_sigpipe;
319  			} else {
320  				final_rc = -errno;
321  				goto cleanup_sigpipe;
322  			}
323  		}
324  	
325  		if (result == 0) {
326  			final_rc = -ENOTCONN;
327  			goto cleanup_sigpipe;
328  		}
329  		processed += result;
330  		to_recv -= result;
331  		if (processed != len) {
332  			goto retry_recv;
333  		}
334  		final_rc = processed;
335  	
336  		if (one_way->type == QB_IPC_SOCKET) {
337  			struct ipc_us_control *ctl = NULL;
338  			ctl = (struct ipc_us_control *)one_way->u.us.shared_data;
339  			if (ctl) {
340  				(void)qb_atomic_int_dec_and_test(&ctl->sent);
341  			}
342  		}
343  	
344  	 cleanup_sigpipe:
345  		sigpipe_ctl(QB_SIGPIPE_DEFAULT);
346  		return final_rc;
347  	}

349  	/*
350  	 * recv a message of unknown size.
351  	 */
352  	static ssize_t
353  	qb_ipc_us_recv_at_most(struct qb_ipc_one_way * one_way,
354  		       void *msg, size_t len, int32_t timeout)
355  	{
356  		int32_t result;
357  		int32_t final_rc = 0;
358  		int32_t processed = 0;
359  		int32_t to_recv = sizeof(struct qb_ipc_request_header);
360  		char *data = msg;
361  		struct ipc_us_control *ctl = NULL;
362  		struct qb_ipc_request_header *hdr = NULL;
363  	
364  		sigpipe_ctl(QB_SIGPIPE_IGNORE);
365  	
366  	retry_recv:
367  		result = recv(one_way->u.us.sock, &data[processed], to_recv,
368  			      MSG_NOSIGNAL | MSG_WAITALL);
369  		if (result == -1) {
370  			if (errno == EAGAIN &&
371  			    (processed > 0 || timeout == -1)) {
372  				/*
373  				 * Don't spin too hard else we can consume too
374  				 * much cpu.
375  				 */
376  				result = qb_ipc_us_ready(one_way,
377  							 100,
378  							 POLLIN);
379  				if (result == 0 || result == -EAGAIN) {
380  					goto retry_recv;
381  				}
382  				final_rc = result;
383  				goto cleanup_sigpipe;
384  			} else {
385  				final_rc = -errno;
386  				goto cleanup_sigpipe;
387  			}
388  		} else if (result == 0) {
389  			final_rc = -ENOTCONN;
390  			goto cleanup_sigpipe;
391  		}
392  	
393  		processed += result;
394  		if (processed >= sizeof(struct qb_ipc_request_header) && hdr == NULL) {
395  			hdr = (struct qb_ipc_request_header*)msg;
396  		}
397  		if (hdr) {
398  			to_recv = hdr->size - processed;
399  		} else {
400  			to_recv = len - processed;
401  		}
402  		if (to_recv > 0) {
403  			goto retry_recv;
404  		}
405  		final_rc = processed;
406  	
407  		ctl = (struct ipc_us_control *)one_way->u.us.shared_data;
408  		if (ctl) {
409  			(void)qb_atomic_int_dec_and_test(&ctl->sent);
410  		}
411  	
412  	 cleanup_sigpipe:
413  		sigpipe_ctl(QB_SIGPIPE_DEFAULT);
414  		return final_rc;
415  	}

418  	static int32_t
419  	qb_ipcc_us_sock_connect(const char *socket_name, int32_t * sock_pt)
420  	{
421  		int32_t request_fd;
422  		struct sockaddr_un address;
423  		int32_t res = 0;
424  	
425  		request_fd = socket(PF_UNIX, SOCK_STREAM, 0);
426  		if (request_fd == -1) {
427  			return -errno;
428  		}
429  	
430  		socket_nosigpipe(request_fd);
431  	
432  		res = qb_sys_fd_nonblock_cloexec_set(request_fd);
433  		if (res < 0) {
434  			goto error_connect;
435  		}
436  	
437  		memset(&address, 0, sizeof(struct sockaddr_un));
438  		address.sun_family = AF_UNIX;
439  	#ifdef HAVE_STRUCT_SOCKADDR_UN_SUN_LEN
440  		address.sun_len = QB_SUN_LEN(&address);
441  	#endif
442  	
443  	#if defined(QB_LINUX) || defined(QB_CYGWIN)
444  		snprintf(address.sun_path + 1, UNIX_PATH_MAX - 1, "%s", socket_name);
445  	#else
446  		snprintf(address.sun_path, UNIX_PATH_MAX, "%s/%s", SOCKETDIR,
447  			 socket_name);
448  	#endif
449  		if (connect(request_fd, (struct sockaddr *)&address,
450  			    QB_SUN_LEN(&address)) == -1) {
451  			res = -errno;
452  			goto error_connect;
453  		}
454  	
455  		*sock_pt = request_fd;
456  		return 0;
457  	
458  	error_connect:
459  		close(request_fd);
460  		*sock_pt = -1;
461  	
462  		return res;
463  	}

472  	int32_t
473  	qb_ipcc_us_setup_connect(struct qb_ipcc_connection *c,
474  				 struct qb_ipc_connection_response *r)
475  	{
476  		int32_t res;
477  		struct qb_ipc_connection_request request;
478  	#ifdef QB_LINUX
479  		int off = 0;
480  		int on = 1;
481  	#endif
482  	
483  		res = qb_ipcc_us_sock_connect(c->name, &c->setup.u.us.sock);
484  		if (res != 0) {
485  			return res;
486  		}
487  	
488  	#ifdef QB_LINUX
489  		setsockopt(c->setup.u.us.sock, SOL_SOCKET, SO_PASSCRED, &on, sizeof(on));
490  	#endif
491  	
492  		memset(&request, 0, sizeof(request));
493  		request.hdr.id = QB_IPC_MSG_AUTHENTICATE;
494  		request.hdr.size = sizeof(request);
495  		request.max_msg_size = c->setup.max_msg_size;
496  		res = qb_ipc_us_send(&c->setup, &request, request.hdr.size);
497  		if (res < 0) {
498  			qb_ipcc_us_sock_close(c->setup.u.us.sock);
499  			return res;
500  		}
501  	
502  	#ifdef QB_LINUX
503  		setsockopt(c->setup.u.us.sock, SOL_SOCKET, SO_PASSCRED, &off, sizeof(off));
504  	#endif
505  	
506  		res =
507  		    qb_ipc_us_recv(&c->setup, r,
508  				   sizeof(struct qb_ipc_connection_response), -1);
509  		if (res < 0) {
510  			return res;
511  		}
512  	
513  		if (r->hdr.error != 0) {
514  			return r->hdr.error;
515  		}
516  		return 0;
517  	}

519  	static void
520  	qb_ipcc_us_disconnect(struct qb_ipcc_connection *c)
521  	{
522  		munmap(c->request.u.us.shared_data, SHM_CONTROL_SIZE);
523  		unlink(c->request.u.us.shared_file_name);
524  		close(c->request.u.us.sock);
525  		close(c->event.u.us.sock);
526  	}

528  	int32_t
529  	qb_ipcc_us_connect(struct qb_ipcc_connection *c,
530  			   struct qb_ipc_connection_response *r)
531  	{
532  		int32_t res;
533  		struct qb_ipc_event_connection_request request;
534  		char path[PATH_MAX];
535  		int32_t fd_hdr;
536  		char * shm_ptr;
537  	
538  		qb_atomic_init();
539  	
540  		c->needs_sock_for_poll = QB_FALSE;
541  		c->funcs.send = qb_ipc_us_send;
542  		c->funcs.sendv = qb_ipc_us_sendv;
543  		c->funcs.recv = qb_ipc_us_recv_at_most;
544  		c->funcs.fc_get = qb_ipc_us_fc_get;
545  		c->funcs.disconnect = qb_ipcc_us_disconnect;
546  	
547  		c->request.u.us.sock = c->setup.u.us.sock;
548  		c->response.u.us.sock = c->setup.u.us.sock;
549  		c->setup.u.us.sock = -1;
550  	
551  		fd_hdr = qb_sys_mmap_file_open(path, r->request,
552  					       SHM_CONTROL_SIZE, O_RDWR);
553  		if (fd_hdr < 0) {
554  			res = fd_hdr;
555  			errno = -fd_hdr;
556  			qb_util_perror(LOG_ERR, "couldn't open file for mmap");
557  			return res;
558  		}
559  		(void)strlcpy(c->request.u.us.shared_file_name, r->request, NAME_MAX);
560  		shm_ptr = mmap(0, SHM_CONTROL_SIZE,
561  			       PROT_READ | PROT_WRITE, MAP_SHARED, fd_hdr, 0);
562  	
563  		if (shm_ptr == MAP_FAILED) {
564  			res = -errno;
565  			qb_util_perror(LOG_ERR, "couldn't create mmap for header");
566  			goto cleanup_hdr;
567  		}
568  		c->request.u.us.shared_data = shm_ptr;
569  		c->response.u.us.shared_data = shm_ptr + sizeof(struct ipc_us_control);
570  		c->event.u.us.shared_data =  shm_ptr + (2 * sizeof(struct ipc_us_control));
571  	
572  		close(fd_hdr);
573  	
574  		res = qb_ipcc_us_sock_connect(c->name, &c->event.u.us.sock);
575  		if (res != 0) {
576  			goto cleanup_hdr;
577  		}
578  	
579  		memset(&request, 0, sizeof(request));
580  		request.hdr.id = QB_IPC_MSG_NEW_EVENT_SOCK;
581  		request.hdr.size = sizeof(request);
582  		request.connection = r->connection;
583  		res = qb_ipc_us_send(&c->event, &request, request.hdr.size);
584  		if (res < 0) {
585  			qb_ipcc_us_sock_close(c->event.u.us.sock);
586  			goto cleanup_hdr;
587  		}
588  	
589  		return 0;
590  	
591  	cleanup_hdr:
592  		close(fd_hdr);
593  		unlink(r->request);
594  		munmap(c->request.u.us.shared_data, SHM_CONTROL_SIZE);
595  		return res;
596  	}

603  	int32_t
604  	qb_ipcs_us_publish(struct qb_ipcs_service * s)
605  	{
606  		struct sockaddr_un un_addr;
607  		int32_t res;
608  	
609  		/*
610  		 * Create socket for IPC clients, name socket, listen for connections
611  		 */
612  		s->server_sock = socket(PF_UNIX, SOCK_STREAM, 0);
613  		if (s->server_sock == -1) {
614  			res = -errno;
615  			qb_util_perror(LOG_ERR, "Cannot create server socket");
616  			return res;
617  		}
618  	
619  		res = qb_sys_fd_nonblock_cloexec_set(s->server_sock);
620  		if (res < 0) {
621  			goto error_close;
622  		}
623  	
624  		memset(&un_addr, 0, sizeof(struct sockaddr_un));
625  		un_addr.sun_family = AF_UNIX;
626  	#if defined(QB_BSD) || defined(QB_DARWIN)
627  		un_addr.sun_len = SUN_LEN(&un_addr);
628  	#endif
629  	
630  		qb_util_log(LOG_INFO, "server name: %s", s->name);
631  	#if defined(QB_LINUX) || defined(QB_CYGWIN)
632  		snprintf(un_addr.sun_path + 1, UNIX_PATH_MAX - 1, "%s", s->name);
633  	#else
634  		{
635  			struct stat stat_out;
636  			res = stat(SOCKETDIR, &stat_out);
637  			if (res == -1 || (res == 0 && !S_ISDIR(stat_out.st_mode))) {
638  				res = -errno;
639  				qb_util_log(LOG_CRIT,
640  					    "Required directory not present %s",
641  					    SOCKETDIR);
642  				goto error_close;
643  			}
644  			snprintf(un_addr.sun_path, UNIX_PATH_MAX, "%s/%s", SOCKETDIR,
645  				 s->name);
646  			unlink(un_addr.sun_path);
647  		}
648  	#endif
649  	
650  		res = bind(s->server_sock, (struct sockaddr *)&un_addr,
651  			   QB_SUN_LEN(&un_addr));
652  		if (res) {
653  			res = -errno;
654  			qb_util_perror(LOG_ERR, "Could not bind AF_UNIX (%s)",
655  				       un_addr.sun_path);
656  			goto error_close;
657  		}
658  	
659  		/*
660  		 * Allow everyone to write to the socket since the IPC layer handles
661  		 * security automatically
662  		 */
663  	#if !defined(QB_LINUX) && !defined(QB_CYGWIN)
664  		res = chmod(un_addr.sun_path, S_IRWXU | S_IRWXG | S_IRWXO);
665  	#endif
666  		if (listen(s->server_sock, SERVER_BACKLOG) == -1) {
667  			qb_util_perror(LOG_ERR, "socket listen failed");
668  		}
669  	
670  		res = s->poll_fns.dispatch_add(s->poll_priority, s->server_sock,
671  					       POLLIN | POLLPRI | POLLNVAL,
672  					       s, qb_ipcs_us_connection_acceptor);
673  		return res;
674  	
675  	error_close:
676  		close(s->server_sock);
677  		return res;
678  	}

681  	qb_ipcs_us_withdraw(struct qb_ipcs_service * s)
682  	{
683  		qb_util_log(LOG_INFO, "withdrawing server sockets");
684  		shutdown(s->server_sock, SHUT_RDWR);

689  	static int32_t
690  	handle_new_connection(struct qb_ipcs_service *s,
691  			      int32_t auth_result,
692  			      int32_t sock,
693  			      void *msg, size_t len, struct ipc_auth_ugp *ugp)
694  	{
695  		struct qb_ipcs_connection *c = NULL;
696  		struct qb_ipc_connection_request *req = msg;
697  		int32_t res = auth_result;
698  		int32_t res2 = 0;
699  		struct qb_ipc_connection_response response;
700  	
701  		c = qb_ipcs_connection_alloc(s);
702  		if (c == NULL) {
703  			qb_ipcc_us_sock_close(sock);
704  			return -ENOMEM;
705  		}
706  	
707  		c->receive_buf = calloc(1, req->max_msg_size);
708  		if (c->receive_buf == NULL) {
709  			free(c);
710  			qb_ipcc_us_sock_close(sock);
711  			return -ENOMEM;
712  		}
713  		c->setup.u.us.sock = sock;
714  		c->request.max_msg_size = req->max_msg_size;
715  		c->response.max_msg_size = req->max_msg_size;
716  		c->event.max_msg_size = req->max_msg_size;
717  		c->pid = ugp->pid;
718  		c->auth.uid = c->euid = ugp->uid;
719  		c->auth.gid = c->egid = ugp->gid;
720  		c->auth.mode = 0600;
721  		c->stats.client_pid = ugp->pid;
722  		snprintf(c->description, CONNECTION_DESCRIPTION,
723  			 "%d-%d-%d", s->pid, ugp->pid,
724  			 c->setup.u.us.sock);
725  	
726  		if (auth_result == 0 && c->service->serv_fns.connection_accept) {
727  			res = c->service->serv_fns.connection_accept(c,
728  								     c->euid, c->egid);
729  		}
730  		if (res != 0) {
731  			goto send_response;
732  		}
733  	
734  		qb_util_log(LOG_DEBUG, "IPC credentials authenticated (%s)",
735  			    c->description);
736  	
737  		memset(&response, 0, sizeof(response));
738  		if (s->funcs.connect) {
739  			res = s->funcs.connect(s, c, &response);
740  			if (res != 0) {
741  				goto send_response;
742  			}
743  		}
744  		/*
745  		 * The connection is good, add it to the active connection list
746  		 */
747  		c->state = QB_IPCS_CONNECTION_ACTIVE;
748  		qb_list_add(&c->list, &s->connections);
749  	
750  		if (s->needs_sock_for_poll) {
751  			qb_ipcs_connection_ref(c);
752  			res = s->poll_fns.dispatch_add(s->poll_priority,
753  						       c->setup.u.us.sock,
754  						       POLLIN | POLLPRI | POLLNVAL,
755  						       c,
756  						       qb_ipcs_dispatch_connection_request);
757  			if (res < 0) {
758  				qb_util_log(LOG_ERR,
759  					    "Error adding socket to mainloop (%s).",
760  					    c->description);
761  			}
762  		}
763  		if (s->type == QB_IPC_SOCKET) {
764  			c->request.u.us.sock = c->setup.u.us.sock;
765  			c->response.u.us.sock = c->setup.u.us.sock;
766  		}
767  	
768  	send_response:
769  		response.hdr.id = QB_IPC_MSG_AUTHENTICATE;
770  		response.hdr.size = sizeof(response);
771  		response.hdr.error = res;
772  		if (res == 0) {
773  			response.connection = (intptr_t) c;
774  			response.connection_type = s->type;
775  			response.max_msg_size = c->request.max_msg_size;
776  			s->stats.active_connections++;
777  		}
778  	
779  		res2 = qb_ipc_us_send(&c->setup, &response, response.hdr.size);
780  		if (res == 0 && res2 != response.hdr.size) {
781  			res = res2;
782  		}
783  	
784  		if (res == 0) {
785  			if (s->type != QB_IPC_SOCKET) {
786  				qb_ipcs_connection_ref(c);
787  				if (s->serv_fns.connection_created) {
788  					s->serv_fns.connection_created(c);
789  				}
790  				if (c->state == QB_IPCS_CONNECTION_ACTIVE) {
791  					c->state = QB_IPCS_CONNECTION_ESTABLISHED;
792  				}
793  				qb_ipcs_connection_unref(c);
794  			}
795  		} else {
796  			if (res == -EACCES) {
797  				qb_util_log(LOG_ERR, "Invalid IPC credentials (%s).",
798  					    c->description);
799  			} else {
800  				errno = -res;
801  				qb_util_perror(LOG_ERR, "Error in connection setup (%s)",
802  					       c->description);
803  			}
804  			qb_ipcs_disconnect(c);
805  		}
806  		return res;
807  	}

809  	static void
810  	handle_connection_new_sock(struct qb_ipcs_service *s, int32_t sock, void *msg)
811  	{
812  		struct qb_ipcs_connection *c = NULL;
813  		struct qb_ipc_event_connection_request *req = msg;
814  	
815  		c = (struct qb_ipcs_connection *)req->connection;
816  		qb_ipcs_connection_ref(c);
817  		c->event.u.us.sock = sock;
818  		if (c->state == QB_IPCS_CONNECTION_ACTIVE) {
819  			c->state = QB_IPCS_CONNECTION_ESTABLISHED;
820  		}
821  		if (s->serv_fns.connection_created) {
822  			s->serv_fns.connection_created(c);
823  		}
824  	
825  		if (c->state == QB_IPCS_CONNECTION_ESTABLISHED &&
826  		    s->type == QB_IPC_SOCKET) {
827  			int32_t res;
828  			qb_ipcs_connection_ref(c);
829  			res = s->poll_fns.dispatch_add(s->poll_priority,
830  						       c->request.u.us.sock,
831  						       POLLIN | POLLPRI | POLLNVAL,
832  						       c,
833  						       qb_ipcs_dispatch_connection_request);
834  			if (res < 0) {
835  				qb_util_log(LOG_ERR,
836  					    "Error adding socket to mainloop (%s).",
837  					    c->description);
838  			}
839  		}
840  	
841  		qb_ipcs_connection_unref(c);
842  	}

844  	static int32_t
845  	qb_ipcs_uc_recv_and_auth(int32_t sock, void *msg, size_t len,
846  				 struct ipc_auth_ugp *ugp)
847  	{
848  		int32_t res = 0;
849  		struct msghdr msg_recv;
850  		struct iovec iov_recv;
851  	
852  	#ifdef SO_PASSCRED
853  		char cmsg_cred[CMSG_SPACE(sizeof(struct ucred))];
854  		int off = 0;
855  		int on = 1;
856  	#endif
857  		msg_recv.msg_iov = &iov_recv;
858  		msg_recv.msg_iovlen = 1;
859  		msg_recv.msg_name = 0;
860  		msg_recv.msg_namelen = 0;
861  	#ifdef SO_PASSCRED
862  		msg_recv.msg_control = (void *)cmsg_cred;
863  		msg_recv.msg_controllen = sizeof(cmsg_cred);
864  	#endif
865  	#ifdef QB_SOLARIS
866  		msg_recv.msg_accrights = 0;
867  		msg_recv.msg_accrightslen = 0;
868  	#else
869  		msg_recv.msg_flags = 0;
870  	#endif /* QB_SOLARIS */
871  	
872  		iov_recv.iov_base = msg;
873  		iov_recv.iov_len = len;
874  	#ifdef SO_PASSCRED
875  		setsockopt(sock, SOL_SOCKET, SO_PASSCRED, &on, sizeof(on));
876  	#endif
877  	
878  		res = qb_ipc_us_recv_msghdr(sock, &msg_recv, msg, len);
879  		if (res < 0) {
880  			goto cleanup_and_return;
881  		}
882  		if (res != len) {
883  			res = -EIO;
884  			goto cleanup_and_return;
885  		}
886  	
887  		/*
888  		 * currently support getpeerucred, getpeereid, and SO_PASSCRED credential
889  		 * retrieval mechanisms for various Platforms
890  		 */
891  	#ifdef HAVE_GETPEERUCRED
892  		/*
893  		 * Solaris and some BSD systems
894  		 */
895  		{
896  			ucred_t *uc = NULL;
897  	
898  			if (getpeerucred(sock, &uc) == 0) {
899  				res = 0;
900  				ugp->uid = ucred_geteuid(uc);
901  				ugp->gid = ucred_getegid(uc);
902  				ugp->pid = ucred_getpid(uc);
903  				ucred_free(uc);
904  			} else {
905  				res = -errno;
906  			}
907  		}
908  	#elif HAVE_GETPEEREID
909  		/*
910  		 * Usually MacOSX systems
911  		 */
912  		{
913  			/*
914  			 * TODO get the peer's pid.
915  			 * c->pid = ?;
916  			 */
917  			if (getpeereid(sock, &ugp->uid, &ugp->gid) == 0) {
918  				res = 0;
919  			} else {
920  				res = -errno;
921  			}
922  		}
923  	
924  	#elif SO_PASSCRED
925  		/*
926  		 * Usually Linux systems
927  		 */
928  		{
929  			struct ucred cred;
930  			struct cmsghdr *cmsg;
931  	
932  			res = -EINVAL;
933  			for (cmsg = CMSG_FIRSTHDR(&msg_recv); cmsg != NULL; cmsg = CMSG_NXTHDR(&msg_recv, cmsg)) {
934  				if (cmsg->cmsg_type != SCM_CREDENTIALS)
935  					continue;
936  	
937  				memcpy(&cred, CMSG_DATA(cmsg), sizeof(struct ucred));
938  				res = 0;
939  				ugp->pid = cred.pid;
940  				ugp->uid = cred.uid;
941  				ugp->gid = cred.gid;
942  				break;
943  			}
944  		}
945  	#else /* no credentials */
946  		ugp->pid = 0;
947  		ugp->uid = 0;
948  		ugp->gid = 0;
949  		res = -ENOTSUP;
950  	#endif /* no credentials */
951  	
952  	cleanup_and_return:
953  	
954  	#ifdef SO_PASSCRED
955  		setsockopt(sock, SOL_SOCKET, SO_PASSCRED, &off, sizeof(off));
956  	#endif
957  	
958  		return res;
959  	}

961  	static int32_t
962  	qb_ipcs_us_connection_acceptor(int fd, int revent, void *data)
963  	{
964  		struct sockaddr_un un_addr;
965  		int32_t new_fd;
966  		struct qb_ipcs_service *s = (struct qb_ipcs_service *)data;
967  		int32_t res;
968  		struct qb_ipc_connection_request setup_msg;
969  		struct ipc_auth_ugp ugp;
970  		socklen_t addrlen = sizeof(struct sockaddr_un);
971  	
972  		if (revent & (POLLNVAL|POLLHUP|POLLERR)) {
973  			/*
974  			 * handle shutdown more cleanly.
975  			 */
976  			return -1;
977  		}
978  	
979  	retry_accept:
980  		errno = 0;
981  		new_fd = accept(fd, (struct sockaddr *)&un_addr, &addrlen);
982  		if (new_fd == -1 && errno == EINTR) {
983  			goto retry_accept;
984  		}
985  	
986  		if (new_fd == -1 && errno == EBADF) {
(1) Event tainted_string_warning: Using a non string-literal such as "descriptor.format" as a format-string argument can lead to a vulnerability if a user can control its content. [details]
Also see events: [tainted_string_warning]
987  			qb_util_perror(LOG_ERR,
988  				       "Could not accept client connection from fd:%d",
989  				       fd);
990  			return -1;
991  		}
992  		if (new_fd == -1) {
(2) Event tainted_string_warning: Using a non string-literal such as "descriptor.format" as a format-string argument can lead to a vulnerability if a user can control its content. [details]
Also see events: [tainted_string_warning]
993  			qb_util_perror(LOG_ERR, "Could not accept client connection");
994  			/* This is an error, but -1 would indicate disconnect
995  			 * from the poll loop
996  			 */
997  			return 0;
998  		}
999  	
1000 		res = qb_sys_fd_nonblock_cloexec_set(new_fd);
1001 		if (res < 0) {
1002 			close(new_fd);
1003 			/* This is an error, but -1 would indicate disconnect
1004 			 * from the poll loop
1005 			 */
1006 			return 0;
1007 		}
1008 	
1009 		res = qb_ipcs_uc_recv_and_auth(new_fd, &setup_msg, sizeof(setup_msg),
1010 					       &ugp);
1011 		if (res < 0) {
1012 			close(new_fd);
1013 			/* This is an error, but -1 would indicate disconnect
1014 			 * from the poll loop
1015 			 */
1016 			return 0;
1017 		}
1018 	
1019 		if (setup_msg.hdr.id == QB_IPC_MSG_AUTHENTICATE) {
1020 			(void)handle_new_connection(s, res, new_fd, &setup_msg,
1021 						    sizeof(setup_msg), &ugp);
1022 		} else if (setup_msg.hdr.id == QB_IPC_MSG_NEW_EVENT_SOCK) {
1023 			if (res == 0) {
1024 				handle_connection_new_sock(s, new_fd, &setup_msg);
1025 			} else {
1026 				close(new_fd);
1027 			}
1028 		} else {
1029 			close(new_fd);
1030 		}
1031 	
1032 		return 0;
1033 	}

1035 	static int32_t
1036 	qb_ipcs_us_connect(struct qb_ipcs_service *s,
1037 			   struct qb_ipcs_connection *c,
1038 			   struct qb_ipc_connection_response *r)
1039 	{
1040 		char path[PATH_MAX];
1041 		int32_t fd_hdr;
1042 		int32_t res = 0;
1043 		struct ipc_us_control *ctl;
1044 		char * shm_ptr;
1045 	
1046 		qb_util_log(LOG_DEBUG, "connecting to client (%s)",
1047 			    c->description);
1048 	
1049 		snprintf(r->request, NAME_MAX, "qb-%s-control-%s",
1050 			 s->name, c->description);
1051 	
1052 		fd_hdr = qb_sys_mmap_file_open(path, r->request,
1053 					       SHM_CONTROL_SIZE,
1054 					       O_CREAT | O_TRUNC | O_RDWR);
1055 		if (fd_hdr < 0) {
1056 			res = fd_hdr;
1057 			errno = -fd_hdr;
1058 			qb_util_perror(LOG_ERR, "couldn't create file for mmap (%s)",
1059 				       c->description);
1060 			return res;
1061 		}
1062 		(void)strlcpy(r->request, path, PATH_MAX);
1063 		(void)strlcpy(c->request.u.us.shared_file_name, r->request, NAME_MAX);
1064 		res = chown(r->request, c->auth.uid, c->auth.gid);
1065 		if (res != 0) {
1066 			/* ignore res, this is just for the compiler warnings.
1067 			 */
1068 			res = 0;
1069 		}
1070 		res = chmod(r->request, c->auth.mode);
1071 		if (res != 0) {
1072 			/* ignore res, this is just for the compiler warnings.
1073 			*/
1074 			res = 0;
1075 		}
1076 	
1077 		shm_ptr = mmap(0, SHM_CONTROL_SIZE,
1078 			       PROT_READ | PROT_WRITE, MAP_SHARED, fd_hdr, 0);
1079 	
1080 		if (shm_ptr == MAP_FAILED) {
1081 			res = -errno;
1082 			qb_util_perror(LOG_ERR, "couldn't create mmap for header (%s)",
1083 				       c->description);
1084 			goto cleanup_hdr;
1085 		}
1086 		c->request.u.us.shared_data = shm_ptr;
1087 		c->response.u.us.shared_data = shm_ptr + sizeof(struct ipc_us_control);
1088 		c->event.u.us.shared_data =  shm_ptr + (2 * sizeof(struct ipc_us_control));
1089 	
1090 		ctl = (struct ipc_us_control *)c->request.u.us.shared_data;
1091 		ctl->sent = 0;
1092 		ctl->flow_control = 0;
1093 		ctl = (struct ipc_us_control *)c->response.u.us.shared_data;
1094 		ctl->sent = 0;
1095 		ctl->flow_control = 0;
1096 		ctl = (struct ipc_us_control *)c->event.u.us.shared_data;
1097 		ctl->sent = 0;
1098 		ctl->flow_control = 0;
1099 	
1100 		close(fd_hdr);
1101 		return res;
1102 	
1103 	cleanup_hdr:
1104 		close(fd_hdr);
1105 		unlink(r->request);
1106 		munmap(c->request.u.us.shared_data, SHM_CONTROL_SIZE);
1107 		return res;
1108 	}
