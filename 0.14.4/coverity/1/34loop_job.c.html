29   	struct qb_loop_job {
30   		struct qb_loop_item item;
31   		qb_loop_job_dispatch_fn dispatch_fn;
32   	};
33   	
34   	static void
35   	job_dispatch(struct qb_loop_item *item, enum qb_loop_priority p)
36   	{
37   		struct qb_loop_job *job = qb_list_entry(item, struct qb_loop_job, item);
38   	
39   		job->dispatch_fn(job->item.user_data);
40   		free(job);
41   	
42   		/*
43   		 * this is a one-shot so don't re-add
44   		 */
45   	}
46   	
47   	static int32_t
48   	get_more_jobs(struct qb_loop_source *s, int32_t ms_timeout)
49   	{
50   		int32_t p;
51   		int32_t new_jobs = 0;
52   		int32_t level_jobs = 0;
53   	
54   		/*
55   		 * this is simple, move jobs from wait_head to job_head
56   		 */
57   		for (p = QB_LOOP_LOW; p <= QB_LOOP_HIGH; p++) {
58   			if (!qb_list_empty(&s->l->level[p].wait_head)) {
59   				level_jobs = qb_list_length(&s->l->level[p].wait_head);
60   				new_jobs += level_jobs;
61   				qb_list_splice_tail(&s->l->level[p].wait_head,
62   					            &s->l->level[p].job_head);
63   				qb_list_init(&s->l->level[p].wait_head);
64   				s->l->level[p].todo += level_jobs;
65   			}
66   		}
67   		return new_jobs;
68   	}
69   	
70   	struct qb_loop_source *
71   	qb_loop_jobs_create(struct qb_loop *l)
72   	{
73   		struct qb_loop_source *s = malloc(sizeof(struct qb_loop_source));
74   		if (s == NULL) {
75   			return NULL;
76   		}
77   		s->l = l;
78   		s->dispatch_and_take_back = job_dispatch;
79   		s->poll = get_more_jobs;
80   	
81   		return s;
82   	}
83   	
84   	void
85   	qb_loop_jobs_destroy(struct qb_loop *l)
86   	{
87   		free(l->job_source);
88   	}

123  	int32_t
124  	qb_loop_job_del(struct qb_loop *lp,
125  			enum qb_loop_priority p,
126  			void *data, qb_loop_job_dispatch_fn dispatch_fn)
127  	{
128  		struct qb_loop_job *job;
129  		struct qb_loop_item *item;
130  		struct qb_loop *l = lp;
131  	
132  		if (l == NULL) {
133  			l = qb_loop_default_get();
134  		}
135  		if (l == NULL || dispatch_fn == NULL) {
136  			return -EINVAL;
137  		}
138  		if (p > QB_LOOP_HIGH) {
139  			return -EINVAL;
140  		}
141  	
142  		qb_list_for_each_entry(item, &l->level[p].wait_head, list) {
143  			job = (struct qb_loop_job *)item;
144  			if (job->dispatch_fn == dispatch_fn &&
145  			    job->item.user_data == data &&
146  			    job->item.type == QB_LOOP_JOB) {
147  				qb_list_del(&job->item.list);
148  				free(job);
149  				return 0;
150  			}
151  		}
152  	
153  		qb_list_for_each_entry(item, &l->level[p].job_head, list) {
154  	
155  			if (item->type != QB_LOOP_JOB) {
156  				continue;
157  			}
158  			job = (struct qb_loop_job *)item;
159  			if (job->dispatch_fn == dispatch_fn &&
160  			    job->item.user_data == data) {
161  				qb_loop_level_item_del(&l->level[p], item);
162  				qb_util_log(LOG_DEBUG, "deleting job in JOBLIST");
163  				return 0;
164  			}
165  		}
166  	
167  		return -ENOENT;
168  	}
