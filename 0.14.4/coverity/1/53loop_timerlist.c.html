32   	struct qb_loop_timer {
33   		struct qb_loop_item item;
34   		qb_loop_timer_dispatch_fn dispatch_fn;
35   		enum qb_loop_priority p;
36   		timer_handle timerlist_handle;
37   		enum qb_poll_entry_state state;
38   		uint32_t check;
39   		uint32_t install_pos;
40   	};
41   	
42   	struct qb_timer_source {
43   		struct qb_loop_source s;
44   		struct timerlist timerlist;
45   		qb_array_t *timers;
46   		size_t timer_entry_count;
47   	};
48   	
49   	static void
50   	timer_dispatch(struct qb_loop_item *item, enum qb_loop_priority p)
51   	{
52   		struct qb_loop_timer *timer = (struct qb_loop_timer *)item;
53   	
54   		assert(timer->state == QB_POLL_ENTRY_JOBLIST);
55   		timer->check = 0;
56   		timer->dispatch_fn(timer->item.user_data);
57   		timer->state = QB_POLL_ENTRY_EMPTY;
58   	}
59   	
60   	static int32_t expired_timers;
61   	static void
62   	make_job_from_tmo(void *data)
63   	{
64   		struct qb_loop_timer *t = (struct qb_loop_timer *)data;
65   		struct qb_loop *l = t->item.source->l;
66   	
67   		assert(t->state == QB_POLL_ENTRY_ACTIVE);
68   		qb_loop_level_item_add(&l->level[t->p], &t->item);
69   		t->state = QB_POLL_ENTRY_JOBLIST;
70   		expired_timers++;
71   	}
72   	
73   	static int32_t
74   	expire_the_timers(struct qb_loop_source *s, int32_t ms_timeout)
75   	{
76   		struct qb_timer_source *ts = (struct qb_timer_source *)s;
77   		expired_timers = 0;
78   		timerlist_expire(&ts->timerlist);
79   		return expired_timers;
80   	}
81   	
82   	int32_t
83   	qb_loop_timer_msec_duration_to_expire(struct qb_loop_source * timer_source)
84   	{
85   		struct qb_timer_source *my_src = (struct qb_timer_source *)timer_source;
86   		uint64_t left = timerlist_msec_duration_to_expire(&my_src->timerlist);
87   		if (left != -1 && left > 0xFFFFFFFF) {
88   			left = 0xFFFFFFFE;
89   		}
90   		return left;
91   	}
92   	
93   	struct qb_loop_source *
94   	qb_loop_timer_create(struct qb_loop *l)
95   	{
96   		struct qb_timer_source *my_src = malloc(sizeof(struct qb_timer_source));
97   		if (my_src == NULL) {
98   			return NULL;
99   		}
100  		my_src->s.l = l;
101  		my_src->s.dispatch_and_take_back = timer_dispatch;
102  		my_src->s.poll = expire_the_timers;
103  	
104  		timerlist_init(&my_src->timerlist);
105  		my_src->timers = qb_array_create_2(16, sizeof(struct qb_loop_timer), 16);
106  		my_src->timer_entry_count = 0;
107  	
108  		return (struct qb_loop_source *)my_src;
109  	}

111  	void
112  	qb_loop_timer_destroy(struct qb_loop *l)
113  	{
114  		struct qb_timer_source *my_src =
115  		    (struct qb_timer_source *)l->timer_source;
116  		qb_array_free(my_src->timers);
117  		free(l->timer_source);
118  	}

120  	static int32_t
121  	_timer_from_handle_(struct qb_timer_source *s,
122  			    qb_loop_timer_handle handle_in,
123  			    struct qb_loop_timer **timer_pt)
124  	{
125  		int32_t rc;
126  		uint32_t check;
127  		uint32_t install_pos;
128  		struct qb_loop_timer *timer;
129  	
130  		if (handle_in == 0) {
131  			return -EINVAL;
132  		}
133  	
134  		check = ((uint32_t) (((uint64_t) handle_in) >> 32));
135  		install_pos = handle_in & 0xffffffff;
136  	
137  		rc = qb_array_index(s->timers, install_pos, (void **)&timer);
138  		if (rc != 0) {
139  			return rc;
140  		}
141  		if (timer->check != check) {
142  			return -EINVAL;
143  		}
144  		*timer_pt = timer;
145  		return 0;
146  	}

148  	static int32_t
149  	_get_empty_array_position_(struct qb_timer_source *s)
150  	{
151  		int32_t install_pos;
152  		int32_t res = 0;
153  		struct qb_loop_timer *timer;
154  	
155  		for (install_pos = 0; install_pos < s->timer_entry_count; install_pos++) {
156  			assert(qb_array_index(s->timers, install_pos, (void **)&timer)
157  			       == 0);
158  			if (timer->state == QB_POLL_ENTRY_EMPTY) {
159  				return install_pos;
160  			}
161  		}
162  	
163  		res = qb_array_grow(s->timers, s->timer_entry_count + 1);
164  		if (res != 0) {
165  			return res;
166  		}
167  	
168  		s->timer_entry_count++;
169  		install_pos = s->timer_entry_count - 1;
170  		return install_pos;
171  	}

173  	int32_t
174  	qb_loop_timer_add(struct qb_loop * lp,
175  			  enum qb_loop_priority p,
176  			  uint64_t nsec_duration,
177  			  void *data,
178  			  qb_loop_timer_dispatch_fn timer_fn,
179  			  qb_loop_timer_handle * timer_handle_out)
180  	{
181  		struct qb_loop_timer *t;
182  		struct qb_timer_source *my_src;
183  		int32_t i;
184  		struct qb_loop *l = lp;
185  	
186  		if (l == NULL) {
187  			l = qb_loop_default_get();
188  		}
189  	
190  		if (l == NULL || timer_fn == NULL) {
191  			return -EINVAL;
192  		}
193  		my_src = (struct qb_timer_source *)l->timer_source;
194  	
195  		i = _get_empty_array_position_(my_src);
196  		assert(qb_array_index(my_src->timers, i, (void **)&t) >= 0);
197  		t->state = QB_POLL_ENTRY_ACTIVE;
198  		t->install_pos = i;
199  		t->item.user_data = data;
200  		t->item.source = (struct qb_loop_source *)my_src;
201  		t->dispatch_fn = timer_fn;
202  		t->p = p;
203  		qb_list_init(&t->item.list);
204  	
205  		for (i = 0; i < 200; i++) {
(1) Event secure_coding: [VERY RISKY]. Using "random" can cause a insecure series of predictable numbers when done incorrectly. random() should not be used for security related applications. Use /dev/random or /dev/urandom instead. 
206  			t->check = random();
207  	
208  			if (t->check != 0 && t->check != 0xffffffff) {
209  				break;
210  			}
211  		}
212  	
213  		if (timer_handle_out) {
214  			*timer_handle_out = (((uint64_t) (t->check)) << 32) | t->install_pos;
215  		}
216  		return timerlist_add_duration(&my_src->timerlist,
217  					      make_job_from_tmo, t,
218  					      nsec_duration, &t->timerlist_handle);
219  	}

221  	int32_t
222  	qb_loop_timer_del(struct qb_loop * lp, qb_loop_timer_handle th)
223  	{
224  		struct qb_timer_source *s;
225  		struct qb_loop_timer *t;
226  		int32_t res;
227  		struct qb_loop *l = lp;
228  	
229  		if (l == NULL) {
230  			l = qb_loop_default_get();
231  		}
232  		s = (struct qb_timer_source *)l->timer_source;
233  	
234  		res = _timer_from_handle_(s, th, &t);
235  		if (res != 0) {
236  			return res;
237  		}
238  	
239  		if (t->state == QB_POLL_ENTRY_DELETED) {
240  			qb_util_log(LOG_WARNING, "timer already deleted");
241  			return 0;
242  		}
243  		if (t->state != QB_POLL_ENTRY_ACTIVE &&
244  		    t->state != QB_POLL_ENTRY_JOBLIST) {
245  			return -EINVAL;
246  		}
247  		if (t->state == QB_POLL_ENTRY_JOBLIST) {
248  			qb_loop_level_item_del(&l->level[t->p], &t->item);
249  		}
250  	
251  		if (t->timerlist_handle) {
252  			timerlist_del(&s->timerlist, t->timerlist_handle);
253  		}
254  		t->state = QB_POLL_ENTRY_EMPTY;
255  		return 0;
256  	}

258  	uint64_t
259  	qb_loop_timer_expire_time_get(struct qb_loop * lp, qb_loop_timer_handle th)
260  	{
261  		struct qb_timer_source *s;
262  		struct qb_loop_timer *t;
263  		int32_t res;
264  		struct qb_loop *l = lp;
265  	
266  		if (l == NULL) {
267  			l = qb_loop_default_get();
268  		}
269  		s = (struct qb_timer_source *)l->timer_source;
270  	
271  		res = _timer_from_handle_(s, th, &t);
272  		if (res != 0) {
273  			return 0;
274  		}
275  	
276  		if (t->state != QB_POLL_ENTRY_ACTIVE) {
277  			return 0;
278  		}
279  	
280  		return timerlist_expire_time(&s->timerlist, t->timerlist_handle);
281  	}
