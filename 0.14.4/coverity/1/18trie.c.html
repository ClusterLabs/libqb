63   	 * characters are stored in reverse to make accessing the
64   	 * more common case (non-control chars) more space efficient.
65   	 */
66   	#define TRIE_CHAR2INDEX(ch) (126 - ch)
67   	#define TRIE_INDEX2CHAR(idx) (126 - idx)

70   	static int32_t
71   	trie_node_alive(struct trie_node *node)
72   	{
73   		if (node->value == NULL ||
74   		    node->refcount <= 0) {
75   			return QB_FALSE;
76   		}
77   		return QB_TRUE;
78   	}

80   	static struct trie_node *
81   	trie_node_next(struct trie_node *node, struct trie_node *root, int all)
82   	{
83   		struct trie_node *c = node;
84   		struct trie_node *n;
85   		struct trie_node *p;
86   		int i;
87   	
88   	keep_going:
89   		n = NULL;
90   	
91   		/* child/outward
92   		 */
93   		for (i = c->num_children - 1; i >= 0; i--) {
94   			if (c->children[i]) {
95   				n = c->children[i];
96   				break;
97   			}
98   		}
99   		if (n) {
100  			if (all || trie_node_alive(n)) {
101  				return n;
102  			} else {
103  				c = n;
104  				goto keep_going;
105  			}
106  		}
107  		/* sibling/parent
108  		 */
109  		if (c == root) {
110  			return NULL;
111  		}
112  		p = c;
113  		do {
114  			for (i = p->idx - 1; i >= 0; i--) {
115  				if (p->parent->children[i]) {
116  					n = p->parent->children[i];
117  					break;
118  				}
119  			}
120  			if (n == NULL) {
121  				p = p->parent;
122  			}
123  		} while (n == NULL && p != root);
124  	
125  		if (n) {
126  			if (all || trie_node_alive(n)) {
127  				return n;
128  			}
129  			if (n == root) {
130  				return NULL;
131  			}
132  			c = n;
133  			goto keep_going;
134  		}
135  	
136  		return n;
137  	}

170  	static struct trie_node *
171  	trie_node_split(struct trie *t, struct trie_node *cur_node, int seg_cnt)
172  	{
173  		struct trie_node *split_node;
174  		struct trie_node ** children = cur_node->children;
175  		uint32_t num_children = cur_node->num_children;
176  		int i;
177  		int s;
178  	
179  		cur_node->children = NULL;
180  		cur_node->num_children = 0;
181  		split_node = new_child_node(t, cur_node, cur_node->segment[seg_cnt]);
182  		if (split_node == NULL) {
183  			return NULL;
184  		}
185  		split_node->children = children;
186  		split_node->num_children = num_children;
187  		for (i = 0; i < split_node->num_children; i++) {
188  			if (split_node->children[i]) {
189  				split_node->children[i]->parent = split_node;
190  			}
191  		}
192  		split_node->value = cur_node->value;
193  		split_node->key = cur_node->key;
194  		split_node->refcount = cur_node->refcount;
195  		cur_node->value = NULL;
196  		cur_node->key = NULL;
197  		cur_node->refcount = 0;
198  		/* move notifier list to split */
199  		qb_list_replace(&cur_node->notifier_head, &split_node->notifier_head);
200  		qb_list_init(&cur_node->notifier_head);
201  	
202  		if (seg_cnt < cur_node->num_segments) {
203  			split_node->num_segments = cur_node->num_segments - seg_cnt - 1;
204  			split_node->segment = malloc(split_node->num_segments * sizeof(char));
205  			if (split_node->segment == NULL) {
206  				free(split_node);
207  				return NULL;
208  			}
209  			for (i = (seg_cnt + 1); i < cur_node->num_segments; i++) {
210  				s = i - seg_cnt - 1;
211  				split_node->segment[s] = cur_node->segment[i];
212  				cur_node->segment[i] = '\0';
213  			}
214  			cur_node->num_segments = seg_cnt;
215  		}
216  		return cur_node;
217  	}

219  	static struct trie_node *
220  	trie_insert(struct trie *t, const char *key)
221  	{
222  		struct trie_node *cur_node = t->header;
223  		struct trie_node *new_node;
224  		char *cur = (char *)key;
225  		int idx = TRIE_CHAR2INDEX(key[0]);
226  		int seg_cnt = 0;
227  	
228  		do {
229  			new_node = NULL;
230  			if (cur_node->num_segments > 0 &&
231  			    seg_cnt < cur_node->num_segments) {
232  				if (cur_node->segment[seg_cnt] == *cur) {
233  					/* we found the char in the segment */
234  					seg_cnt++;
235  				} else {
236  					cur_node = trie_node_split(t, cur_node, seg_cnt);
237  					if (cur_node == NULL) {
238  						return NULL;
239  					}
240  					new_node = new_child_node(t, cur_node, *cur);
241  					if (new_node == NULL) {
242  						return NULL;
243  					}
244  				}
245  			} else if (idx < cur_node->num_children &&
246  			    cur_node->children[idx]) {
247  				/* the char can be found on the next node */
248  				new_node = cur_node->children[idx];
249  			} else if (cur_node == t->header) {
250  				/* the root node is empty so make it on the next node */
251  				new_node = new_child_node(t, cur_node, *cur);
252  				if (new_node == NULL) {
253  					return NULL;
254  				}
255  			} else if (cur_node->value == NULL &&
256  				   qb_list_empty(&cur_node->notifier_head) &&
257  				   cur_node->num_children == 0 &&
258  				   seg_cnt == cur_node->num_segments) {
259  				/* we are on a leaf (with no value) so just add it as a segment */
260  				cur_node->segment = realloc(cur_node->segment, cur_node->num_segments + 1);
261  				cur_node->segment[cur_node->num_segments] = *cur;
262  				t->mem_used += sizeof(char);
263  				cur_node->num_segments++;
264  				seg_cnt++;
265  			} else if (seg_cnt == cur_node->num_segments) {
266  				/* on the last segment need to make a new node */
267  				new_node = new_child_node(t, cur_node, *cur);
268  				if (new_node == NULL) {
269  					return NULL;
270  				}
271  			} else /* need_to_split */ {
272  				cur_node = trie_node_split(t, cur_node, seg_cnt);
273  				if (cur_node == NULL) {
274  					return NULL;
275  				}
276  				new_node = new_child_node(t, cur_node, *cur);
277  				if (new_node == NULL) {
278  					return NULL;
279  				}
280  			}
281  			if (new_node) {
282  				seg_cnt = 0;
283  				cur_node = new_node;
284  			}
285  			cur++;
286  			idx = TRIE_CHAR2INDEX(*cur);
287  		} while (*cur != '\0');
288  	
289  		if (cur_node->num_segments > 0 &&
290  		    seg_cnt < cur_node->num_segments) {
291  			/* we need to split */
292  			cur_node = trie_node_split(t, cur_node, seg_cnt);
293  			if (cur_node == NULL) {
294  				return NULL;
295  			}
296  			new_node = new_child_node(t, cur_node, *cur);
297  			if (new_node == NULL) {
298  				return NULL;
299  			}
300  		}
301  	
302  		return cur_node;
303  	}

305  	static struct trie_node *
306  	trie_lookup(struct trie *t, const char *key, int exact_match)
307  	{
308  		struct trie_node *cur_node = t->header;
309  		char *cur = (char *)key;
310  		int idx = TRIE_CHAR2INDEX(key[0]);
311  		int seg_cnt = 0;
312  	
313  		do {
314  			if (cur_node->num_segments > 0 &&
315  			    seg_cnt < cur_node->num_segments) {
316  				if (cur_node->segment[seg_cnt] == *cur) {
317  					/* we found the char in the segment */
318  					seg_cnt++;
319  				} else {
320  					return NULL;
321  				}
322  			} else if (idx < cur_node->num_children &&
323  			    cur_node->children[idx]) {
324  				/* the char can be found on the next node */
325  				cur_node = cur_node->children[idx];
326  				seg_cnt = 0;
327  			} else {
328  				return NULL;
329  			}
330  			cur++;
331  			idx = TRIE_CHAR2INDEX(*cur);
332  		} while (*cur != '\0');
333  	
334  		if (exact_match &&
335  		    cur_node->num_segments > 0 &&
336  		    seg_cnt < cur_node->num_segments) {
337  			return NULL;
338  		}
339  	
340  		return cur_node;
341  	}

343  	static void
344  	trie_node_release(struct trie *t, struct trie_node *node)
345  	{
346  		int i;
347  		int empty = QB_FALSE;
348  	
349  		if (node->key == NULL &&
350  		    node->parent != NULL &&
351  		    qb_list_empty(&node->notifier_head)) {
352  			struct trie_node *p = node->parent;
353  	
354  			if (node->num_children == 0) {
355  				empty = QB_TRUE;
356  			} else {
357  				empty = QB_TRUE;
358  				for (i = node->num_children - 1; i >= 0; i--) {
359  					if (node->children[i]) {
360  						empty = QB_FALSE;
361  						break;
362  					}
363  				}
364  			}
365  			if (!empty) {
366  				return;
367  			}
368  	
369  			/*
370  			 * unlink the node from the parent
371  			 */
372  			p->children[node->idx] = NULL;
373  			free(node->segment);
374  			free(node->children);
375  			free(node);
376  			t->num_nodes--;
377  			t->mem_used -= sizeof(struct trie_node);
378  	
379  			trie_node_release(t, p);
380  		}
381  	}

383  	static void
384  	trie_node_destroy(struct trie *t, struct trie_node *n)
385  	{
386  		if (n->value == NULL) {
387  			return;
388  		}
389  		trie_notify(n, QB_MAP_NOTIFY_DELETED, n->key, n->value, NULL);
390  	
391  		n->key = NULL;
392  		n->value = NULL;
393  	
394  		trie_node_release(t, n);
395  	}

397  	static void
398  	trie_print_node(struct trie_node *n, struct trie_node *r, const char *suffix)
399  	{
400  		int i;
401  	
402  		if (n->parent) {
403  			trie_print_node(n->parent, n, suffix);
404  		}
405  		if (n->idx == 0) {
406  			return;
407  		}
408  	
409  		printf("[%c", TRIE_INDEX2CHAR(n->idx));
410  		for (i = 0; i < n->num_segments; i++) {
411  			printf("%c", n->segment[i]);
412  		}
413  		if (n == r) {
414  			printf("] (%d) %s\n", n->refcount, suffix);
415  		} else {
416  			printf("] ");
417  		}
418  	}

420  	static void
421  	trie_node_ref(struct trie *t, struct trie_node *node)
422  	{
423  		if (t->header == node) {
424  			return;
425  		}
426  		node->refcount++;
427  	}

429  	static void
430  	trie_node_deref(struct trie *t, struct trie_node *node)
431  	{
432  		if (!trie_node_alive(node)) {
433  			return;
434  		}
435  		node->refcount--;
436  		if (node->refcount > 0) {
437  			return;
438  		}
439  		trie_node_destroy(t, node);
440  	}

442  	static void
443  	trie_destroy(struct qb_map *map)
444  	{
445  		struct trie *t = (struct trie *)map;
446  	
447  		struct trie_node *cur_node = t->header;
448  		struct trie_node *fwd_node;
449  	
450  		do {
451  			fwd_node = trie_node_next(cur_node, t->header, QB_FALSE);
452  			trie_node_destroy(t, cur_node);
453  		} while ((cur_node = fwd_node));
454  	
455  		free(t);
456  	}

458  	static struct trie_node *
459  	trie_new_node(struct trie *t, struct trie_node *parent)
460  	{
461  		struct trie_node *new_node = calloc(1, sizeof(struct trie_node));
462  	
463  		if (new_node == NULL) {
464  			return NULL;
465  		}
466  	
467  		new_node->parent = parent;
468  		new_node->num_children = 0;
469  		new_node->children = NULL;
470  		new_node->num_segments = 0;
471  		new_node->segment = NULL;
472  		t->num_nodes++;
473  		t->mem_used += sizeof(struct trie_node);
474  		qb_list_init(&new_node->notifier_head);
475  		return new_node;
476  	}

478  	void
479  	qb_trie_dump(qb_map_t* m)
480  	{
481  		struct trie * t = (struct trie*)m;
482  		struct trie_node *n;
483  	
484  		if (t == NULL) {
485  			return;
486  		}
487  	
488  		printf("nodes: %d, bytes: %d\n", t->num_nodes, t->mem_used);
489  	
490  		n = t->header;
491  		do {
492  			if (n->num_children == 0) {
493  				trie_print_node(n, n, " ");
494  			}
495  			n = trie_node_next(n, t->header, QB_FALSE);
496  		} while (n);
497  	}

499  	static void
500  	trie_put(struct qb_map *map, const char *key, const void *value)
501  	{
502  		struct trie *t = (struct trie *)map;
503  		struct trie_node *n = trie_insert(t, key);
504  		if (n) {
505  			const char *old_value = n->value;
506  			const char *old_key = n->key;
507  	
508  			n->key = (char *)key;
509  			n->value = (void *)value;
510  	
511  			if (old_value == NULL) {
512  				trie_node_ref(t, n);
513  				t->length++;
514  				trie_notify(n, QB_MAP_NOTIFY_INSERTED,
515  					    n->key, NULL, n->value);
516  			} else {
517  				trie_notify(n, QB_MAP_NOTIFY_REPLACED,
518  					    (char *)old_key, (void *)old_value,
519  					    (void *)value);
520  			}
521  		}
522  	}

524  	static int32_t
525  	trie_rm(struct qb_map *map, const char *key)
526  	{
527  		struct trie *t = (struct trie *)map;
528  		struct trie_node *n = trie_lookup(t, key, QB_TRUE);
529  		if (n) {
530  			trie_node_deref(t, n);
531  			t->length--;
532  			return QB_TRUE;
533  		} else {
534  			return QB_FALSE;
535  		}
536  	}

566  	static void
567  	trie_notify(struct trie_node *n,
568  		    uint32_t event, const char *key, void *old_value, void *value)
569  	{
570  		struct trie_node *c = n;
571  		struct qb_list_head *list;
572  		struct qb_list_head *next;
573  		struct qb_map_notifier *tn;
574  	
575  		do {
576  			qb_list_for_each_safe(list, next, &c->notifier_head) {
577  				tn = qb_list_entry(list, struct qb_map_notifier, list);
578  				trie_notify_ref(tn);
579  	
580  				if ((tn->events & event) &&
581  				    ((tn->events & QB_MAP_NOTIFY_RECURSIVE) ||
582  				     (n == c))) {
583  					tn->callback(event, (char *)key, old_value,
584  						     value, tn->user_data);
585  				}
586  				if (((event & QB_MAP_NOTIFY_DELETED) ||
587  				     (event & QB_MAP_NOTIFY_REPLACED)) &&
588  				    (tn->events & QB_MAP_NOTIFY_FREE)) {
589  					tn->callback(QB_MAP_NOTIFY_FREE, (char *)key,
590  						     old_value, value, tn->user_data);
591  				}
592  				trie_notify_deref(tn);
593  			}
594  			c = c->parent;
595  		} while (c);
596  	}

598  	static int32_t
599  	trie_notify_add(qb_map_t * m, const char *key,
600  			qb_map_notify_fn fn, int32_t events, void *user_data)
601  	{
602  		struct trie *t = (struct trie *)m;
603  		struct qb_map_notifier *f;
604  		struct trie_node *n;
605  		struct qb_list_head *list;
606  		int add_to_tail = QB_FALSE;
607  	
608  		if (key) {
609  			n = trie_lookup(t, key, QB_TRUE);
610  			if (n == NULL) {
611  				n = trie_insert(t, key);
612  			}
613  		} else {
614  			n = t->header;
615  		}
616  		if (n) {
617  			qb_list_for_each(list, &n->notifier_head) {
618  				f = qb_list_entry(list, struct qb_map_notifier, list);
619  	
620  				if (events & QB_MAP_NOTIFY_FREE &&
621  				    f->events == events) {
622  					/* only one free notifier */
623  					return -EEXIST;
624  				}
625  				if (f->events == events &&
626  				    f->callback == fn &&
627  				    f->user_data == user_data) {
628  					return -EEXIST;
629  				}
630  			}
631  	
632  			f = malloc(sizeof(struct qb_map_notifier));
633  			if (f == NULL) {
634  				return -errno;
635  			}
636  			f->events = events;
637  			f->user_data = user_data;
638  			f->callback = fn;
639  			f->refcount = 1;
640  			qb_list_init(&f->list);
641  			if (key) {
642  				if (events & QB_MAP_NOTIFY_RECURSIVE) {
643  					add_to_tail = QB_TRUE;
644  				}
645  			} else {
646  				if (events & QB_MAP_NOTIFY_FREE) {
647  					add_to_tail = QB_TRUE;
648  				}
649  			}
650  			if (add_to_tail) {
651  				qb_list_add_tail(&f->list, &n->notifier_head);
652  			} else {
653  				qb_list_add(&f->list, &n->notifier_head);
654  			}
655  			return 0;
656  		}
657  		return -EINVAL;
658  	}

660  	static int32_t
661  	trie_notify_del(qb_map_t * m, const char *key,
662  			qb_map_notify_fn fn, int32_t events,
663  			int32_t cmp_userdata, void *user_data)
664  	{
665  		struct trie *t = (struct trie *)m;
666  		struct qb_map_notifier *f;
667  		struct trie_node *n;
668  		struct qb_list_head *list;
669  		struct qb_list_head *next;
670  		int32_t found = QB_FALSE;
671  	
672  		if (key) {
673  			n = trie_lookup(t, key, QB_FALSE);
674  		} else {
675  			n = t->header;
676  		}
677  		if (n == NULL) {
678  			return -ENOENT;
679  		}
680  		qb_list_for_each_safe(list, next, &n->notifier_head) {
681  			f = qb_list_entry(list, struct qb_map_notifier, list);
682  			trie_notify_ref(f);
683  	
684  			if (f->events == events && f->callback == fn) {
685  				if (cmp_userdata && (f->user_data == user_data)) {
686  					trie_notify_deref(f);
687  					found = QB_TRUE;
688  				} else if (!cmp_userdata) {
689  					trie_notify_deref(f);
690  					found = QB_TRUE;
691  				}
692  			}
693  			trie_notify_deref(f);
694  		}
695  		if (found) {
696  			trie_node_release(t, n);
697  			return 0;
698  		} else {
699  			return -ENOENT;
700  		}
701  	}

703  	static qb_map_iter_t *
704  	trie_iter_create(struct qb_map *map, const char *prefix)
705  	{
706  		struct trie_iter *i = malloc(sizeof(struct trie_iter));
707  		struct trie *t = (struct trie *)map;
708  		if (i == NULL) {
709  			return NULL;
710  		}
711  		i->i.m = map;
712  		i->prefix = prefix;
713  		i->n = t->header;
714  		i->root = t->header;
715  		return (qb_map_iter_t *) i;
716  	}
717  	
718  	static const char *
719  	trie_iter_next(qb_map_iter_t * i, void **value)
720  	{
721  		struct trie_iter *si = (struct trie_iter *)i;
722  		struct trie_node *p = si->n;
723  		struct trie *t = (struct trie *)(i->m);
724  	
725  		if (p == NULL) {
726  			return NULL;
727  		}
728  	
729  		if (p->parent == NULL && si->prefix) {
730  			si->root = trie_lookup(t, si->prefix, QB_FALSE);
731  			if (si->root == NULL) {
732  				si->n = NULL;
733  			} else if (si->root->value == NULL) {
734  				si->n = trie_node_next(si->root, si->root, QB_FALSE);
735  			} else {
736  				si->n = si->root;
737  			}
738  		} else {
739  			si->n = trie_node_next(p, si->root, QB_FALSE);
740  		}
741  		if (si->n == NULL) {
742  			trie_node_deref(t, p);
743  			return NULL;
744  		}
745  		trie_node_ref(t, si->n);
746  		trie_node_deref(t, p);
747  		*value = si->n->value;
748  		return si->n->key;
749  	}
750  	
751  	static void
752  	trie_iter_free(qb_map_iter_t * i)
753  	{
754  		struct trie_iter *si = (struct trie_iter *)i;
755  		struct trie *t = (struct trie *)(i->m);
756  	
757  		if (si->n != NULL) {
758  			/* if free'ing the iterator before getting to the last
759  			 * node make sure we de-ref the current node.
760  			 */
761  			trie_node_deref(t, si->n);
762  		}
763  		free(i);
764  	}

766  	static size_t
767  	trie_count_get(struct qb_map *map)
768  	{
769  		struct trie *list = (struct trie *)map;
770  		return list->length;
771  	}

773  	qb_map_t *
774  	qb_trie_create(void)
775  	{
776  		struct trie *t = malloc(sizeof(struct trie));
777  		if (t == NULL) {
778  			return NULL;
779  		}
780  	
781  		t->map.put = trie_put;
782  		t->map.get = trie_get;
783  		t->map.rm = trie_rm;
784  		t->map.count_get = trie_count_get;
785  		t->map.iter_create = trie_iter_create;
786  		t->map.iter_next = trie_iter_next;
787  		t->map.iter_free = trie_iter_free;
788  		t->map.destroy = trie_destroy;
789  		t->map.notify_add = trie_notify_add;
790  		t->map.notify_del = trie_notify_del;
791  		t->length = 0;
792  		t->num_nodes = 0;
793  		t->mem_used = sizeof(struct trie);
794  		t->header = trie_new_node(t, NULL);
795  	
796  		return (qb_map_t *) t;
797  	}
