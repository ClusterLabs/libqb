104  	int
105  	main(int argc, char *argv[])
106  	{
107  		int sock;
108  		int true_opt = 1;
109  		struct sockaddr_in server_addr;
110  		qb_loop_t *ml = qb_loop_create();
111  	
112  		if ((sock = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
113  			perror("Socket");
114  			exit(1);
115  		}
116  	
117  		if (setsockopt(sock,
118  			       SOL_SOCKET,
119  			       SO_REUSEADDR, &true_opt, sizeof(int)) == -1) {
120  			perror("Setsockopt");
121  			exit(1);
122  		}
123  	
124  		server_addr.sin_family = AF_INET;
125  		server_addr.sin_port = htons(5000);
126  		server_addr.sin_addr.s_addr = INADDR_ANY;
127  		bzero(&(server_addr.sin_zero), 8);
128  	
129  		printf("TCPServer binding to port 5000\n");
130  		if (bind(sock,
131  			 (struct sockaddr *)&server_addr,
132  			 sizeof(struct sockaddr)) == -1) {
133  			perror("Unable to bind");
134  			exit(1);
135  		}
136  	
137  		printf("TCPServer Waiting for client on port 5000\n");
138  	
139  		if (listen(sock, 5) == -1) {
140  			perror("Listen");
141  			exit(1);
142  		}
143  	
144  		qb_loop_poll_add(ml, QB_LOOP_MED, sock, POLLIN, ml, sock_accept_fn);
145  	
146  		qb_loop_signal_add(ml, QB_LOOP_HIGH, SIGINT, ml, please_exit_fn, NULL);
147  		qb_loop_run(ml);
148  	
149  		close(sock);
150  		return 0;
151  	}
