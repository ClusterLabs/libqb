40   	struct bm_ctx {
41   		qb_ipcc_connection_t *conn;
42   		qb_util_stopwatch_t *sw;
43   		float mbs;
44   		float secs;
45   		int32_t multi;
46   		uint32_t counter;
47   	};
48   	
49   	static void bm_start(struct bm_ctx *ctx)
50   	{
51   		qb_util_stopwatch_start(ctx->sw);
52   	}
53   	
54   	static void bm_finish(struct bm_ctx *ctx, const char *operation, int32_t size)
55   	{
56   		qb_util_stopwatch_stop(ctx->sw);
57   		ctx->secs = qb_util_stopwatch_sec_elapsed_get(ctx->sw);
58   	
59   		ctx->mbs =
60   		    ((((float)ctx->counter) * size) / ctx->secs) / (1024.0 *
61   								    1024.0);
62   	}
63   	
64   	static void bmc_connect(struct bm_ctx *ctx)
65   	{
66   		ctx->sw = qb_util_stopwatch_create();
67   		ctx->conn = qb_ipcc_connect("bm1", QB_MAX(1000 * (100 + THREADS),
68   							  1024*1024));
69   		if (ctx->conn == NULL) {
70   			perror("qb_ipcc_connect");
71   			exit(-1);
72   		}
73   	}
74   	
75   	static void bmc_disconnect(struct bm_ctx *ctx)
76   	{
77   		qb_ipcc_disconnect(ctx->conn);
78   		qb_util_stopwatch_free(ctx->sw);
79   	}
