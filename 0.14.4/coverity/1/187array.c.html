101  	int32_t
102  	qb_array_index(struct qb_array * a, int32_t idx, void **element_out)
103  	{
104  		int32_t b;
105  		int32_t elem;
106  		char *bin;
107  		int32_t rc = 0;
108  	
109  		if (a == NULL || element_out == NULL) {
110  			return -EINVAL;
111  		}
112  		if (idx < 0) {
113  			return -ERANGE;
114  		}
115  		if (idx >= a->max_elements) {
116  			if (a->autogrow_elements == 0) {
117  				return -ERANGE;
118  			} else {
119  				rc = qb_array_grow(a, idx + 1);
120  				if (rc != 0) {
121  					return rc;
122  				}
123  			}
124  		}
125  		b = BIN_NUM_GET(idx);

129  			int32_t bin_alloced = QB_FALSE;
130  	
131  			(void)qb_thread_lock(a->grow_lock);
132  	
133  			if (b >= a->num_bins) {
134  				rc = _grow_bin_array(a, b + 1);
135  				if (rc < 0) {
136  					goto unlock_error;
137  				}
138  			}
139  			if (a->bin[b] == NULL) {
140  				a->bin[b] = calloc(MAX_ELEMENTS_PER_BIN, a->element_size);
141  				if (a->bin[b] == NULL) {
142  					rc = -errno;

146  			}
147

196  	int32_t
197  	qb_array_grow(struct qb_array * a, size_t max_elements)
198  	{
199  		int32_t b;
200  		int32_t rc = 0;
201  	
202  		if (a == NULL || max_elements > (MAX_ELEMENTS_PER_BIN * MAX_BINS)) {
203  			return -EINVAL;
204  		}
205  		if (max_elements <= a->max_elements) {
206  			return 0;
207  		}
208  		a->max_elements = max_elements;
209  		b = QB_MIN((max_elements / MAX_ELEMENTS_PER_BIN) + 1, MAX_BINS);
210  		if (b > a->num_bins) {
211  			(void)qb_thread_lock(a->grow_lock);
212  			if (b >= a->num_bins) {
213  				rc = _grow_bin_array(a, b + 1);
214  			}
215  			(void)qb_thread_unlock(a->grow_lock);
216  		}
217  		return rc;
218  	}
219  	
220  	void
221  	qb_array_free(struct qb_array *a)
222  	{
223  		int32_t i;
224  		for (i = 0; i < a->num_bins; i++) {
225  			free(a->bin[i]);
226  		}
227  		free(a->bin);
228  		(void)qb_thread_lock_destroy(a->grow_lock);
229  		free(a);
230  	}
