26   	enum QB_HDB_HANDLE_STATE {
27   		QB_HDB_HANDLE_STATE_EMPTY,
28   		QB_HDB_HANDLE_STATE_PENDINGREMOVAL,
29   		QB_HDB_HANDLE_STATE_ACTIVE
30   	};
31   	
32   	static void
33   	qb_hdb_create_first_run(struct qb_hdb *hdb)
34   	{
35   		if (hdb->first_run == QB_TRUE) {
36   			hdb->first_run = QB_FALSE;
37   			qb_atomic_init();
38   			hdb->handles = qb_array_create(32, sizeof(struct qb_hdb_handle));
39   		}
40   	}
41   	
42   	void
43   	qb_hdb_create(struct qb_hdb *hdb)
44   	{
45   		memset(hdb, 0, sizeof(struct qb_hdb));
46   		hdb->first_run = QB_TRUE;
47   		qb_hdb_create_first_run(hdb);
48   	}
49   	
50   	void
51   	qb_hdb_destroy(struct qb_hdb *hdb)
52   	{
53   		qb_array_free(hdb->handles);
54   		memset(hdb, 0, sizeof(struct qb_hdb));
55   	}

57   	int32_t
58   	qb_hdb_handle_create(struct qb_hdb *hdb, int32_t instance_size,
59   			     qb_handle_t * handle_id_out)
60   	{
61   		int32_t handle;
62   		int32_t res = 0;
63   		uint32_t check;
64   		int32_t found = QB_FALSE;
65   		void *instance;
66   		int32_t i;
67   		struct qb_hdb_handle *entry = NULL;
68   		int32_t handle_count;
69   	
70   		qb_hdb_create_first_run(hdb);
71   	
72   		handle_count = qb_atomic_int_get(&hdb->handle_count);
73   		for (handle = 0; handle < handle_count; handle++) {
74   			if (qb_array_index(hdb->handles, handle, (void**)&entry) == 0 &&
75   			    entry->state == QB_HDB_HANDLE_STATE_EMPTY) {
76   				found = QB_TRUE;
77   				qb_atomic_int_inc(&entry->ref_count);
78   				break;
79   			}
80   		}
81   	
82   		if (found == QB_FALSE) {
83   			res = qb_array_grow(hdb->handles, handle_count + 1);
84   			if (res != 0) {
85   				return res;
86   			}
87   			res = qb_array_index(hdb->handles, handle_count,
88   					     (void **)&entry);
89   			if (res != 0) {
90   				return res;
91   			}
92   			qb_atomic_int_inc((int32_t *)&hdb->handle_count);
93   		}
94   	
95   		instance = malloc(instance_size);
96   		if (instance == 0) {
97   			return -ENOMEM;
98   		}
99   	
100  		/*
101  		 * This code makes sure the random number isn't zero
102  		 * We use 0 to specify an invalid handle out of the 1^64 address space
103  		 * If we get 0 200 times in a row, the RNG may be broken
104  		 */
105  		for (i = 0; i < 200; i++) {
(1) Event secure_coding: [VERY RISKY]. Using "random" can cause a insecure series of predictable numbers when done incorrectly. random() should not be used for security related applications. Use /dev/random or /dev/urandom instead. 
106  			check = random();
107  	
108  			if (check != 0 && check != 0xffffffff) {
109  				break;
110  			}
111  		}
112  	
113  		memset(instance, 0, instance_size);
114  	
115  		entry->state = QB_HDB_HANDLE_STATE_ACTIVE;
116  		entry->instance = instance;
117  		entry->ref_count = 1;
118  		entry->check = check;
119  	
120  		*handle_id_out = (((uint64_t) (check)) << 32) | handle;
121  	
122  		return res;
123  	}

125  	int32_t
126  	qb_hdb_handle_get(struct qb_hdb * hdb, qb_handle_t handle_in, void **instance)
127  	{
128  		uint32_t check = ((uint32_t) (((uint64_t) handle_in) >> 32));
129  		uint32_t handle = handle_in & 0xffffffff;
130  		struct qb_hdb_handle *entry;
131  		int32_t handle_count;
132  	
133  		qb_hdb_create_first_run(hdb);
134  	
135  		*instance = NULL;
136  		handle_count = qb_atomic_int_get(&hdb->handle_count);
137  		if (handle >= handle_count) {
138  			return (-EBADF);
139  		}
140  	
141  		if (qb_array_index(hdb->handles, handle, (void **)&entry) != 0 ||
142  		    entry->state != QB_HDB_HANDLE_STATE_ACTIVE) {
143  			return (-EBADF);
144  		}
145  	
146  		if (check != 0xffffffff && check != entry->check) {
147  			return (-EBADF);
148  		}
149  		qb_atomic_int_inc(&entry->ref_count);
150  	
151  		*instance = entry->instance;
152  	
153  		return (0);
154  	}

156  	int32_t
157  	qb_hdb_handle_get_always(struct qb_hdb * hdb, qb_handle_t handle_in,
158  				 void **instance)
159  	{
160  		return qb_hdb_handle_get(hdb, handle_in, instance);
161  	}

163  	int32_t
164  	qb_hdb_handle_put(struct qb_hdb * hdb, qb_handle_t handle_in)
165  	{
166  		uint32_t check = ((uint32_t) (((uint64_t) handle_in) >> 32));
167  		uint32_t handle = handle_in & 0xffffffff;
168  		struct qb_hdb_handle *entry;
169  		int32_t handle_count;
170  	
171  		qb_hdb_create_first_run(hdb);
172  	
173  		handle_count = qb_atomic_int_get(&hdb->handle_count);
174  		if (handle >= handle_count) {
175  			return (-EBADF);
176  		}
177  	
178  		if (qb_array_index(hdb->handles, handle, (void **)&entry) != 0 ||
179  		    (check != 0xffffffff && check != entry->check)) {
180  			return (-EBADF);
181  		}
182  	
183  		if (qb_atomic_int_dec_and_test(&entry->ref_count)) {
184  			if (hdb->destructor) {
185  				hdb->destructor(entry->instance);
186  			}
187  			free(entry->instance);
188  			memset(entry, 0, sizeof(struct qb_hdb_handle));
189  		}
190  		return (0);
191  	}

193  	int32_t
194  	qb_hdb_handle_destroy(struct qb_hdb * hdb, qb_handle_t handle_in)
195  	{
196  		uint32_t check = ((uint32_t) (((uint64_t) handle_in) >> 32));
197  		uint32_t handle = handle_in & 0xffffffff;
198  		int32_t res;
199  		struct qb_hdb_handle *entry;
200  		int32_t handle_count;
201  	
202  		qb_hdb_create_first_run(hdb);
203  	
204  		handle_count = qb_atomic_int_get(&hdb->handle_count);
205  		if (handle >= handle_count) {
206  			return (-EBADF);
207  		}
208  	
209  		if (qb_array_index(hdb->handles, handle, (void **)&entry) != 0 ||
210  		    (check != 0xffffffff && check != entry->check)) {
211  			return (-EBADF);
212  		}
213  	
214  		entry->state = QB_HDB_HANDLE_STATE_PENDINGREMOVAL;
215  		res = qb_hdb_handle_put(hdb, handle_in);
216  		return (res);
217  	}

220  	qb_hdb_handle_refcount_get(struct qb_hdb * hdb, qb_handle_t handle_in)
221  	{
222  		uint32_t check = ((uint32_t) (((uint64_t) handle_in) >> 32));
223  		uint32_t handle = handle_in & 0xffffffff;
224  		struct qb_hdb_handle *entry;
225  		int32_t handle_count;
226  		int32_t refcount = 0;
227  	
228  		qb_hdb_create_first_run(hdb);
229  	
230  		handle_count = qb_atomic_int_get(&hdb->handle_count);
231  		if (handle >= handle_count) {
232  			return (-EBADF);
233  		}
234  	
235  		if (qb_array_index(hdb->handles, handle, (void **)&entry) != 0 ||
236  		    (check != 0xffffffff && check != entry->check)) {
237  			return (-EBADF);
238  		}
239  	
240  		refcount = qb_atomic_int_get(&entry->ref_count);
241  	
242  		return (refcount);
243  	}

251  	int32_t
252  	qb_hdb_iterator_next(struct qb_hdb *hdb, void **instance, qb_handle_t * handle)
253  	{
254  		int32_t res = -1;
255  		uint64_t checker;
256  		struct qb_hdb_handle *entry;
257  		int32_t handle_count;
258  	
259  		handle_count = qb_atomic_int_get(&hdb->handle_count);
260  		while (hdb->iterator < handle_count) {
261  			res = qb_array_index(hdb->handles,
262  					     hdb->iterator,
263  					     (void **)&entry);
264  			if (res != 0) {
265  				break;
266  			}
267  			checker = (uint64_t) (entry->check);
268  			*handle = (checker << 32) | hdb->iterator;
269  			res = qb_hdb_handle_get(hdb, *handle, instance);
270  	
271  			hdb->iterator += 1;
272  			if (res == 0) {
273  				break;
274  			}
275  		}
276  		return (res);
277  	}
