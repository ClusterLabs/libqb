87   	void
88   	qb_log_format_init(void)
89   	{
90   		int32_t i;
91   		struct qb_log_target *t;
92   	
93   		i = pthread_rwlock_init(&_formatlock, NULL);
94   		assert(i == 0);
95   	
96   		for (i = 0; i < QB_LOG_TARGET_MAX; i++) {
97   			t = qb_log_target_get(i);
98   			t->format = strdup("[%p] %b");
99   		}
100  	}

102  	void
103  	qb_log_format_fini(void)
104  	{
105  		struct qb_log_target *t;
106  		int32_t i;
107  	
108  		pthread_rwlock_destroy(&_formatlock);
109  	
110  		for (i = 0; i < QB_LOG_TARGET_MAX; i++) {
111  			t = qb_log_target_get(i);
112  			free(t->format);
113  		}
114  	}

116  	void
117  	qb_log_format_set(int32_t target, const char *format)
118  	{
119  		char modified_format[256];
120  		struct qb_log_target *t = qb_log_target_get(target);
121  	
122  		pthread_rwlock_wrlock(&_formatlock);
123  	
124  		free(t->format);
125  	
126  		if (format) {
127  			qb_log_target_format_static(target, format, modified_format);
128  			t->format = strdup(modified_format);
129  		} else {
130  			t->format = strdup("[%p] %b");
131  		}
132  		assert(t->format != NULL);
133  	
134  		pthread_rwlock_unlock(&_formatlock);
135  	}

137  	/* Convert string "auth" to equivalent number "LOG_AUTH" etc. */
138  	int32_t
139  	qb_log_facility2int(const char *fname)
140  	{
141  		int32_t i;
142  	
143  		if (fname == NULL) {
144  			return -EINVAL;
145  		}
146  	
147  		for (i = 0; facilitynames[i].c_name != NULL; i++) {
148  			if (strcmp(fname, facilitynames[i].c_name) == 0) {
149  				return facilitynames[i].c_val;
150  			}
151  		}
152  		return -EINVAL;
153  	}

155  	/* Convert number "LOG_AUTH" to equivalent string "auth" etc. */
156  	const char *
157  	qb_log_facility2str(int32_t fnum)
158  	{
159  		int32_t i;
160  	
161  		for (i = 0; facilitynames[i].c_name != NULL; i++) {
162  			if (facilitynames[i].c_val == fnum) {
163  				return facilitynames[i].c_name;
164  			}
165  		}
166  		return NULL;
167  	}

169  	const char *
170  	qb_log_priority2str(uint8_t priority)
171  	{
172  		if (priority > LOG_TRACE) {
173  			return prioritynames[LOG_TRACE].c_name;
174  		}
175  		return prioritynames[priority].c_name;

184  	static int
185  	_strcpy_cutoff(char *dest, const char *src, size_t cutoff, int ralign,
186  		       size_t buf_len)
187  	{
188  		size_t len = strlen(src);
189  		if (buf_len <= 1) {
190  			if (buf_len == 0)
191  				dest[0] = 0;
192  			return 0;
193  		}
194  	
195  		if (cutoff == 0) {
196  			cutoff = len;
197  		}
198  	
199  		cutoff = QB_MIN(cutoff, buf_len - 1);
200  		len = QB_MIN(len, cutoff);
201  		if (ralign) {
202  			memset(dest, ' ', cutoff - len);
203  			memcpy(dest + cutoff - len, src, len);
204  		} else {
205  			memcpy(dest, src, len);
206  			memset(dest + len, ' ', cutoff - len);
207  		}
208  	
209  		dest[cutoff] = '\0';
210  	
211  		return cutoff;
212  	}

214  	/*
215  	 * This function will do static formatting (for things that don't
216  	 * change on each log message).
217  	 *
218  	 * %P PID
219  	 * %N name passed into qb_log_init
220  	 * %H hostname
221  	 *
222  	 * any number between % and character specify field length to pad or chop
223  	 */
224  	void
225  	qb_log_target_format_static(int32_t target, const char * format,
226  				    char *output_buffer)
227  	{
228  		char tmp_buf[255];
229  		unsigned int format_buffer_idx = 0;
230  		unsigned int output_buffer_idx = 0;
231  		size_t cutoff;
232  		uint32_t len;
233  		int ralign;
234  		int c;
235  		struct qb_log_target *t = qb_log_target_get(target);
236  	
237  		if (format == NULL) {
238  			return;
239  		}
240  	
241  		while ((c = format[format_buffer_idx])) {
242  			cutoff = 0;
243  			ralign = QB_FALSE;
244  			if (c != '%') {
245  				output_buffer[output_buffer_idx++] = c;
246  				format_buffer_idx++;
247  			} else {
248  				const char *p;
249  				unsigned int percent_buffer_idx = format_buffer_idx;
250  	
251  				format_buffer_idx += 1;
252  				if (format[format_buffer_idx] == '-') {
253  					ralign = QB_TRUE;
254  					format_buffer_idx += 1;
255  				}
256  	
257  				if (isdigit(format[format_buffer_idx])) {
258  					cutoff = atoi(&format[format_buffer_idx]);
259  				}
260  				while (isdigit(format[format_buffer_idx])) {
261  					format_buffer_idx += 1;
262  				}
263  	
264  				switch (format[format_buffer_idx]) {
265  				case 'P':
266  					snprintf(tmp_buf, 30, "%d", getpid());
267  					p = tmp_buf;
268  					break;
269  	
270  				case 'N':
271  					p = t->name;
272  					break;
273  	
274  				case 'H':
275  					if (gethostname(tmp_buf, 255) == 0) {
276  						tmp_buf[254] = '\0';
277  					} else {
278  						(void)strlcpy(tmp_buf, "localhost", 255);
279  					}
280  					p = tmp_buf;
281  					break;
282  	
283  				default:
284  					p = &format[percent_buffer_idx];
285  					cutoff = (format_buffer_idx - percent_buffer_idx + 1);
286  					ralign = QB_FALSE;
287  					break;
288  				}
289  				len = _strcpy_cutoff(output_buffer + output_buffer_idx,
290  						     p, cutoff, ralign,
291  						     (QB_LOG_MAX_LEN -
292  						      output_buffer_idx));
293  				output_buffer_idx += len;
294  				format_buffer_idx += 1;
295  			}
296  			if (output_buffer_idx >= QB_LOG_MAX_LEN - 1) {
297  				break;
298  			}
299  		}
300  	
301  		output_buffer[output_buffer_idx] = '\0';
302  	}

304  	/*
305  	 * %n FUNCTION NAME
306  	 * %f FILENAME
307  	 * %l FILELINE
308  	 * %p PRIORITY
309  	 * %t TIMESTAMP
310  	 * %b BUFFER
311  	 * %g SUBSYSTEM
312  	 *
313  	 * any number between % and character specify field length to pad or chop
314  	 */
315  	void
316  	qb_log_target_format(int32_t target,
317  			     struct qb_log_callsite *cs,
318  			     time_t current_time,
319  			     const char *formatted_message, char *output_buffer)
320  	{
321  		char tmp_buf[128];
322  		struct tm tm_res;
323  		unsigned int format_buffer_idx = 0;
324  		unsigned int output_buffer_idx = 0;
325  		size_t cutoff;
326  		uint32_t len;
327  		int ralign;
328  		int c;
329  		struct qb_log_target *t = qb_log_target_get(target);
330  	
331  		pthread_rwlock_rdlock(&_formatlock);
332  		if (t->format == NULL) {
333  			pthread_rwlock_unlock(&_formatlock);
334  			return;
335  		}
336  	
337  		while ((c = t->format[format_buffer_idx])) {
338  			cutoff = 0;
339  			ralign = QB_FALSE;
340  			if (c != '%') {
341  				output_buffer[output_buffer_idx++] = c;
342  				format_buffer_idx++;
343  			} else {
344  				const char *p;
345  	
346  				format_buffer_idx += 1;
347  				if (t->format[format_buffer_idx] == '-') {
348  					ralign = QB_TRUE;
349  					format_buffer_idx += 1;
350  				}
351  	
352  				if (isdigit(t->format[format_buffer_idx])) {
353  					cutoff = atoi(&t->format[format_buffer_idx]);
354  				}
355  				while (isdigit(t->format[format_buffer_idx])) {
356  					format_buffer_idx += 1;
357  				}
358  	
359  				switch (t->format[format_buffer_idx]) {
360  				case 'g':
361  					if (_user_tags_stringify_fn) {
362  						p = _user_tags_stringify_fn(cs->tags);
363  					} else {
364  						p = "";
365  					}
366  					break;
367  	
368  				case 'n':
369  					p = cs->function;
370  					break;
371  	
372  				case 'f':
373  	#ifdef BUILDING_IN_PLACE
374  					p = cs->filename;
375  	#else
376  					p = strrchr(cs->filename, '/');
377  					if (p == NULL) {
378  						p = cs->filename;
379  					} else {
380  						p++; /* move past the "/" */
381  					}
382  	#endif /* BUILDING_IN_PLACE */
383  					break;
384  	
385  				case 'l':
386  					snprintf(tmp_buf, 30, "%d", cs->lineno);
387  					p = tmp_buf;
388  					break;
389  	
390  				case 't':
391  					(void)localtime_r(&current_time, &tm_res);
392  					snprintf(tmp_buf, TIME_STRING_SIZE,
393  						 "%s %02d %02d:%02d:%02d",
394  						 log_month_name[tm_res.tm_mon],
395  						 tm_res.tm_mday, tm_res.tm_hour,
396  						 tm_res.tm_min, tm_res.tm_sec);
397  					p = tmp_buf;
398  					break;
399  	
400  				case 'b':
401  					p = formatted_message;
402  					break;
403  	
404  				case 'p':
405  					if (cs->priority > LOG_TRACE) {
406  						p = prioritynames[LOG_TRACE].c_name;
407  					} else {
408  						p = prioritynames[cs->priority].c_name;
409  					}
410  					break;
411  	
412  				default:
413  					p = "";
414  					break;
415  				}
416  				len = _strcpy_cutoff(output_buffer + output_buffer_idx,
417  						     p, cutoff, ralign,
418  						     (QB_LOG_MAX_LEN -
419  						      output_buffer_idx));
420  				output_buffer_idx += len;
421  				format_buffer_idx += 1;
422  			}
423  			if (output_buffer_idx >= QB_LOG_MAX_LEN - 1) {
424  				break;
425  			}
426  		}
427  		pthread_rwlock_unlock(&_formatlock);
428  	
429  		if (output_buffer[output_buffer_idx - 1] == '\n') {
430  			output_buffer[output_buffer_idx - 1] = '\0';
431  		} else {
432  			output_buffer[output_buffer_idx] = '\0';
433  		}
434  	}

437  	/*
438  	 * These wrappers around strl* functions just return the
439  	 * number of characters written, not the number of characters
440  	 * requested to be written.
441  	 */
442  	static size_t
443  	my_strlcpy(char *dest, const char * src, size_t maxlen)
444  	{
445  		size_t rc = strlcpy(dest, src, maxlen);
446  		/* maxlen includes NUL, so -1 */
447  		return QB_MIN(rc, maxlen-1);
448  	}
449  	
450  	static size_t
451  	my_strlcat(char *dest, const char * src, size_t maxlen)
452  	{
453  		size_t rc = strlcat(dest, src, maxlen);
454  		return QB_MIN(rc, maxlen-1);
455  	}

457  	size_t
458  	qb_vsnprintf_serialize(char *serialize, size_t max_len,
459  			       const char *fmt, va_list ap)
460  	{
461  		char *format;
462  		char *p;
463  		int type_long = QB_FALSE;
464  		int type_longlong = QB_FALSE;
465  	        int sformat_length = 0;
466  	        int sformat_precision = QB_FALSE;
467  		uint32_t location = my_strlcpy(serialize, fmt, max_len) + 1;
468  	
469  		format = (char *)fmt;
470  		for (;;) {
471  			type_long = QB_FALSE;
472  			type_longlong = QB_FALSE;
473  			p = strchrnul((const char *)format, '%');
474  			if (*p == '\0') {
475  				break;
476  			}
477  			format = p + 1;
478  	reprocess:
479  			switch (format[0]) {
480  			case '#': /* alternate form conversion, ignore */
481  			case '-': /* left adjust, ignore */
482  			case ' ': /* a space, ignore */
483  			case '+': /* a sign should be used, ignore */
484  			case '\'': /* group in thousands, ignore */
485  			case 'I': /* glibc-ism locale alternative, ignore */
486  	                    format++;
487  	                    goto reprocess;
488  			case '.': /* precision, ignore */
489  	                    format++;
490  	                    sformat_precision = QB_TRUE;
491  	                    goto reprocess;
492  			case '0': /* field width, ignore */
493  			case '1': /* field width, ignore */
494  			case '2': /* field width, ignore */
495  			case '3': /* field width, ignore */
496  			case '4': /* field width, ignore */
497  			case '5': /* field width, ignore */
498  			case '6': /* field width, ignore */
499  			case '7': /* field width, ignore */
500  			case '8': /* field width, ignore */
501  			case '9': /* field width, ignore */
502  	                        if (sformat_precision) {
503  	                            sformat_length *= 10;
504  	                            sformat_length += (format[0] - '0');
505  	                        }
506  				format++;
507  				goto reprocess;
508  			case '*': /* variable field width, save */ {
509  				int arg_int = va_arg(ap, int);
510  				if (location + sizeof (int) > max_len) {
511  					return max_len;
512  				}
513  				memcpy(&serialize[location], &arg_int, sizeof (int));
514  				location += sizeof(int);
515  				format++;
516  				goto reprocess;
517  			}
518  			case 'l':
519  				format++;
520  				type_long = QB_TRUE;
521  				if (*format == 'l') {
522  					type_long = QB_FALSE;
523  					type_longlong = QB_TRUE;
524  					format++;
525  				}
526  				goto reprocess;
527  			case 'd': /* int argument */
528  			case 'i': /* int argument */
529  			case 'o': /* unsigned int argument */
530  			case 'u':
531  			case 'x':
532  			case 'X':
533  				if (type_long) {
534  					long int arg_int;
535  	
536  					if (location + sizeof (long int) > max_len) {
537  						return max_len;
538  					}
539  					arg_int = va_arg(ap, long int);
540  					memcpy(&serialize[location], &arg_int,
541  					       sizeof(long int));
542  					location += sizeof(long int);
543  					format++;
544  					break;
545  				} else if (type_longlong) {
546  					long long int arg_int;
547  	
548  					if (location + sizeof (long long int) > max_len) {
549  						return max_len;
550  					}
551  					arg_int = va_arg(ap, long long int);
552  					memcpy(&serialize[location], &arg_int,
553  					       sizeof(long long int));
554  					location += sizeof(long long int);
555  					format++;
556  					break;
557  				} else {
558  					int arg_int;
559  	
560  					if (location + sizeof (int) > max_len) {
561  						return max_len;
562  					}
563  					arg_int = va_arg(ap, int);
564  					memcpy(&serialize[location], &arg_int,
565  					       sizeof(int));
566  					location += sizeof(int);
567  					format++;
568  					break;
569  				}
570  			case 'e':
571  			case 'E':
572  			case 'f':
573  			case 'F':
574  			case 'g':
575  			case 'G':
576  			case 'a':
577  			case 'A':
578  				{
579  				double arg_double;
580  	
581  				if (location + sizeof (double) > max_len) {
582  					return max_len;
583  				}
584  				arg_double = va_arg(ap, double);
585  				memcpy (&serialize[location], &arg_double, sizeof (double));
586  				location += sizeof(double);
587  				format++;
588  				break;
589  				}
590  			case 'c':
591  				{
592  				int arg_int;
593  				unsigned char arg_char;
594  	
595  				if (location + sizeof (unsigned int) > max_len) {
596  					return max_len;
597  				}
598  				arg_int = va_arg(ap, unsigned int);
599  				arg_char = (unsigned char)arg_int;
600  				memcpy (&serialize[location], &arg_char, sizeof (unsigned char));
601  				location += sizeof(unsigned char);
602  				break;
603  				}
604  			case 's':
605  				{
606  				char *arg_string;
607  				arg_string = va_arg(ap, char *);
608  				if (arg_string == NULL) {
609  					location += my_strlcpy(&serialize[location],
610  							   "(null)",
611  							   QB_MIN(strlen("(null)") + 1,
612  								  max_len - location));
613  				} else if (sformat_length) {
614  					location += my_strlcpy(&serialize[location],
615  							   arg_string,
616  							   QB_MIN(sformat_length + 1,
617  							   (max_len - location)));
618  				} else {
619  					location += my_strlcpy(&serialize[location],
620  							   arg_string,
621  							   QB_MIN(strlen(arg_string) + 1,
622  								  max_len - location));
623  				}
624  				location++;
625  				break;
626  				}
627  			case 'p':
628  				{
629  				ptrdiff_t arg_pointer = va_arg(ap, ptrdiff_t);
630  				if (location + sizeof (ptrdiff_t) > max_len) {
631  					return max_len;
632  				}
633  				memcpy(&serialize[location], &arg_pointer, sizeof(ptrdiff_t));
634  				location += sizeof(ptrdiff_t);
635  				break;
636  				}
637  			case '%':
638  				if (location + 1 > max_len) {
639  					return max_len;
640  				}
641  				serialize[location++] = '%';
642  	                        sformat_length = 0;
643  	                        sformat_precision = QB_FALSE;
644  				break;
645  	
646  			}
647  		}
648  		return (location);
649  	}

652  	
653  	size_t
654  	qb_vsnprintf_deserialize(char *string, size_t str_len, const char *buf)
655  	{
656  		char *p;
657  		char *format;
658  		char fmt[MINI_FORMAT_STR_LEN];
659  		int fmt_pos;
