29   	static int32_t do_benchmark = QB_FALSE;
30   	static int32_t use_events = QB_FALSE;
31   	static int alarm_notice;
32   	static qb_util_stopwatch_t *sw;
33   	#define ONE_MEG 1048576
34   	#define MAX_MSG_SIZE ONE_MEG
35   	static char data[ONE_MEG];
36   	
37   	struct my_req {
38   		struct qb_ipc_request_header hdr;
39   		char message[256];
40   	};
41   	
42   	struct my_res {
43   		struct qb_ipc_response_header hdr;
44   		char message[256];
45   	};

47   	static void sigalrm_handler (int num)
48   	{
49   		alarm_notice = 1;
50   	}
51   	
52   	static void
53   	_benchmark(qb_ipcc_connection_t *conn, int write_size)
54   	{
55   		struct iovec iov[2];
56   		unsigned int res;
57   		struct qb_ipc_request_header hdr;
58   		int write_count = 0;
59   		float secs;
60   	
61   		alarm_notice = 0;
62   		hdr.size = write_size;
63   		hdr.id = QB_IPC_MSG_USER_START + 1;
64   	
65   		iov[0].iov_base = &hdr;
66   		iov[0].iov_len = sizeof(struct qb_ipc_request_header);
67   	
68   		iov[1].iov_base = data;
69   		iov[1].iov_len = write_size - sizeof(struct qb_ipc_request_header);
70   	
71   		alarm (10);
72   	
73   		qb_util_stopwatch_start(sw);
74   		do {
75   			res = qb_ipcc_sendv(conn, iov, 2);
76   			if (res == write_size) {
77   				write_count++;
78   			}
79   		} while (alarm_notice == 0 && (res == write_size || res == -EAGAIN));
(1) Event unsigned_compare: This less-than-zero comparison of an unsigned value is never true. "res < 0U". 
80   		if (res < 0) {
81   			perror("qb_ipcc_sendv");
82   		}
83   		qb_util_stopwatch_stop(sw);
84   		secs = qb_util_stopwatch_sec_elapsed_get(sw);
85   	
86   		printf ("%5d messages sent ", write_count);
87   		printf ("%5d bytes per write ", write_size);
88   		printf ("%7.3f Seconds runtime ", secs);
89   		printf ("%9.3f TP/s ",
90   			((float)write_count) / secs);
91   		printf ("%7.3f MB/s.\n",
92   			((float)write_count) * ((float)write_size) / secs);
93   	}

96   	static void
97   	do_throughput_benchmark(qb_ipcc_connection_t *conn)
98   	{
99   		ssize_t size = 64;
100  		int i;
101  	
102  		signal (SIGALRM, sigalrm_handler);
103  		sw =  qb_util_stopwatch_create();
104  	
105  		for (i = 0; i < 10; i++) { /* number of repetitions - up to 50k */
106  			_benchmark (conn, size);
107  			signal (SIGALRM, sigalrm_handler);
108  			size *= 5;
109  			if (size >= (ONE_MEG - 100)) {
110  				break;
111  			}
112  		}
113  	}

115  	static void
116  	do_echo(qb_ipcc_connection_t *conn)
117  	{
118  		struct my_req req;
119  		struct my_res res;
120  		char *newline;
121  		int32_t rc;
122  		int32_t send_ten_events;
123  	
124  		while (1) {
125  			printf("SEND (q or Q to quit) : ");
126  			if (fgets(req.message, 256, stdin) == NULL) {
127  				continue;
128  			}
129  			newline = strrchr(req.message, '\n');
130  			if (newline) {
131  				*newline = '\0';
132  			}
133  	
134  			if (strcasecmp(req.message, "q") == 0) {
135  				break;
136  			} else {
137  				req.hdr.id = QB_IPC_MSG_USER_START + 3;
138  				req.hdr.size = sizeof(struct my_req);
139  				rc = qb_ipcc_send(conn, &req, req.hdr.size);
140  				if (rc < 0) {
141  					perror("qb_ipcc_send");
142  					exit(0);
143  				}
144  			}
145  	
146  			send_ten_events = (strcasecmp(req.message, "events") == 0);
147  	
148  			if (rc > 0) {
149  				if (use_events && !send_ten_events) {
150  					printf("waiting for event recv\n");
151  					rc = qb_ipcc_event_recv(conn, &res, sizeof(res), -1);
152  				} else {
153  					printf("waiting for recv\n");
154  					rc = qb_ipcc_recv(conn, &res, sizeof(res), -1);
155  				}
156  				printf("recv %d\n", rc);
157  				if (rc < 0) {
158  					perror("qb_ipcc_recv");
159  					exit(0);
160  				}
161  				if (send_ten_events) {
162  					int32_t i;
163  					printf("waiting for 10 events\n");
164  					for (i = 0; i < 10; i++) {
165  						rc = qb_ipcc_event_recv(conn, &res, sizeof(res), -1);
166  						if (rc < 0) {
167  							perror("qb_ipcc_event_recv");
168  						} else {
169  							printf("got event %d rc:%d\n", i, rc);
170  						}
171  					}
172  				}
173  				printf("Response[%d]: %s \n", res.hdr.id, res.message);
174  			}
175  		}
176  	}

178  	static void
179  	show_usage(const char *name)
180  	{
181  		printf("usage: \n");
182  		printf("%s <options>\n", name);
183  		printf("\n");
184  		printf("  options:\n");
185  		printf("\n");
186  		printf("  -h             show this help text\n");
187  		printf("  -b             benchmark\n");
188  		printf("  -e             use events instead of responses\n");
189  		printf("\n");
190  	}
191  	
192  	int
193  	main(int argc, char *argv[])
194  	{
195  		qb_ipcc_connection_t *conn;
196  		const char *options = "eb";
197  		int32_t opt;
198  	
199  		while ((opt = getopt(argc, argv, options)) != -1) {
200  			switch (opt) {
201  			case 'b':
202  				do_benchmark = QB_TRUE;
203  				break;
204  			case 'e':
205  				use_events = QB_TRUE;
206  				break;
207  			case 'h':
208  			default:
209  				show_usage(argv[0]);
210  				exit(0);
211  				break;
212  			}
213  		}
214  	
215  	
216  		qb_log_init("ipcclient", LOG_USER, LOG_TRACE);
217  		qb_log_ctl(QB_LOG_SYSLOG, QB_LOG_CONF_ENABLED, QB_FALSE);
218  		qb_log_filter_ctl(QB_LOG_STDERR, QB_LOG_FILTER_ADD,
219  				  QB_LOG_FILTER_FILE, "*", LOG_TRACE);
220  		qb_log_format_set(QB_LOG_STDERR, "%f:%l [%p] %b");
221  		qb_log_ctl(QB_LOG_STDERR, QB_LOG_CONF_ENABLED, QB_TRUE);
222  	
223  		conn = qb_ipcc_connect("ipcserver", MAX_MSG_SIZE);
224  		if (conn == NULL) {
225  			perror("qb_ipcc_connect");
226  			exit(1);
227  		}
228  	
229  		if (do_benchmark) {
230  			do_throughput_benchmark(conn);
231  		} else {
232  			do_echo(conn);
233  		}
234  	
235  		qb_ipcc_disconnect(conn);
236  		return EXIT_SUCCESS;
237  	}
