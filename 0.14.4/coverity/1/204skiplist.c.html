61   	static int8_t
62   	skiplist_level_generate(void)
63   	{
64   		/* This constant is found by 1 / P, where P = 0.25. */
65   		enum { P_INVERSE = 4 };
66   	
67   		/* The original algorithm's random number is in the range [0, 1), so
68   		 * max M = 1. Its ceiling C = M * P = 1 * P = P.
69   		 *
70   		 * Our random number is in the range [0, UINT16_MAX], so M = UINT16_MAX. Therefore,
71   		 * C = UINT16_MAX * P = UINT16_MAX / P_INVERSE.
72   		 */
73   		enum { P_CEIL = UINT16_MAX / P_INVERSE };
74   		int8_t level = SKIPLIST_LEVEL_MIN;
75   	
76   		while ((uint16_t) (random()) < P_CEIL)
77   			level++;
78   	
79   		if (level < SKIPLIST_LEVEL_MAX)
80   			return level;
81   	
82   		return SKIPLIST_LEVEL_MAX;
83   	}

85   	static struct skiplist_node *
86   	skiplist_node_next(const struct skiplist_node *node)
87   	{
88   		const struct skiplist_node *n = node;
89   		do {
90   			n = n->forward[SKIPLIST_LEVEL_MIN];
91   		} while (n && n->refcount == 0);
92   		return (struct skiplist_node *)n;
93   	}

95   	/*
96   	 * Create a new level @level node with value @value. The node should eventually
97   	 * be destroyed with @skiplist_node_destroy.
98   	 *
99   	 * return: a new node on success and NULL otherwise.
100  	 */
101  	static struct skiplist_node *
102  	skiplist_node_new(const int8_t level, const char *key, const void *value)
103  	{
104  		struct skiplist_node *new_node = (struct skiplist_node *)
105  		    (malloc(sizeof(struct skiplist_node)));
106  	
107  		if (!new_node)
108  			return NULL;
109  	
110  		new_node->value = (void *)value;
111  		new_node->key = key;
112  		new_node->level = level;
113  		new_node->refcount = 1;
114  		qb_list_init(&new_node->notifier_head);
115  	
116  		/* A level 0 node still needs to hold 1 forward pointer, etc. */
117  		new_node->forward = (struct skiplist_node **)
118  		    (calloc(level + 1, sizeof(struct skiplist_node *)));
119  	
120  		if (new_node->forward == NULL) {
121  			free(new_node);
122  			return NULL;
123  		}
124  	
125  		return new_node;
126  	}

134  	/* An operation to perform after comparing a user value or search with a
135  	 * node's value
136  	 */
137  	typedef enum {
138  		OP_GOTO_NEXT_LEVEL,
139  		OP_GOTO_NEXT_NODE,
140  		OP_FINISH,
141  	} op_t;
142  	
143  	static op_t
144  	op_search(const struct skiplist *list,
145  		  const struct skiplist_node *fwd_node, const void *search)
146  	{
147  		int32_t cmp;
148  	
149  		if (!fwd_node)
150  			return OP_GOTO_NEXT_LEVEL;
151  	
152  		cmp = strcmp(fwd_node->key, search);
153  		if (cmp < 0) {
154  			return OP_GOTO_NEXT_NODE;
155  		} else if (cmp == 0) {
156  			return OP_FINISH;
157  		}
158  	
159  		return OP_GOTO_NEXT_LEVEL;
160  	}

162  	static struct skiplist_node *
163  	skiplist_lookup(struct skiplist *list, const char *key)
164  	{
165  		struct skiplist_node *cur_node = list->header;
166  		int8_t level = list->level;
167  	
168  		while (level >= SKIPLIST_LEVEL_MIN) {
169  			struct skiplist_node *fwd_node = cur_node->forward[level];
170  	
171  			switch (op_search(list, fwd_node, key)) {
172  			case OP_FINISH:
173  				return fwd_node;
174  			case OP_GOTO_NEXT_NODE:
175  				cur_node = fwd_node;
176  				break;
177  			case OP_GOTO_NEXT_LEVEL:
178  				level--;
179  			}
180  		}
181  	
182  		return NULL;
183  	}

185  	static void
186  	skiplist_notify(struct skiplist *l, struct skiplist_node *n,
187  			uint32_t event, char *key, void *old_value, void *value)
188  	{
189  		struct qb_list_head *list;
190  		struct qb_map_notifier *tn;
191  	
192  		/* node callbacks
193  		 */
194  		qb_list_for_each(list, &n->notifier_head) {
195  			tn = qb_list_entry(list, struct qb_map_notifier, list);
196  	
197  			if (tn->events & event) {
198  				tn->callback(event, key, old_value, value,
199  					     tn->user_data);
200  			}
201  		}
202  		/* global callbacks
203  		 */
204  		qb_list_for_each(list, &l->header->notifier_head) {
205  			tn = qb_list_entry(list, struct qb_map_notifier, list);
206  	
207  			if (tn->events & event) {
208  				tn->callback(event, key, old_value, value,
209  					     tn->user_data);
210  			}
211  			if (((event & QB_MAP_NOTIFY_DELETED) ||
212  			     (event & QB_MAP_NOTIFY_REPLACED)) &&
213  			    (tn->events & QB_MAP_NOTIFY_FREE)) {
214  				tn->callback(QB_MAP_NOTIFY_FREE, (char *)key,
215  					     old_value, value, tn->user_data);
216  			}
217  		}
218  	
219  	}

221  	static void
222  	skiplist_node_destroy(struct skiplist_node *node, struct skiplist *list)
223  	{
224  		struct qb_list_head *pos;
225  		struct qb_list_head *next;
226  		struct qb_map_notifier *tn;
227  	
228  		skiplist_notify(list, node,
229  				QB_MAP_NOTIFY_DELETED,
230  				(char *)node->key, node->value, NULL);
231  	
232  		qb_list_for_each_safe(pos, next, &node->notifier_head) {
233  			tn = qb_list_entry(pos, struct qb_map_notifier, list);
234  			qb_list_del(&tn->list);
235  			free(tn);
236  		}
237  	
238  		free(node->forward);
239  		free(node);
240  	}

242  	static void
243  	skiplist_node_deref(struct skiplist_node *node, struct skiplist *list)
244  	{
245  		node->refcount--;
246  		if (node->refcount == 0) {
247  			skiplist_node_destroy(node, list);
248  		}
249  	}

251  	static int32_t
252  	skiplist_notify_add(qb_map_t * m, const char *key,
253  			    qb_map_notify_fn fn, int32_t events, void *user_data)
254  	{
255  		struct skiplist *t = (struct skiplist *)m;
256  		struct qb_map_notifier *f;
257  		struct skiplist_node *n;
258  		struct qb_list_head *list;
259  		int add_to_tail = QB_FALSE;
260  	
261  		if (key) {
262  			n = skiplist_lookup(t, key);
263  		} else {
264  			n = t->header;
265  		}
266  		if (events & QB_MAP_NOTIFY_FREE) {
267  			add_to_tail = QB_TRUE;
268  		}
269  		if (n) {
270  			qb_list_for_each(list, &n->notifier_head) {
271  				f = qb_list_entry(list, struct qb_map_notifier, list);
272  	
273  				if (events & QB_MAP_NOTIFY_FREE &&
274  				    f->events == events) {
275  					/* only one free notifier */
276  					return -EEXIST;
277  				}
278  				if (f->events == events &&
279  				    f->callback == fn &&
280  				    f->user_data == user_data) {
281  					return -EEXIST;
282  				}
283  			}
284  	
285  			f = malloc(sizeof(struct qb_map_notifier));
286  			if (f == NULL) {
287  				return -errno;
288  			}
289  			f->events = events;
290  			f->user_data = user_data;
291  			f->callback = fn;
292  			qb_list_init(&f->list);
293  			if (add_to_tail) {
294  				qb_list_add_tail(&f->list, &n->notifier_head);
295  			} else {
296  				qb_list_add(&f->list, &n->notifier_head);
297  			}
298  			return 0;
299  		}
300  		return -EINVAL;
301  	}

303  	static int32_t
304  	skiplist_notify_del(qb_map_t * m, const char *key,
305  			    qb_map_notify_fn fn, int32_t events,
306  			    int32_t cmp_userdata, void *user_data)
307  	{
308  		struct skiplist *t = (struct skiplist *)m;
309  		struct skiplist_node *n;
310  		struct qb_map_notifier *f;
311  		struct qb_list_head *head = NULL;
312  		struct qb_list_head *list;
313  		struct qb_list_head *next;
314  		int32_t found = QB_FALSE;
315  	
316  		if (key) {
317  			n = skiplist_lookup(t, key);
318  			if (n) {
319  				head = &n->notifier_head;
320  			}
321  		} else {
322  			head = &t->header->notifier_head;
323  		}
324  		if (head == NULL) {
325  			return -ENOENT;
326  		}
327  		qb_list_for_each_safe(list, next, head) {
328  			f = qb_list_entry(list, struct qb_map_notifier, list);
329  	
330  			if (f->events == events && f->callback == fn) {
331  				if (cmp_userdata && (f->user_data == user_data)) {
332  					found = QB_TRUE;
333  					qb_list_del(&f->list);
334  					free(f);
335  				} else if (!cmp_userdata) {
336  					found = QB_TRUE;
337  					qb_list_del(&f->list);
338  					free(f);
339  				}
340  			}
341  		}
342  		if (found) {
343  			return 0;
344  		} else {
345  			return -ENOENT;
346  		}
347  	}

349  	static void
350  	skiplist_destroy(struct qb_map *map)
351  	{
352  		struct skiplist *list = (struct skiplist *)map;
353  		struct skiplist_node *cur_node;
354  		struct skiplist_node *fwd_node;
355  	
356  		for (cur_node = skiplist_node_next(list->header);
357  		     cur_node; cur_node = fwd_node) {
358  			fwd_node = skiplist_node_next(cur_node);
359  			skiplist_node_destroy(cur_node, list);
360  		}
361  		skiplist_node_destroy(list->header, list);
362  		free(list);
363  	}

427  	static int32_t
428  	skiplist_rm(struct qb_map *map, const char *key)
429  	{
430  		struct skiplist *list = (struct skiplist *)map;
431  		struct skiplist_node *found_node;
432  		struct skiplist_node *cur_node = list->header;
433  		int8_t level = list->level;
434  		int8_t update_level;
435  		skiplist_update_t update;
436  	
437  		while ((update_level = level) >= SKIPLIST_LEVEL_MIN) {
438  			struct skiplist_node *fwd_node = cur_node->forward[level];
439  	
440  			switch (op_search(list, fwd_node, key)) {
441  			case OP_GOTO_NEXT_NODE:
442  				cur_node = fwd_node;
443  				break;
444  			case OP_GOTO_NEXT_LEVEL:
445  			default:
446  				level--;
447  				break;
448  			}
449  	
450  			update[update_level] = cur_node;
451  		}
452  	
453  		/* The immediate forward node should be the matching node... */
454  		found_node = skiplist_node_next(cur_node);
455  	
456  		/* ...unless we're at the end of the list or the value doesn't exist. */
457  		if (!found_node || strcmp(found_node->key, key) != 0) {
458  			return QB_FALSE;
459  		}
460  	
461  		/* Splice found_node out of list. */
462  		for (level = SKIPLIST_LEVEL_MIN; level <= list->level; level++)
463  			if (update[level]->forward[level] == found_node)
464  				update[level]->forward[level] =
465  				    found_node->forward[level];
466  	
467  		skiplist_node_deref(found_node, list);
468  	
469  		/* Remove unused levels from @list -- stop removing levels as soon as a
470  		 * used level is found. Unused levels can occur if @found_node had the
471  		 * highest level.
472  		 */
473  		for (level = list->level; level >= SKIPLIST_LEVEL_MIN; level--) {
474  			if (list->header->forward[level])
475  				break;
476  	
477  			list->level--;
478  		}
479  	
480  		list->length--;
481  	
482  		return QB_TRUE;
483  	}

486  	skiplist_get(struct qb_map *map, const char *key)
487  	{
488  		struct skiplist *list = (struct skiplist *)map;
489  		struct skiplist_node *n = skiplist_lookup(list, key);

493  	
494  		return NULL;
495  	}

497  	static qb_map_iter_t *
498  	skiplist_iter_create(struct qb_map *map, const char *prefix)
499  	{
500  		struct skiplist_iter *i = malloc(sizeof(struct skiplist_iter));
501  		struct skiplist *list = (struct skiplist *)map;
502  		if (i == NULL) {
503  			return NULL;
504  		}
505  		i->i.m = map;
506  		i->n = list->header;
507  		i->n->refcount++;
508  		return (qb_map_iter_t *) i;
509  	}

511  	static const char *
512  	skiplist_iter_next(qb_map_iter_t * i, void **value)
513  	{
514  		struct skiplist_iter *si = (struct skiplist_iter *)i;
515  		struct skiplist_node *p = si->n;
516  	
517  		if (p == NULL) {
518  			return NULL;
519  		}
520  		si->n = skiplist_node_next(p);
521  		if (si->n == NULL) {
522  			skiplist_node_deref(p, (struct skiplist *)i->m);
523  			return NULL;
524  		}
525  		si->n->refcount++;
526  		skiplist_node_deref(p, (struct skiplist *)i->m);
527  		*value = si->n->value;
528  		return si->n->key;
529  	}
