34   	static int32_t
35   	_poll_to_epoll_event_(int32_t event)
36   	{
37   		int32_t out = 0;
38   		if (event & POLLIN)
39   			out |= EPOLLIN;
40   		if (event & POLLOUT)
41   			out |= EPOLLOUT;
42   		if (event & POLLPRI)
43   			out |= EPOLLPRI;
44   		if (event & POLLERR)
45   			out |= EPOLLERR;
46   		if (event & POLLHUP)
47   			out |= EPOLLHUP;
48   		if (event & POLLNVAL)
49   			out |= EPOLLERR;
50   		return out;
51   	}
52   	
53   	static int32_t
54   	_epoll_to_poll_event_(int32_t event)
55   	{
56   		int32_t out = 0;
57   		if (event & EPOLLIN)
58   			out |= POLLIN;
59   		if (event & EPOLLOUT)
60   			out |= POLLOUT;
61   		if (event & EPOLLPRI)
62   			out |= POLLPRI;
63   		if (event & EPOLLERR)
64   			out |= POLLERR;
65   		if (event & EPOLLHUP)
66   			out |= POLLHUP;
67   		return out;
68   	}

70   	static void
71   	_fini(struct qb_poll_source *s)
72   	{
73   		if (s->epollfd != -1) {
74   			close(s->epollfd);
75   			s->epollfd = -1;
76   		}
77   	}

79   	static int32_t
80   	_add(struct qb_poll_source *s, struct qb_poll_entry *pe, int32_t fd, int32_t events)
81   	{
82   		struct epoll_event ev;
83   		int32_t res = 0;
84   	
85   		ev.events = _poll_to_epoll_event_(events);
86   		ev.data.u64 = (((uint64_t) (pe->check)) << 32) | pe->install_pos;
87   		if (epoll_ctl(s->epollfd, EPOLL_CTL_ADD, fd, &ev) == -1) {
88   			res = -errno;

89   			qb_util_perror(LOG_ERR, "epoll_ctl(add)");
90   		}
91   		return res;
92   	}

98   		struct epoll_event ev;
99   		int32_t res = 0;
100  	
101  		ev.events = _poll_to_epoll_event_(events);
102  		ev.data.u64 = (((uint64_t) (pe->check)) << 32) | pe->install_pos;
103  		if (epoll_ctl(s->epollfd, EPOLL_CTL_MOD, fd, &ev) == -1) {
104  			res = -errno;

122  	static int32_t
123  	_poll_entry_from_handle_(struct qb_poll_source *s,
124  				 uint64_t handle_in, struct qb_poll_entry **pe_pt)
125  	{
126  		int32_t res = 0;
127  		uint32_t check = ((uint32_t) (((uint64_t) handle_in) >> 32));
128  		uint32_t handle = handle_in & 0xffffffff;
129  		struct qb_poll_entry *pe;
130  	
131  		res = qb_array_index(s->poll_entries, handle, (void **)&pe);
132  		if (res != 0) {
133  			return res;
134  		}
135  		if (pe->check != check) {
136  			return -EINVAL;
137  		}
138  		*pe_pt = pe;
139  		return 0;
140  	}

142  	static int32_t
143  	_poll_and_add_to_jobs_(struct qb_loop_source *src, int32_t ms_timeout)
144  	{
145  		int32_t i;
146  		int32_t res;
147  		int32_t event_count;
148  		int32_t new_jobs = 0;
149  		struct qb_poll_entry *pe = NULL;
150  		struct qb_poll_source *s = (struct qb_poll_source *)src;
151  		struct epoll_event events[MAX_EVENTS];
152  	
153  		qb_poll_fds_usage_check_(s);
154  	
155  	retry_poll:
156  	
157  		event_count = epoll_wait(s->epollfd, events, MAX_EVENTS, ms_timeout);
158  	
159  		if (errno == EINTR && event_count == -1) {
160  			goto retry_poll;
161  		} else if (event_count == -1) {
162  			return -errno;
163  		}
164  	
165  		for (i = 0; i < event_count; i++) {
166  			res = _poll_entry_from_handle_(s, events[i].data.u64, &pe);
167  			if (res != 0) {
168  				qb_util_log(LOG_WARNING,
169  					    "can't find poll entry for new event.");
170  				continue;
171  			}
172  			if (pe->ufd.fd == -1 || pe->state == QB_POLL_ENTRY_DELETED) {
173  				qb_util_log(LOG_WARNING,
174  					    "can't post new event to a deleted entry.");
175  				/*
176  				 * empty/deleted
177  				 */
178  				continue;
179  			}
180  			if (events[i].events == pe->ufd.revents ||
181  			    pe->state == QB_POLL_ENTRY_JOBLIST) {
182  				/*
183  				 * entry already in the job queue.
184  				 */
185  				continue;
186  			}
187  			pe->ufd.revents = _epoll_to_poll_event_(events[i].events);
188  	
189  			new_jobs += pe->add_to_jobs(src->l, pe);
190  		}
191  	
192  		return new_jobs;
193  	}

195  	int32_t
196  	qb_epoll_init(struct qb_poll_source *s)
197  	{
198  		s->epollfd = epoll_create1(EPOLL_CLOEXEC);
199  		if (s->epollfd < 0) {
200  			return -errno;
201  		}
202  		s->driver.fini = _fini;
203  		s->driver.add = _add;
204  		s->driver.mod = _mod;
205  		s->driver.del = _del;
206  		s->s.poll = _poll_and_add_to_jobs_;
207  		return 0;
208  	}
