78   	static int32_t
79   	_cs_matches_filter_(struct qb_log_callsite *cs,
80   			    enum qb_log_filter_type type,
81   			    const char *text,
82   			    uint8_t high_priority,
83   			    uint8_t low_priority)
84   	{
85   		int32_t match = QB_FALSE;
86   	
87   		if (cs->priority > low_priority ||
88   		    cs->priority < high_priority) {
89   			return QB_FALSE;
90   		}
91   		if (strcmp(text, "*") == 0) {
92   			return QB_TRUE;
93   		}
94   		if (type == QB_LOG_FILTER_FILE ||
95   		    type == QB_LOG_FILTER_FUNCTION) {
96   			char token[500];
97   			const char *offset = NULL;
98   			const char *next = text;
99   	
100  			do {
101  				offset = next;
102  				next = strchrnul(offset, ',');
103  				snprintf(token, 499, "%.*s", (int)(next - offset), offset);
104  	
105  				if (type == QB_LOG_FILTER_FILE) {
106  					match = (strstr(cs->filename, token) != NULL);
107  				} else {
108  					match = (strstr(cs->function, token) != NULL);
109  				}
110  				if (!match && next[0] != 0) {
111  					next++;
112  				}
113  	
114  			} while (match == QB_FALSE && next != NULL && next[0] != 0);
115  		} else if (type == QB_LOG_FILTER_FORMAT) {
116  			if (strstr(cs->format, text)) {
117  				match = QB_TRUE;
118  			}
119  		}
120  		return match;
121  	}

216  	void
217  	qb_log_thread_log_write(struct qb_log_callsite *cs,
218  				time_t timestamp, const char *buffer)
219  	{
220  		struct qb_log_target *t;
221  		int32_t pos;
222  	
223  		for (pos = 0; pos <= conf_active_max; pos++) {
224  			t = &conf[pos];
225  			if (t->state != QB_LOG_STATE_ENABLED) {
226  				continue;
227  			}
228  			if (!t->threaded) {
229  				continue;
230  			}
231  			if (qb_bit_is_set(cs->targets, t->pos)) {
232  				t->logger(t->pos, cs, timestamp, buffer);
233  			}
234  		}
235  	}

237  	struct qb_log_callsite*
238  	qb_log_callsite_get(const char *function,
239  			    const char *filename,
240  			    const char *format,
241  			    uint8_t priority,
242  			    uint32_t lineno,
243  			    uint32_t tags)
244  	{
245  		struct qb_log_target *t;
246  		struct qb_log_filter *flt;
247  		struct qb_log_callsite *cs;
248  		int32_t new_dcs = QB_FALSE;
249  		struct qb_list_head *f_item;
250  		int32_t pos;
251  	
252  		if (!logger_inited) {
253  			return NULL;
254  		}
255  	
256  		cs = qb_log_dcs_get(&new_dcs, function, filename,
257  				    format, priority, lineno, tags);
258  		if (cs == NULL) {
259  			return NULL;
260  		}
261  	
262  		if (new_dcs) {
263  			pthread_rwlock_rdlock(&_listlock);
264  			for (pos = 0; pos <= conf_active_max; pos++) {
265  				t = &conf[pos];
266  				if (t->state != QB_LOG_STATE_ENABLED) {
267  					continue;
268  				}
269  				qb_list_for_each(f_item, &t->filter_head) {
270  					flt = qb_list_entry(f_item, struct qb_log_filter, list);
271  					_log_filter_apply_to_cs(cs, t->pos, flt->conf, flt->type,
272  								flt->text, flt->high_priority,
273  								flt->low_priority);
274  				}
275  			}
276  			if (tags == 0) {
277  				qb_list_for_each(f_item, &tags_head) {
278  					flt = qb_list_entry(f_item, struct qb_log_filter, list);
279  					_log_filter_apply_to_cs(cs, flt->new_value, flt->conf, flt->type,
280  								flt->text, flt->high_priority,
281  								flt->low_priority);
282  				}
283  			} else {
284  				cs->tags = tags;
285  			}
286  			if (_custom_filter_fn) {
287  				_custom_filter_fn(cs);
288  			}
289  			pthread_rwlock_unlock(&_listlock);
290  		} else if (cs->tags != tags) {
291  			cs->tags = tags;
292  			if (_custom_filter_fn) {
293  				_custom_filter_fn(cs);
294  			}
295  		}
296  		return cs;
297  	}

338  	int32_t
339  	qb_log_callsites_register(struct qb_log_callsite *_start,
340  				  struct qb_log_callsite *_stop)
341  	{
342  		struct callsite_section *sect;
343  		struct qb_log_callsite *cs;
344  		struct qb_log_target *t;
345  		struct qb_log_filter *flt;
346  		int32_t pos;
347  	
348  		if (_start == NULL || _stop == NULL) {
349  			return -EINVAL;
350  		}
351  	
352  		pthread_rwlock_rdlock(&_listlock);
353  		qb_list_for_each_entry(sect, &callsite_sections, list) {
354  			if (sect->start == _start || sect->stop == _stop) {
355  				pthread_rwlock_unlock(&_listlock);
356  				return -EEXIST;
357  			}
358  		}
359  		pthread_rwlock_unlock(&_listlock);
360  	
361  		sect = calloc(1, sizeof(struct callsite_section));
362  		if (sect == NULL) {
363  			return -ENOMEM;
364  		}
365  		sect->start = _start;
366  		sect->stop = _stop;
367  		qb_list_init(&sect->list);
368  	
369  		pthread_rwlock_wrlock(&_listlock);
370  		qb_list_add(&sect->list, &callsite_sections);
371  	
372  		/*
373  		 * Now apply the filters on these new callsites
374  		 */
375  		for (pos = 0; pos <= conf_active_max; pos++) {
376  			t = &conf[pos];
377  			if (t->state != QB_LOG_STATE_ENABLED) {
378  				continue;
379  			}
380  			qb_list_for_each_entry(flt, &t->filter_head, list) {
381  				_log_filter_apply(sect, t->pos, flt->conf,
382  						  flt->type, flt->text,
383  						  flt->high_priority, flt->low_priority);
384  			}
385  		}
386  		qb_list_for_each_entry(flt, &tags_head, list) {
387  			_log_filter_apply(sect, flt->new_value, flt->conf,
388  					  flt->type, flt->text,
389  					  flt->high_priority, flt->low_priority);
390  		}
391  		pthread_rwlock_unlock(&_listlock);
392  		if (_custom_filter_fn) {
393  			for (cs = sect->start; cs < sect->stop; cs++) {
394  				if (cs->lineno > 0) {
395  					_custom_filter_fn(cs);
396  				}
397  			}
398  		}
399  		/* qb_log_callsites_dump_sect(sect); */
400  	
401  		return 0;
402  	}

414  				       cs->targets, cs->tags);
415  			}
416  		}

435  	static int32_t
436  	_log_filter_exists(struct qb_list_head *list_head,
437  			   enum qb_log_filter_type type,
438  			   const char *text,
439  			   uint8_t high_priority,
440  			   uint8_t low_priority,
441  			   uint32_t new_value)
442  	{
443  		struct qb_log_filter *flt;
444  	
445  		qb_list_for_each_entry(flt, list_head, list) {
446  			if (flt->type == type &&
447  			    flt->high_priority == high_priority &&
448  			    flt->low_priority == low_priority &&
449  			    flt->new_value == new_value &&
450  			    strcmp(flt->text, text) == 0) {
451  				return QB_TRUE;
452  			}
453  		}
454  		return QB_FALSE;
455  	}

457  	static int32_t
458  	_log_filter_store(uint32_t t, enum qb_log_filter_conf c,
459  			  enum qb_log_filter_type type,
460  			  const char *text,
461  			  uint8_t high_priority,
462  			  uint8_t low_priority)
463  	{
464  		struct qb_log_filter *flt;
465  		struct qb_list_head *iter;
466  		struct qb_list_head *next;
467  		struct qb_list_head *list_head;
468  	
469  		switch (c) {
470  		case QB_LOG_FILTER_ADD:
471  		case QB_LOG_FILTER_REMOVE:
472  		case QB_LOG_FILTER_CLEAR_ALL:
473  			list_head = &conf[t].filter_head;
474  			break;
475  	
476  		case QB_LOG_TAG_SET:
477  		case QB_LOG_TAG_CLEAR:
478  		case QB_LOG_TAG_CLEAR_ALL:
479  			list_head = &tags_head;
480  			break;
481  		default:
482  			return -ENOSYS;
483  		}
484  	
485  		if (c == QB_LOG_FILTER_ADD || c == QB_LOG_TAG_SET) {
486  			if (text == NULL) {
487  				return -EINVAL;
488  			}
489  			if (_log_filter_exists(list_head, type, text,
490  					       high_priority, low_priority, t)) {
491  				return -EEXIST;
492  			}
493  			flt = calloc(1, sizeof(struct qb_log_filter));
494  			if (flt == NULL) {
495  				return -ENOMEM;
496  			}
497  			qb_list_init(&flt->list);
498  			flt->conf = c;
499  			flt->type = type;
500  			flt->text = strdup(text);
501  			if (flt->text == NULL) {
502  				free(flt);
503  				return -ENOMEM;
504  			}
505  			flt->high_priority = high_priority;
506  			flt->low_priority = low_priority;
507  			flt->new_value = t;
508  			qb_list_add_tail(&flt->list, list_head);
509  		} else if (c == QB_LOG_FILTER_REMOVE || c == QB_LOG_TAG_CLEAR) {
510  			qb_list_for_each_safe(iter, next, list_head) {
511  				flt = qb_list_entry(iter, struct qb_log_filter, list);
512  				if (flt->type == type &&
513  				    flt->low_priority <= low_priority &&
514  				    flt->high_priority >= high_priority &&
515  				    (strcmp(flt->text, text) == 0 ||
516  				     strcmp("*", text) == 0)) {
517  					qb_list_del(iter);
518  					free(flt->text);
519  					free(flt);
520  					return 0;
521  				}
522  			}
523  	
524  		} else if (c == QB_LOG_FILTER_CLEAR_ALL || c == QB_LOG_TAG_CLEAR_ALL) {
525  			qb_list_for_each_safe(iter, next, list_head) {
526  				flt = qb_list_entry(iter, struct qb_log_filter, list);
527  				qb_list_del(iter);
528  				free(flt->text);
529  				free(flt);
530  			}
531  		}
532  		return 0;
533  	}

535  	static void
536  	_log_filter_apply(struct callsite_section *sect,
537  			  uint32_t t, enum qb_log_filter_conf c,
538  			  enum qb_log_filter_type type,
539  			  const char *text,
540  			  uint8_t high_priority, uint8_t low_priority)
541  	{
542  		struct qb_log_callsite *cs;
543  	
544  		for (cs = sect->start; cs < sect->stop; cs++) {
545  			if (cs->lineno > 0) {
546  				_log_filter_apply_to_cs(cs, t, c, type, text,
547  						    high_priority, low_priority);
548  			}
549  		}
550  	}

552  	/* #define _QB_FILTER_DEBUGGING_ 1 */
553  	static void
554  	_log_filter_apply_to_cs(struct qb_log_callsite *cs,
555  				uint32_t t, enum qb_log_filter_conf c,
556  				enum qb_log_filter_type type,
557  				const char *text,
558  				uint8_t high_priority, uint8_t low_priority)
559  	{
560  	
561  		if (c == QB_LOG_FILTER_CLEAR_ALL) {
562  			qb_bit_clear(cs->targets, t);
563  			return;
564  		} else if (c == QB_LOG_TAG_CLEAR_ALL) {
565  			cs->tags = 0;
566  			return;
567  		}
568  	
569  		if (_cs_matches_filter_(cs, type, text, high_priority, low_priority)) {
570  	#ifdef _QB_FILTER_DEBUGGING_
571  			uint32_t old_targets = cs->targets;
572  			uint32_t old_tags = cs->tags;
573  	#endif /* _QB_FILTER_DEBUGGING_ */
574  			if (c == QB_LOG_FILTER_ADD) {
575  				qb_bit_set(cs->targets, t);
576  			} else if (c == QB_LOG_FILTER_REMOVE) {
577  				qb_bit_clear(cs->targets, t);
578  			} else if (c == QB_LOG_TAG_SET) {
579  				cs->tags = t;
580  			} else if (c == QB_LOG_TAG_CLEAR) {
581  				cs->tags = 0;
582  			}
583  	#ifdef _QB_FILTER_DEBUGGING_
584  			if (old_targets != cs->targets) {
585  				printf("targets: %s:%u value(%d) %d -> %d\n",
586  				       cs->filename, cs->lineno, t,
587  				       old_targets, cs->targets);
588  			}
589  			if (old_tags != cs->tags) {
590  				printf("tags: %s:%u value(%d) %d -> %d\n",
591  				       cs->filename, cs->lineno, t, old_tags, cs->tags);
592  			}
593  	#endif /* _QB_FILTER_DEBUGGING_ */
594  		}
595  	}

597  	int32_t
598  	qb_log_filter_ctl2(int32_t t, enum qb_log_filter_conf c,
599  			   enum qb_log_filter_type type, const char * text,
600  			   uint8_t high_priority, uint8_t low_priority)
601  	{
602  		struct callsite_section *sect;
603  		int32_t rc;
604  	
605  		if (!logger_inited) {
606  			return -EINVAL;
607  		}
608  	
609  		if (c == QB_LOG_FILTER_ADD ||
610  		    c == QB_LOG_FILTER_CLEAR_ALL ||
611  		    c == QB_LOG_FILTER_REMOVE) {
612  			if (t < 0 || t >= QB_LOG_TARGET_MAX ||
613  			    conf[t].state == QB_LOG_STATE_UNUSED) {
614  				return -EBADF;
615  			}
616  		}
617  	
618  		if (text == NULL ||
619  		    low_priority < high_priority ||
620  		    type > QB_LOG_FILTER_FORMAT ||
621  		    c > QB_LOG_TAG_CLEAR_ALL) {
622  			return -EINVAL;
623  		}
624  		pthread_rwlock_rdlock(&_listlock);
625  		rc = _log_filter_store(t, c, type, text, high_priority, low_priority);
626  		if (rc < 0) {
627  			pthread_rwlock_unlock(&_listlock);
628  			return rc;
629  		}
630  	
631  		qb_list_for_each_entry(sect, &callsite_sections, list) {
632  			_log_filter_apply(sect, t, c, type, text, high_priority, low_priority);
633  		}
634  		pthread_rwlock_unlock(&_listlock);
635  		return 0;
636  	}

638  	int32_t
639  	qb_log_filter_fn_set(qb_log_filter_fn fn)
640  	{
641  		struct callsite_section *sect;
642  		struct qb_log_callsite *cs;
643  	
644  		if (!logger_inited) {
645  			return -EINVAL;
646  		}
647  		_custom_filter_fn = fn;
648  		if (_custom_filter_fn == NULL) {
649  			return 0;
650  		}
651  	
652  		qb_list_for_each_entry(sect, &callsite_sections, list) {
653  			for (cs = sect->start; cs < sect->stop; cs++) {
654  				if (cs->lineno > 0) {
655  					_custom_filter_fn(cs);
656  				}
657  			}
658  		}
659  		return 0;
660  	}

671  	static int32_t
672  	_log_so_walk_callback(struct dl_phdr_info *info, size_t size, void *data)
673  	{
674  		if (strlen(info->dlpi_name) > 0) {
675  			void *handle;
676  			void *start;
677  			void *stop;
678  			const char *error;
679  	
680  			handle = dlopen(info->dlpi_name, RTLD_LAZY);
681  			error = dlerror();
682  			if (!handle || error) {
683  				qb_log(LOG_ERR, "%s", error);
684  				if (handle) {
685  					dlclose(handle);
686  				}
687  				return 0;
688  			}
689  	
690  			start = dlsym(handle, "__start___verbose");
691  			error = dlerror();
692  			if (error) {
693  				goto done;
694  			}
695  	
696  			stop = dlsym(handle, "__stop___verbose");
697  			error = dlerror();
698  			if (error) {
699  				goto done;
700  	
701  			} else {
702  				qb_log_callsites_register(start, stop);
703  			}
704  	done:
705  			dlclose(handle);
706  		}
707  		return 0;
708  	}

711  	static void
712  	_log_target_state_set(struct qb_log_target *t, enum qb_log_target_state s)
713  	{
714  		int32_t i;
715  		int32_t a_set = QB_FALSE;
716  		int32_t u_set = QB_FALSE;
717  	
718  		t->state = s;
719  	
720  		for (i = 31; i >= 0; i--) {
721  			if (!a_set && conf[i].state == QB_LOG_STATE_ENABLED) {
722  				a_set = QB_TRUE;
723  				conf_active_max = i;
724  			}
725  			if (!u_set && conf[i].state != QB_LOG_STATE_UNUSED) {
726  				u_set = QB_TRUE;
727  			}
728  		}
729  	}

731  	void
732  	qb_log_init(const char *name, int32_t facility, uint8_t priority)
733  	{
734  		int32_t i;
735  	
736  		i = pthread_rwlock_init(&_listlock, NULL);
737  		assert(i == 0);
738  		qb_log_format_init();
739  	
740  		for (i = 0; i < QB_LOG_TARGET_MAX; i++) {
741  			conf[i].pos = i;
742  			conf[i].debug = QB_FALSE;
743  			conf[i].file_sync = QB_FALSE;
744  			conf[i].state = QB_LOG_STATE_UNUSED;
745  			(void)strlcpy(conf[i].name, name, PATH_MAX);
746  			conf[i].facility = facility;
747  			qb_list_init(&conf[i].filter_head);
748  		}
749  	
750  		qb_log_dcs_init();
751  	#ifdef QB_HAVE_ATTRIBUTE_SECTION
752  		qb_log_callsites_register(__start___verbose, __stop___verbose);
753  		dl_iterate_phdr(_log_so_walk_callback, NULL);
754  	#endif /* QB_HAVE_ATTRIBUTE_SECTION */
755  	
756  		conf[QB_LOG_STDERR].state = QB_LOG_STATE_DISABLED;
757  		conf[QB_LOG_BLACKBOX].state = QB_LOG_STATE_DISABLED;
758  		conf[QB_LOG_STDOUT].state = QB_LOG_STATE_DISABLED;
759  	
760  		logger_inited = QB_TRUE;
761  		(void)qb_log_syslog_open(&conf[QB_LOG_SYSLOG]);
762  		_log_target_state_set(&conf[QB_LOG_SYSLOG], QB_LOG_STATE_ENABLED);
763  		(void)qb_log_filter_ctl(QB_LOG_SYSLOG, QB_LOG_FILTER_ADD,
764  					QB_LOG_FILTER_FILE, "*", priority);
765  	}

767  	void
768  	qb_log_fini(void)
769  	{
770  		struct qb_log_target *t;
771  		struct qb_log_filter *flt;
772  		struct callsite_section *s;
773  		struct qb_list_head *iter;
774  		struct qb_list_head *iter2;
775  		struct qb_list_head *next;
776  		struct qb_list_head *next2;
777  		int32_t pos;
778  	
779  		if (!logger_inited) {
780  			return;
781  		}
782  		logger_inited = QB_FALSE;
783  		qb_log_thread_stop();
784  		pthread_rwlock_destroy(&_listlock);
785  	
786  		for (pos = 0; pos <= conf_active_max; pos++) {
787  			t = &conf[pos];
788  			_log_target_disable(t);
789  			qb_list_for_each_safe(iter2, next2, &t->filter_head) {
790  				flt = qb_list_entry(iter2, struct qb_log_filter, list);
791  				qb_list_del(iter2);
792  				free(flt->text);
793  				free(flt);
794  			}
795  		}
796  		qb_log_format_fini();
797  		qb_log_dcs_fini();
798  		qb_list_for_each_safe(iter, next, &callsite_sections) {
799  			s = qb_list_entry(iter, struct callsite_section, list);
800  			qb_list_del(iter);
801  			free(s);
802  		}
803  		qb_list_for_each_safe(iter, next, &tags_head) {
804  			flt = qb_list_entry(iter, struct qb_log_filter, list);
805  			qb_list_del(iter);
806  			free(flt->text);
807  			free(flt);
808  		}
809  	}

914  	static int32_t
915  	_log_target_enable(struct qb_log_target *t)
916  	{
917  		int32_t rc = 0;
918  	
919  		if (t->state == QB_LOG_STATE_ENABLED) {
920  			return 0;
921  		}
922  		if (t->pos == QB_LOG_STDERR ||
923  		    t->pos == QB_LOG_STDOUT) {
924  			rc = qb_log_stderr_open(t);
925  		} else if (t->pos == QB_LOG_SYSLOG) {
926  			rc = qb_log_syslog_open(t);
927  		} else if (t->pos == QB_LOG_BLACKBOX) {
928  			rc = qb_log_blackbox_open(t);
929  		}
930  		if (rc == 0) {
931  			_log_target_state_set(t, QB_LOG_STATE_ENABLED);
932  		}
933  		return rc;
934  	}

936  	static void
937  	_log_target_disable(struct qb_log_target *t)
938  	{
939  		if (t->state != QB_LOG_STATE_ENABLED) {
940  			return;
941  		}
942  		_log_target_state_set(t, QB_LOG_STATE_DISABLED);
943  		if (t->close) {
944  			in_logger = QB_TRUE;
945  			t->close(t->pos);
946  			in_logger = QB_FALSE;
947  		}
948  	}

950  	int32_t
951  	qb_log_ctl(int32_t t, enum qb_log_conf c, int32_t arg)
952  	{
953  		int32_t rc = 0;
954  		int32_t need_reload = QB_FALSE;
955  	
956  		if (!logger_inited) {
957  			return -EINVAL;
958  		}
959  		if (t < 0 || t >= QB_LOG_TARGET_MAX ||
960  		    conf[t].state == QB_LOG_STATE_UNUSED) {
961  			return -EBADF;
962  		}
963  		switch (c) {
964  		case QB_LOG_CONF_ENABLED:
965  			if (arg) {
966  				rc = _log_target_enable(&conf[t]);
967  			} else {
968  				_log_target_disable(&conf[t]);
969  			}
970  			break;
971  		case QB_LOG_CONF_STATE_GET:
972  			rc = conf[t].state;
973  			break;
974  		case QB_LOG_CONF_FACILITY:
975  			conf[t].facility = arg;
976  			if (t == QB_LOG_SYSLOG) {
977  				need_reload = QB_TRUE;
978  			}
979  			break;
980  		case QB_LOG_CONF_FILE_SYNC:
981  			conf[t].file_sync = arg;
982  			break;
983  		case QB_LOG_CONF_PRIORITY_BUMP:
984  			conf[t].priority_bump = arg;
985  			break;
986  		case QB_LOG_CONF_SIZE:
987  			if (t == QB_LOG_BLACKBOX) {
988  				if (arg <= 0) {
989  					return -EINVAL;
990  				}
991  				conf[t].size = arg;
992  				need_reload = QB_TRUE;
993  			} else {
994  				return -ENOSYS;
995  			}
996  			break;
997  		case QB_LOG_CONF_THREADED:
998  			conf[t].threaded = arg;
999  			break;
1000 	
1001 		default:
1002 			rc = -EINVAL;
1003 		}
1004 		if (rc == 0 && need_reload && conf[t].reload) {
1005 			in_logger = QB_TRUE;
1006 			conf[t].reload(t);
1007 			in_logger = QB_FALSE;
1008 		}
1009 		return rc;
1010 	}
