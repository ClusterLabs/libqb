57   	static void *qb_logt_worker_thread(void *data) __attribute__ ((noreturn));
58   	static void *
59   	qb_logt_worker_thread(void *data)
60   	{
61   		struct qb_log_record *rec;
62   		int dropped = 0;
63   		int res;
64   	
65   		/*
66   		 * Signal wthread_create that the initialization process may continue
67   		 */
68   		sem_post(&logt_thread_start);
69   		for (;;) {
70   	retry_sem_wait:
71   			res = sem_wait(&logt_print_finished);
72   			if (res == -1 && errno == EINTR) {
73   				goto retry_sem_wait;
74   			} else if (res == -1) {
75   				/*
76   				 * This case shouldn't happen
77   				 */
78   				pthread_exit(NULL);
79   			}
80   	
81   			(void)qb_thread_lock(logt_wthread_lock);
82   			if (wthread_should_exit) {
83   				int value = -1;
84   	
85   				(void)sem_getvalue(&logt_print_finished, &value);
86   				if (value == 0) {
87   					(void)qb_thread_unlock(logt_wthread_lock);
88   					pthread_exit(NULL);
89   				}
90   			}
91   	
92   			rec =
93   			    qb_list_first_entry(&logt_print_finished_records,
94   					  struct qb_log_record, list);
95   			qb_list_del(&rec->list);
96   			logt_memory_used = logt_memory_used - strlen(rec->buffer) -
97   			    sizeof(struct qb_log_record) - 1;
98   			dropped = logt_dropped_messages;
99   			logt_dropped_messages = 0;
100  			(void)qb_thread_unlock(logt_wthread_lock);
101  			if (dropped) {
102  				printf("%d messages lost\n", dropped);
103  			}
104  	
105  			qb_log_thread_log_write(rec->cs, rec->timestamp, rec->buffer);
106  			free(rec->buffer);
107  			free(rec);
108  		}
109  	}

111  	int32_t
112  	qb_log_thread_priority_set(int32_t policy, int32_t priority)
113  	{
114  		int res = 0;
115  	
116  	#if defined(HAVE_PTHREAD_SETSCHEDPARAM) && defined(HAVE_SCHED_GET_PRIORITY_MAX)
117  	
118  		logt_sched_policy = policy;
119  	
120  		if (policy == SCHED_OTHER
121  	#ifdef SCHED_IDLE
122  		    || policy == SCHED_IDLE
123  	#endif
124  	#if defined(SCHED_BATCH) && !defined(QB_DARWIN)
125  		    || policy == SCHED_BATCH
126  	#endif
127  		    ) {
128  			logt_sched_param.sched_priority = 0;
129  		} else {
130  			logt_sched_param.sched_priority = priority;
131  		}
132  		if (wthread_active == QB_FALSE) {
133  			logt_sched_param_queued = QB_TRUE;
134  		} else {
135  			res = pthread_setschedparam(logt_thread_id, policy,
136  						    &logt_sched_param);
137  			if (res != 0) {
138  				res = -res;
139  			}
140  		}
141  	#endif
142  		return res;
143  	}

145  	int32_t
146  	qb_log_thread_start(void)
147  	{
148  		int res;
149  	
150  		if (wthread_active) {
151  			return 0;
152  		}
153  	
154  		wthread_active = QB_TRUE;
155  		sem_init(&logt_thread_start, 0, 0);
156  		sem_init(&logt_print_finished, 0, 0);
157  		res = pthread_create(&logt_thread_id, NULL,
158  				     qb_logt_worker_thread, NULL);
159  		if (res != 0) {
160  			wthread_active = QB_FALSE;
161  			return -res;
162  		}
163  		sem_wait(&logt_thread_start);
164  	
165  		if (logt_sched_param_queued) {
166  			res = qb_log_thread_priority_set(logt_sched_policy,
167  			                                 logt_sched_param.sched_priority);
168  			if (res != 0) {
169  				goto cleanup_pthread;
170  			}
171  			logt_sched_param_queued = QB_FALSE;
172  		}
173  		logt_wthread_lock = qb_thread_lock_create(QB_THREAD_LOCK_SHORT);
174  		if (logt_wthread_lock == NULL) {
175  			goto cleanup_pthread;
176  		}
177  	
178  		return 0;
179  	
180  	cleanup_pthread:
181  		wthread_should_exit = QB_TRUE;
182  		sem_post(&logt_print_finished);
183  		pthread_join(logt_thread_id, NULL);
184  		sem_destroy(&logt_print_finished);
185  		sem_destroy(&logt_thread_start);
186  	
187  		return res;
188  	}

238  	void
239  	qb_log_thread_stop(void)
240  	{
241  		int res;
242  		int value;
243  		struct qb_log_record *rec;
244  	
245  		if (wthread_active == QB_FALSE && logt_wthread_lock == NULL) {
246  			return;
247  		}
248  		if (wthread_active == QB_FALSE) {
249  			for (;;) {
250  				res = sem_getvalue(&logt_print_finished, &value);
251  				if (res != 0 || value == 0) {
252  					return;
253  				}
254  				sem_wait(&logt_print_finished);
255  	
256  				(void)qb_thread_lock(logt_wthread_lock);
257  	
258  				rec = qb_list_first_entry(&logt_print_finished_records,
259  						    struct qb_log_record, list);
260  				qb_list_del(&rec->list);
261  				logt_memory_used = logt_memory_used -
262  						   strlen(rec->buffer) -
263  						   sizeof(struct qb_log_record) - 1;
264  				(void)qb_thread_unlock(logt_wthread_lock);
265  	
266  				qb_log_thread_log_write(rec->cs, rec->timestamp,
267  							rec->buffer);
268  				free(rec->buffer);
269  				free(rec);
270  			}
271  		} else {
272  			wthread_should_exit = QB_TRUE;
273  			sem_post(&logt_print_finished);
274  			pthread_join(logt_thread_id, NULL);
275  		}
276  		(void)qb_thread_lock_destroy(logt_wthread_lock);
277  		sem_destroy(&logt_print_finished);
278  		sem_destroy(&logt_thread_start);
279  	}
