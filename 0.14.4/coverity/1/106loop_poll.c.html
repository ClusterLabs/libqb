52   	static void
53   	_poll_entry_check_generate_(struct qb_poll_entry *pe)
54   	{
55   		int32_t i;
56   	
57   		for (i = 0; i < 200; i++) {
58   			pe->check = random();
59   	
60   			if (pe->check != 0 && pe->check != 0xffffffff) {
61   				break;
62   			}
63   		}
64   	}

66   	static void
67   	_poll_entry_mark_deleted_(struct qb_poll_entry *pe)
68   	{
69   		pe->ufd.fd = -1;
70   		pe->state = QB_POLL_ENTRY_DELETED;
71   		pe->check = 0;
72   	}

74   	static void
75   	_poll_entry_empty_(struct qb_poll_entry *pe)
76   	{
77   		memset(pe, 0, sizeof(struct qb_poll_entry));
78   		pe->ufd.fd = -1;
79   	}

81   	static void
82   	_poll_dispatch_and_take_back_(struct qb_loop_item *item,
83   				      enum qb_loop_priority p)
84   	{
85   		struct qb_poll_entry *pe = (struct qb_poll_entry *)item;
86   		int32_t res;
87   	#ifdef DEBUG_DISPATCH_TIME
88   		uint64_t start;
89   		uint64_t stop;
90   		int32_t log_warn = QB_FALSE;
91   	
92   		start = qb_util_nano_current_get();
93   	#endif /* DEBUG_DISPATCH_TIME */
94   	
95   		assert(pe->state == QB_POLL_ENTRY_JOBLIST);
96   		assert(pe->item.type == QB_LOOP_FD);
97   	
98   		res = pe->poll_dispatch_fn(pe->ufd.fd,
99   					   pe->ufd.revents,
100  					   pe->item.user_data);
101  		if (res < 0) {
102  			_poll_entry_mark_deleted_(pe);
103  		} else {
104  			pe->state = QB_POLL_ENTRY_ACTIVE;
105  			pe->ufd.revents = 0;
106  		}
107  	#ifdef DEBUG_DISPATCH_TIME
108  		if (pe->state == QB_POLL_ENTRY_ACTIVE) {
109  			pe->runs++;
110  			if ((pe->runs % 50) == 0) {
111  				log_warn = QB_TRUE;
112  			}
113  			stop = qb_util_nano_current_get();
114  			if ((stop - start) > (10 * QB_TIME_NS_IN_MSEC)) {
115  				log_warn = QB_TRUE;
116  			}
117  	
118  			if (log_warn && pe->item.type == QB_LOOP_FD) {
119  				qb_util_log(LOG_INFO,
120  					    "[fd:%d] dispatch:%p runs:%d duration:%d ms",
121  					    pe->ufd.fd, pe->poll_dispatch_fn,
122  					    pe->runs,
123  					    (int32_t) ((stop -
124  							start) / QB_TIME_NS_IN_MSEC));
125  			}
126  		}
127  	#endif /* DEBUG_DISPATCH_TIME */
128  	}

130  	void
131  	qb_poll_fds_usage_check_(struct qb_poll_source *s)
132  	{
133  		struct rlimit lim;
134  		static int32_t socks_limit = 0;
135  		int32_t send_event = QB_FALSE;
136  		int32_t socks_used = 0;
137  		int32_t socks_avail = 0;
138  		struct qb_poll_entry *pe;
139  		int32_t i;
140  	
141  		if (socks_limit == 0) {
142  			if (getrlimit(RLIMIT_NOFILE, &lim) == -1) {
(1) Event tainted_string_warning: Using a non string-literal such as "descriptor.format" as a format-string argument can lead to a vulnerability if a user can control its content. [details] 
143  				qb_util_perror(LOG_WARNING, "getrlimit");
144  				return;
145  			}
146  			socks_limit = lim.rlim_cur;
147  			socks_limit -= POLL_FDS_USED_MISC;
148  			if (socks_limit < 0) {
149  				socks_limit = 0;
150  			}
151  		}
152  	
153  		for (i = 0; i < s->poll_entry_count; i++) {
154  			assert(qb_array_index(s->poll_entries, i, (void **)&pe) == 0);
155  			if ((pe->state == QB_POLL_ENTRY_ACTIVE ||
156  			     pe->state == QB_POLL_ENTRY_JOBLIST) && pe->ufd.fd != -1) {
157  				socks_used++;
158  			}
159  			if (pe->state == QB_POLL_ENTRY_DELETED) {
160  				_poll_entry_empty_(pe);
161  			}
162  		}
163  	
164  		socks_avail = socks_limit - socks_used;
165  		if (socks_avail < 0) {
166  			socks_avail = 0;
167  		}
168  		send_event = QB_FALSE;
169  		if (s->not_enough_fds) {
170  			if (socks_avail > 2) {
171  				s->not_enough_fds = QB_FALSE;
172  				send_event = QB_TRUE;
173  			}
174  		} else {
175  			if (socks_avail <= 1) {
176  				s->not_enough_fds = QB_TRUE;
177  				send_event = QB_TRUE;
178  			}
179  		}
180  		if (send_event && s->low_fds_event_fn) {
181  			s->low_fds_event_fn(s->not_enough_fds, socks_avail);
182  		}
183  	}

186  	struct qb_loop_source *
187  	qb_loop_poll_create(struct qb_loop *l)
188  	{
189  		struct qb_poll_source *s = malloc(sizeof(struct qb_poll_source));
190  		if (s == NULL) {
191  			return NULL;
192  		}
193  		s->s.l = l;
194  		s->s.dispatch_and_take_back = _poll_dispatch_and_take_back_;
195  	
196  		s->poll_entries = qb_array_create_2(16, sizeof(struct qb_poll_entry), 16);
197  		s->poll_entry_count = 0;
198  		s->low_fds_event_fn = NULL;
199  		s->not_enough_fds = QB_FALSE;
200  	
201  	#ifdef USE_EPOLL
202  		(void)qb_epoll_init(s);
203  	#endif
204  	#ifdef USE_KQUEUE
205  		(void)qb_kqueue_init(s);
206  	#endif
207  	#ifdef USE_POLL
208  		(void)qb_poll_init(s);
209  	#endif /* USE_POLL */
210  	
211  		return (struct qb_loop_source *)s;
212  	}

235  	static int32_t
236  	_get_empty_array_position_(struct qb_poll_source *s)
237  	{
238  		int32_t found = QB_FALSE;
239  		uint32_t install_pos;
240  		int32_t res = 0;
241  		struct qb_poll_entry *pe;
242  	
243  		for (install_pos = 0;
244  		     install_pos < s->poll_entry_count; install_pos++) {
245  			assert(qb_array_index
246  			       (s->poll_entries, install_pos, (void **)&pe) == 0);
247  			if (pe->state == QB_POLL_ENTRY_EMPTY) {
248  				found = QB_TRUE;
249  				break;
250  			}
251  		}
252  	
253  		if (found == QB_FALSE) {
254  	#ifdef USE_POLL
255  			struct pollfd *ufds;
256  			int32_t new_size = (s->poll_entry_count + 1) * sizeof(struct pollfd);
257  			ufds = realloc(s->ufds, new_size);
258  			if (ufds == NULL) {
259  				return -ENOMEM;
260  			}
261  			s->ufds = ufds;
262  	#endif /* USE_POLL */
263  			/*
264  			 * Grow pollfd list
265  			 */
266  			res = qb_array_grow(s->poll_entries, s->poll_entry_count + 1);
267  			if (res != 0) {
268  				return res;
269  			}
270  	
271  			s->poll_entry_count += 1;
272  			install_pos = s->poll_entry_count - 1;
273  		}
274  		return install_pos;
275  	}

277  	static int32_t
278  	_poll_add_(struct qb_loop *l,
279  		   enum qb_loop_priority p,
280  		   int32_t fd, int32_t events, void *data, struct qb_poll_entry **pe_pt)
281  	{
282  		struct qb_poll_entry *pe;
283  		uint32_t install_pos;
284  		int32_t res = 0;
285  		struct qb_poll_source *s;
286  	
287  		if (l == NULL) {
288  			return -EINVAL;
289  		}
290  	
291  		s = (struct qb_poll_source *)l->fd_source;
292  	
293  		install_pos = _get_empty_array_position_(s);
294  	
295  		assert(qb_array_index(s->poll_entries, install_pos, (void **)&pe) == 0);
296  		pe->state = QB_POLL_ENTRY_ACTIVE;
297  		pe->install_pos = install_pos;
298  		_poll_entry_check_generate_(pe);
299  		pe->ufd.fd = fd;
300  		pe->ufd.events = events;
301  		pe->ufd.revents = 0;
302  		pe->item.user_data = data;
303  		pe->item.source = (struct qb_loop_source *)l->fd_source;
304  		pe->p = p;
305  		pe->runs = 0;
306  		res = s->driver.add(s, pe, fd, events);
307  		if (res == 0) {
308  			*pe_pt = pe;
309  			return 0;
310  		} else {
311  			pe->state = QB_POLL_ENTRY_EMPTY;
312  			return res;
313  		}
314  	}

316  	static int32_t
317  	_qb_poll_add_to_jobs_(struct qb_loop *l, struct qb_poll_entry *pe)
318  	{
319  		assert(pe->item.type == QB_LOOP_FD);
320  		qb_loop_level_item_add(&l->level[pe->p], &pe->item);
321  		pe->state = QB_POLL_ENTRY_JOBLIST;
322  		return 1;
323  	}

325  	int32_t
326  	qb_loop_poll_add(struct qb_loop * lp,
327  			 enum qb_loop_priority p,
328  			 int32_t fd,
329  			 int32_t events,
330  			 void *data, qb_loop_poll_dispatch_fn dispatch_fn)
331  	{
332  		struct qb_poll_entry *pe = NULL;
333  		int32_t size;
334  		int32_t new_size;
335  		int32_t res;
336  		struct qb_loop *l = lp;
337  	
338  		if (l == NULL) {
339  			l = qb_loop_default_get();
340  		}
341  	
342  		size = ((struct qb_poll_source *)l->fd_source)->poll_entry_count;
343  		res = _poll_add_(l, p, fd, events, data, &pe);
344  		if (res != 0) {
345  			qb_util_perror(LOG_ERR,
346  				       "couldn't add poll entryfor FD %d", fd);
347  			return res;
348  		}
349  		new_size = ((struct qb_poll_source *)l->fd_source)->poll_entry_count;
350  	
351  		pe->poll_dispatch_fn = dispatch_fn;
352  		pe->item.type = QB_LOOP_FD;
353  		pe->add_to_jobs = _qb_poll_add_to_jobs_;
354  	
355  		if (new_size > size) {
356  			qb_util_log(LOG_TRACE,
357  				    "grown poll array to %d for FD %d", new_size, fd);
358  		}
359  	
360  		return res;
361  	}

363  	int32_t
364  	qb_loop_poll_mod(struct qb_loop * lp,
365  			 enum qb_loop_priority p,
366  			 int32_t fd,
367  			 int32_t events,
368  			 void *data, qb_loop_poll_dispatch_fn dispatch_fn)
369  	{
370  		uint32_t i;
371  		int32_t res = 0;
372  		struct qb_poll_entry *pe;
373  		struct qb_poll_source *s;
374  		struct qb_loop *l = lp;
375  	
376  		if (l == NULL) {
377  			l = qb_loop_default_get();
378  		}
379  		s = (struct qb_poll_source *)l->fd_source;
380  	
381  		/*
382  		 * Find file descriptor to modify events and dispatch function
383  		 */
384  		for (i = 0; i < s->poll_entry_count; i++) {
385  			assert(qb_array_index(s->poll_entries, i, (void **)&pe) == 0);
386  			if (pe->ufd.fd != fd) {
387  				continue;
388  			}
389  			if (pe->state == QB_POLL_ENTRY_DELETED || pe->check == 0) {
390  				qb_util_log(LOG_ERR,
391  					    "poll_mod : can't modify entry already deleted");
392  				return -EBADF;
393  			}
394  			pe->poll_dispatch_fn = dispatch_fn;
395  			pe->item.user_data = data;
396  			pe->p = p;
397  			if (pe->ufd.events != events) {
398  				res = s->driver.mod(s, pe, fd, events);
399  				pe->ufd.events = events;
400  			}
401  			return res;
402  		}
403  	
404  		return -EBADF;
405  	}

407  	int32_t
408  	qb_loop_poll_del(struct qb_loop * lp, int32_t fd)
409  	{
410  		int32_t i;
411  		int32_t res = 0;
412  		struct qb_poll_entry *pe;
413  		struct qb_poll_source *s;
414  		struct qb_loop *l = lp;
415  	
416  		if (l == NULL) {
417  			l = qb_loop_default_get();
418  		}
419  		s = (struct qb_poll_source *)l->fd_source;
420  		for (i = 0; i < s->poll_entry_count; i++) {
421  			assert(qb_array_index(s->poll_entries, i, (void **)&pe) == 0);
422  			if (pe->ufd.fd != fd || pe->item.type != QB_LOOP_FD) {
423  				continue;
424  			}
425  			if (pe->state == QB_POLL_ENTRY_DELETED ||
426  			    pe->state == QB_POLL_ENTRY_EMPTY) {
427  				return 0;
428  			}
429  			if (pe->state == QB_POLL_ENTRY_JOBLIST) {
430  				qb_loop_level_item_del(&l->level[pe->p], &pe->item);
431  			}
432  			res = s->driver.del(s, pe, fd, i);
433  			_poll_entry_mark_deleted_(pe);
434  			return res;
435  		}
436  	
437  		return -EBADF;
438  	}

456  	static void
457  	_handle_real_signal_(int signal_num, siginfo_t * si, void *context)
458  	{
459  		int32_t sig = signal_num;
460  		int32_t res = 0;
461  	
462  		if (pipe_fds[1] > 0) {
463  	try_again:
464  			res = write(pipe_fds[1], &sig, sizeof(int32_t));
465  			if (res == -1 && errno == EAGAIN) {
466  				goto try_again;
467  			} else if (res != sizeof(int32_t)) {
468  				qb_util_log(LOG_ERR,
469  					    "failed to write signal to pipe [%d]", res);
470  			}
471  		}
472  		qb_util_log(LOG_TRACE, "got real signal [%d] sent to pipe", sig);
473  	}

475  	static void
476  	_signal_dispatch_and_take_back_(struct qb_loop_item *item,
477  					enum qb_loop_priority p)
478  	{
479  		struct qb_loop_sig *sig = (struct qb_loop_sig *)item;
480  		int32_t res;
481  	
482  		res = sig->dispatch_fn(sig->signal, sig->item.user_data);
483  		if (res != 0) {
484  			(void)qb_loop_signal_del(sig->cloned_from->item.source->l,
485  						 sig->cloned_from);
486  		}
487  		free(sig);
488  	}

490  	struct qb_loop_source *
491  	qb_loop_signals_create(struct qb_loop *l)
492  	{
493  		int32_t res = 0;
494  		struct qb_poll_entry *pe;
495  		struct qb_signal_source *s = calloc(1, sizeof(struct qb_signal_source));
496  	
497  		if (s == NULL) {
498  			return NULL;
499  		}
500  		s->s.l = l;
501  		s->s.dispatch_and_take_back = _signal_dispatch_and_take_back_;
502  		s->s.poll = NULL;
503  		qb_list_init(&s->sig_head);
504  		sigemptyset(&s->signal_superset);
505  	
506  		if (pipe_fds[0] < 0) {
507  			res = pipe(pipe_fds);
508  			if (res == -1) {
509  				res = -errno;
510  				qb_util_perror(LOG_ERR, "Can't light pipe");
511  				goto error_exit;
512  			}
513  			(void)qb_sys_fd_nonblock_cloexec_set(pipe_fds[0]);
514  			(void)qb_sys_fd_nonblock_cloexec_set(pipe_fds[1]);
515  	
516  			res = _poll_add_(l, QB_LOOP_HIGH,
517  					 pipe_fds[0], POLLIN, NULL, &pe);
518  			if (res == 0) {
519  				pe->poll_dispatch_fn = NULL;
520  				pe->item.type = QB_LOOP_SIG;
521  				pe->add_to_jobs = _qb_signal_add_to_jobs_;
522  			} else {
523  				qb_util_perror(LOG_ERR, "Can't smoke pipe");
524  				goto error_exit;
525  			}
526  		}
527  	
528  		return (struct qb_loop_source *)s;
529  	
530  	error_exit:
531  		errno = -res;
532  		free(s);
533  		if (pipe_fds[0] >= 0) {
534  			close(pipe_fds[0]);
535  		}
536  		if (pipe_fds[1] >= 0) {
537  			close(pipe_fds[1]);
538  		}
539  		return NULL;
540  	}

542  	void
543  	qb_loop_signals_destroy(struct qb_loop *l)
544  	{
545  		struct qb_signal_source *s =
546  		    (struct qb_signal_source *)l->signal_source;
547  		struct qb_list_head *list;
548  		struct qb_list_head *n;
549  		struct qb_loop_item *item;
550  	
551  		close(pipe_fds[0]);
552  		pipe_fds[0] = -1;
553  		close(pipe_fds[1]);
554  		pipe_fds[1] = -1;
555  	
556  		qb_list_for_each_safe(list, n, &s->sig_head) {
557  			item = qb_list_entry(list, struct qb_loop_item, list);
558  			qb_list_del(&item->list);
559  			free(item);
560  		}
561  	
562  		free(l->signal_source);
563  	}

565  	static int32_t
566  	_qb_signal_add_to_jobs_(struct qb_loop *l, struct qb_poll_entry *pe)
567  	{
568  		struct qb_signal_source *s =
569  		    (struct qb_signal_source *)l->signal_source;
570  		struct qb_list_head *list;
571  		struct qb_loop_sig *sig;
572  		struct qb_loop_item *item;
573  		struct qb_loop_sig *new_sig_job;
574  		int32_t the_signal;
575  		ssize_t res;
576  		int32_t jobs_added = 0;
577  	
578  		res = read(pipe_fds[0], &the_signal, sizeof(int32_t));
579  		if (res != sizeof(int32_t)) {
580  			qb_util_perror(LOG_WARNING, "failed to read pipe");
581  			return 0;
582  		}
583  		pe->ufd.revents = 0;
584  	
585  		qb_list_for_each(list, &s->sig_head) {
586  			item = qb_list_entry(list, struct qb_loop_item, list);
587  			sig = (struct qb_loop_sig *)item;
588  			if (sig->signal == the_signal) {
589  				new_sig_job = calloc(1, sizeof(struct qb_loop_sig));
590  				if (new_sig_job == NULL) {
591  					return jobs_added;
592  				}
593  				memcpy(new_sig_job, sig, sizeof(struct qb_loop_sig));
594  	
595  				qb_util_log(LOG_TRACE,
596  					    "adding signal [%d] to job queue %p",
597  					    the_signal, sig);
598  	
599  				new_sig_job->cloned_from = sig;
600  				qb_loop_level_item_add(&l->level[sig->p],
601  						       &new_sig_job->item);
602  				jobs_added++;
603  			}
604  		}
605  		return jobs_added;
606  	}
607  	
608  	static void
609  	_adjust_sigactions_(struct qb_signal_source *s)
610  	{
611  		struct qb_loop_sig *sig;
612  		struct qb_loop_item *item;
613  		struct sigaction sa;
614  		int32_t i;
615  		int32_t needed;
616  	
617  		sa.sa_flags = SA_SIGINFO;
618  		sa.sa_sigaction = _handle_real_signal_;
619  		sigemptyset(&s->signal_superset);
620  		sigemptyset(&sa.sa_mask);
621  	
622  		/* re-set to default */
623  		for (i = 0; i < 30; i++) {
624  			needed = QB_FALSE;
625  			qb_list_for_each_entry(item, &s->sig_head, list) {
626  				sig = (struct qb_loop_sig *)item;
627  				if (i == sig->signal) {
628  					needed = QB_TRUE;
629  					break;
630  				}
631  			}
632  			if (needed) {
633  				sigaddset(&s->signal_superset, i);
634  				sigaction(i, &sa, NULL);
635  			} else {
636  				(void)signal(i, SIG_DFL);
637  			}
638  		}
639  	}

641  	int32_t
642  	qb_loop_signal_add(qb_loop_t * lp,
643  			   enum qb_loop_priority p,
644  			   int32_t the_sig,
645  			   void *data,
646  			   qb_loop_signal_dispatch_fn dispatch_fn,
647  			   qb_loop_signal_handle * handle)
648  	{
649  		struct qb_loop_sig *sig;
650  		struct qb_signal_source *s;
651  		struct qb_loop *l = lp;
652  	
653  		if (l == NULL) {
654  			l = qb_loop_default_get();
655  		}
656  		if (l == NULL || dispatch_fn == NULL) {
657  			return -EINVAL;
658  		}
659  		if (p < QB_LOOP_LOW || p > QB_LOOP_HIGH) {
660  			return -EINVAL;
661  		}
662  		s = (struct qb_signal_source *)l->signal_source;
663  		sig = calloc(1, sizeof(struct qb_loop_sig));
664  		if (sig == NULL) {
665  			return -errno;
666  		}
667  	
668  		sig->dispatch_fn = dispatch_fn;
669  		sig->p = p;
670  		sig->signal = the_sig;
671  		sig->item.user_data = data;
672  		sig->item.source = l->signal_source;
673  		sig->item.type = QB_LOOP_SIG;
674  	
675  		qb_list_init(&sig->item.list);
676  		qb_list_add_tail(&sig->item.list, &s->sig_head);
677  	
678  		if (sigismember(&s->signal_superset, the_sig) != 1) {
679  			_adjust_sigactions_(s);
680  		}
681  		if (handle) {
682  			*handle = sig;
683  		}
684  	
685  		return 0;
686  	}

688  	int32_t
689  	qb_loop_signal_mod(qb_loop_t * lp,
690  			   enum qb_loop_priority p,
691  			   int32_t the_sig,
692  			   void *data,
693  			   qb_loop_signal_dispatch_fn dispatch_fn,
694  			   qb_loop_signal_handle handle)
695  	{
696  		struct qb_signal_source *s;
697  		struct qb_loop_sig *sig = (struct qb_loop_sig *)handle;
698  		struct qb_loop *l = lp;
699  	
700  		if (l == NULL) {
701  			l = qb_loop_default_get();
702  		}
703  		if (l == NULL || dispatch_fn == NULL || handle == NULL) {
704  			return -EINVAL;
705  		}
706  		if (p < QB_LOOP_LOW || p > QB_LOOP_HIGH) {
707  			return -EINVAL;
708  		}
709  		s = (struct qb_signal_source *)l->signal_source;
710  	
711  		sig->item.user_data = data;
712  		sig->item.type = QB_LOOP_SIG;
713  		sig->dispatch_fn = dispatch_fn;
714  		sig->p = p;
715  	
716  		if (sig->signal != the_sig) {
717  			sig->signal = the_sig;
718  			_adjust_sigactions_(s);
719  		}
720  	
721  		return 0;
722  	}

724  	int32_t
725  	qb_loop_signal_del(qb_loop_t * lp, qb_loop_signal_handle handle)
726  	{
727  		struct qb_signal_source *s;
728  		struct qb_loop_sig *sig = (struct qb_loop_sig *)handle;
729  		struct qb_loop_sig *sig_clone;
730  		struct qb_loop *l = lp;
731  		struct qb_loop_item *item;
732  	
733  		if (l == NULL) {
734  			l = qb_loop_default_get();
735  		}
736  		if (l == NULL || handle == NULL) {
737  			return -EINVAL;
738  		}
739  		s = (struct qb_signal_source *)l->signal_source;
740  	
741  		qb_list_for_each_entry(item, &l->level[sig->p].wait_head, list) {
742  			if (item->type != QB_LOOP_SIG) {
743  				continue;
744  			}
745  			sig_clone = (struct qb_loop_sig *)item;
746  			if (sig_clone->cloned_from == sig) {
747  				qb_util_log(LOG_TRACE, "deleting sig in WAITLIST");
748  				qb_list_del(&sig_clone->item.list);
749  				free(sig_clone);
750  				break;
751  			}
752  		}
753  	
754  		qb_list_for_each_entry(item, &l->level[sig->p].job_head, list) {
755  			if (item->type != QB_LOOP_SIG) {
756  				continue;
757  			}
758  			sig_clone = (struct qb_loop_sig *)item;
759  			if (sig_clone->cloned_from == sig) {
760  				qb_loop_level_item_del(&l->level[sig->p], item);
761  				qb_util_log(LOG_TRACE, "deleting sig in JOBLIST");
762  				break;
763  			}
764  		}
765  	
766  		qb_list_del(&sig->item.list);
767  		free(sig);
768  		_adjust_sigactions_(s);
769  		return 0;
770  	}
