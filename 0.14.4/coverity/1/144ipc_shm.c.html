39   	static void
40   	qb_ipcc_shm_disconnect(struct qb_ipcc_connection *c)
41   	{
42   		if (c->is_connected) {
43   			qb_rb_close(c->request.u.shm.rb);
44   			qb_rb_close(c->response.u.shm.rb);
45   			qb_rb_close(c->event.u.shm.rb);
46   		} else {
47   			qb_rb_force_close(c->request.u.shm.rb);
48   			qb_rb_force_close(c->response.u.shm.rb);
49   			qb_rb_force_close(c->event.u.shm.rb);
50   		}
51   	}

53   	static ssize_t
54   	qb_ipc_shm_send(struct qb_ipc_one_way *one_way,
55   			const void *msg_ptr, size_t msg_len)
56   	{
57   		return qb_rb_chunk_write(one_way->u.shm.rb, msg_ptr, msg_len);
58   	}
59   	
60   	static ssize_t
61   	qb_ipc_shm_sendv(struct qb_ipc_one_way *one_way,
62   			 const struct iovec *iov, size_t iov_len)
63   	{
64   		char *dest;
65   		int32_t res = 0;
66   		int32_t total_size = 0;
67   		int32_t i;
68   		char *pt = NULL;
69   	
70   		if (one_way->u.shm.rb == NULL) {
71   			return -ENOTCONN;
72   		}
73   	
74   		for (i = 0; i < iov_len; i++) {
75   			total_size += iov[i].iov_len;
76   		}
77   		dest = qb_rb_chunk_alloc(one_way->u.shm.rb, total_size);
78   		if (dest == NULL) {
79   			return -errno;
80   		}
81   		pt = dest;
82   	
83   		for (i = 0; i < iov_len; i++) {
84   			memcpy(pt, iov[i].iov_base, iov[i].iov_len);
85   			pt += iov[i].iov_len;
86   		}
87   		res = qb_rb_chunk_commit(one_way->u.shm.rb, total_size);
88   		if (res < 0) {
89   			return res;
90   		}
91   		return total_size;
92   	}

94   	static ssize_t
95   	qb_ipc_shm_recv(struct qb_ipc_one_way *one_way,
96   			void *msg_ptr, size_t msg_len, int32_t ms_timeout)
97   	{
98   		if (one_way->u.shm.rb == NULL) {
99   			return -ENOTCONN;
100  		}
101  		return qb_rb_chunk_read(one_way->u.shm.rb,
102  					(void *)msg_ptr, msg_len, ms_timeout);
103  	}
104  	
105  	static ssize_t
106  	qb_ipc_shm_peek(struct qb_ipc_one_way *one_way, void **data_out,
107  			int32_t ms_timeout)
108  	{
109  		ssize_t rc;
110  		if (one_way->u.shm.rb == NULL) {
111  			return -ENOTCONN;
112  		}
113  		rc = qb_rb_chunk_peek(one_way->u.shm.rb, data_out, ms_timeout);
114  		if (rc == 0)  {
115  			return -EAGAIN;
116  		}
117  		return rc;
118  	}

120  	static void
121  	qb_ipc_shm_reclaim(struct qb_ipc_one_way *one_way)
122  	{
123  		if (one_way->u.shm.rb != NULL) {
124  			qb_rb_chunk_reclaim(one_way->u.shm.rb);
125  		}
126  	}

128  	static void
129  	qb_ipc_shm_fc_set(struct qb_ipc_one_way *one_way, int32_t fc_enable)
130  	{
131  		int32_t *fc;
132  		fc = qb_rb_shared_user_data_get(one_way->u.shm.rb);
133  		qb_util_log(LOG_TRACE, "setting fc to %d", fc_enable);
134  		qb_atomic_int_set(fc, fc_enable);
135  	}
136  	
137  	static int32_t
138  	qb_ipc_shm_fc_get(struct qb_ipc_one_way *one_way)
139  	{
140  		int32_t *fc;
141  		int32_t rc = qb_rb_refcount_get(one_way->u.shm.rb);
142  	
143  		if (rc != 2) {
144  			return -ENOTCONN;
145  		}
146  		fc = qb_rb_shared_user_data_get(one_way->u.shm.rb);
147  		return qb_atomic_int_get(fc);
148  	}
149  	
150  	static ssize_t
151  	qb_ipc_shm_q_len_get(struct qb_ipc_one_way *one_way)
152  	{
153  		if (one_way->u.shm.rb == NULL) {
154  			return -ENOTCONN;
155  		}
156  		return qb_rb_chunks_used(one_way->u.shm.rb);
157  	}

224  	static void
225  	qb_ipcs_shm_disconnect(struct qb_ipcs_connection *c)
226  	{
227  		if (c->response.u.shm.rb) {
228  			qb_rb_close(c->response.u.shm.rb);
229  			c->response.u.shm.rb = NULL;
230  		}
231  		if (c->event.u.shm.rb) {
232  			qb_rb_close(c->event.u.shm.rb);
233  			c->event.u.shm.rb = NULL;
234  		}
235  		if (c->request.u.shm.rb) {
236  			qb_rb_close(c->request.u.shm.rb);
237  			c->request.u.shm.rb = NULL;
238  		}
239  	}

241  	static int32_t
242  	qb_ipcs_shm_rb_open(struct qb_ipcs_connection *c,
243  			    struct qb_ipc_one_way *ow,
244  			    const char *rb_name)
245  	{
246  		int32_t res = 0;
247  	
248  		ow->u.shm.rb = qb_rb_open(rb_name,
249  					  ow->max_msg_size,
250  					  QB_RB_FLAG_CREATE |
251  					  QB_RB_FLAG_SHARED_PROCESS,
252  					  sizeof(int32_t));
253  		if (ow->u.shm.rb == NULL) {
254  			res = -errno;
255  			qb_util_perror(LOG_ERR, "qb_rb_open:%s", rb_name);
256  			return res;
257  		}
258  		res = qb_rb_chown(ow->u.shm.rb, c->auth.uid, c->auth.gid);
259  		if (res != 0) {
260  			qb_util_perror(LOG_ERR, "qb_rb_chown:%s", rb_name);
261  			goto cleanup;
262  		}
263  		res = qb_rb_chmod(ow->u.shm.rb, c->auth.mode);
264  		if (res != 0) {
265  			qb_util_perror(LOG_ERR, "qb_rb_chmod:%s", rb_name);
266  			goto cleanup;
267  		}
268  		return res;
269  	
270  	cleanup:
271  		qb_rb_close(ow->u.shm.rb);
272  		return res;
273  	}

275  	static int32_t
276  	qb_ipcs_shm_connect(struct qb_ipcs_service *s,
277  			    struct qb_ipcs_connection *c,
278  			    struct qb_ipc_connection_response *r)
279  	{
280  		int32_t res;
281  	
282  		qb_util_log(LOG_DEBUG, "connecting to client [%d]", c->pid);
283  	
284  		snprintf(r->request, NAME_MAX, "%s-request-%s",
285  			 s->name, c->description);
286  		snprintf(r->response, NAME_MAX, "%s-response-%s",
287  			 s->name, c->description);
288  		snprintf(r->event, NAME_MAX, "%s-event-%s",
289  			 s->name, c->description);
290  	
291  		res = qb_ipcs_shm_rb_open(c, &c->request,
292  					  r->request);
293  		if (res != 0) {
294  			goto cleanup;
295  		}
296  	
297  		res = qb_ipcs_shm_rb_open(c, &c->response,
298  					  r->response);
299  		if (res != 0) {
300  			goto cleanup_request;
301  		}
302  	
303  		res = qb_ipcs_shm_rb_open(c, &c->event,
304  					  r->event);
305  		if (res != 0) {
306  			goto cleanup_request_response;
307  		}
308  	
309  		r->hdr.error = 0;
310  		return 0;
311  	
312  	cleanup_request_response:
313  		qb_rb_close(c->request.u.shm.rb);
314  	
315  	cleanup_request:
316  		qb_rb_close(c->response.u.shm.rb);
317  	
318  	cleanup:
319  		r->hdr.error = res;
320  		errno = -res;
321  		qb_util_perror(LOG_ERR, "shm connection FAILED");
322  	
323  		return res;
324  	}
