27   	static void
28   	_blackbox_reload(int32_t target)
29   	{
30   		struct qb_log_target *t = qb_log_target_get(target);
31   	
32   		if (t->instance == NULL) {
33   			return;
34   		}
35   		qb_rb_close(t->instance);
36   		t->instance = qb_rb_open(t->filename, t->size,
37   					 QB_RB_FLAG_CREATE | QB_RB_FLAG_OVERWRITE, 0);
38   	}
39   	
40   	/* <u32> file lineno
41   	 * <u32> tags
42   	 * <u8> priority
43   	 * <u32> function name length
44   	 * <string> function name
45   	 * <u32> buffer length
46   	 * <string> buffer
47   	 */
48   	static void
49   	_blackbox_vlogger(int32_t target,
50   			  struct qb_log_callsite *cs, time_t timestamp, va_list ap)
51   	{
52   		size_t max_size;
53   		size_t actual_size;
54   		uint32_t fn_size;
55   		char *chunk;
56   		char *msg_len_pt;
57   		uint32_t msg_len;
58   		struct qb_log_target *t = qb_log_target_get(target);
59   	
60   		if (t->instance == NULL) {
61   			return;
62   		}
63   	
64   		fn_size = strlen(cs->function) + 1;
65   	
66   		actual_size = 4 * sizeof(uint32_t) + sizeof(uint8_t) + fn_size + sizeof(time_t);
67   		max_size = actual_size + QB_LOG_MAX_LEN;
68   	
69   		chunk = qb_rb_chunk_alloc(t->instance, max_size);
70   	
71   		/* line number */
72   		memcpy(chunk, &cs->lineno, sizeof(uint32_t));
73   		chunk += sizeof(uint32_t);
74   	
75   		/* tags */
76   		memcpy(chunk, &cs->tags, sizeof(uint32_t));
77   		chunk += sizeof(uint32_t);
78   	
79   		/* log level/priority */
80   		memcpy(chunk, &cs->priority, sizeof(uint8_t));
81   		chunk += sizeof(uint8_t);
82   	
83   		/* function name */
84   		memcpy(chunk, &fn_size, sizeof(uint32_t));
85   		chunk += sizeof(uint32_t);
86   		memcpy(chunk, cs->function, fn_size);
87   		chunk += fn_size;
88   	
89   		/* timestamp */
90   		memcpy(chunk, &timestamp, sizeof(time_t));
91   		chunk += sizeof(time_t);
92   	
93   		/* log message length */
94   		msg_len_pt = chunk;
95   		chunk += sizeof(uint32_t);
96   	
97   		/* log message */
98   		msg_len = qb_vsnprintf_serialize(chunk, QB_LOG_MAX_LEN, cs->format, ap);
99   		if (msg_len >= QB_LOG_MAX_LEN) {
100  		    chunk = msg_len_pt + sizeof(uint32_t); /* Reset */
101  	
102  		    msg_len = qb_vsnprintf_serialize(chunk, QB_LOG_MAX_LEN,
103  			"Log message too long to be stored in the blackbox.  "\
104  			"Maximum is QB_LOG_MAX_LEN" , ap);
105  		    actual_size += msg_len;
106  		}
107  	
108  		actual_size += msg_len;
109  	
110  		/* now that we know the length, write it
111  		 */
112  		memcpy(msg_len_pt, &msg_len, sizeof(uint32_t));
113  	
114  		(void)qb_rb_chunk_commit(t->instance, actual_size);
115  	}
116  	
117  	static void
118  	_blackbox_close(int32_t target)
119  	{
120  		struct qb_log_target *t = qb_log_target_get(target);
121  	
122  		if (t->instance) {
123  			qb_rb_close(t->instance);
124  			t->instance = NULL;
125  		}
126  	}

128  	int32_t
129  	qb_log_blackbox_open(struct qb_log_target *t)
130  	{
131  		if (t->size < 1024) {
132  			return -EINVAL;
133  		}
134  		snprintf(t->filename, PATH_MAX, "%s-blackbox", t->name);
135  	
136  		t->instance = qb_rb_open(t->filename, t->size,
137  					 QB_RB_FLAG_CREATE | QB_RB_FLAG_OVERWRITE, 0);
138  		if (t->instance == NULL) {
139  			return -errno;
140  		}
141  	
142  		t->logger = NULL;
143  		t->vlogger = _blackbox_vlogger;
144  		t->reload = _blackbox_reload;
145  		t->close = _blackbox_close;
146  		return 0;
147  	}

149  	ssize_t
150  	qb_log_blackbox_write_to_file(const char *filename)
151  	{
152  		ssize_t written_size = 0;
153  		struct qb_log_target *t;
154  		int fd = open(filename, O_CREAT | O_RDWR, 0700);
155  	
156  		if (fd < 0) {
157  			return -errno;
158  		}
159  		t = qb_log_target_get(QB_LOG_BLACKBOX);
160  		if (t->instance) {
161  			written_size = qb_rb_write_to_file(t->instance, fd);
162  		} else {
163  			written_size = -ENOENT;
164  		}
165  		close(fd);
166  	
167  		return written_size;
168  	}
