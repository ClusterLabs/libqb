56   	static int32_t
57   	open_mmap_file(char *path, uint32_t file_flags)
58   	{
59   		if (strstr(path, "XXXXXX") != NULL) {
60   			mode_t old_mode = umask(077);
61   			int32_t temp_fd = mkstemp(path);
62   			(void)umask(old_mode);
63   			return temp_fd;
64   		}
65   	
66   		return open(path, file_flags, 0600);
67   	}

69   	int32_t
70   	qb_sys_mmap_file_open(char *path, const char *file, size_t bytes,
71   			       uint32_t file_flags)
72   	{
73   		int32_t fd;
74   		int32_t i;
75   		int32_t res = 0;
76   		ssize_t written;
77   		char *buffer = NULL;
78   		char *is_absolute = strchr(file, '/');
79   	
80   		if (is_absolute) {
81   			(void)strlcpy(path, file, PATH_MAX);
82   		} else {
83   	#if defined(QB_LINUX) || defined(QB_CYGWIN)
84   			snprintf(path, PATH_MAX, "/dev/shm/%s", file);
85   	#else
86   			snprintf(path, PATH_MAX, LOCALSTATEDIR "/run/%s", file);
87   			is_absolute = path;
88   	#endif
89   		}
90   		fd = open_mmap_file(path, file_flags);
91   		if (fd < 0 && !is_absolute) {
92   			qb_util_perror(LOG_ERR, "couldn't open file %s", path);
93   	
94   			snprintf(path, PATH_MAX, LOCALSTATEDIR "/run/%s", file);
95   			fd = open_mmap_file(path, file_flags);
96   			if (fd < 0) {
97   				res = -errno;
98   				qb_util_perror(LOG_ERR, "couldn't open file %s", path);
99   				return res;
100  			}
101  		} else if (fd < 0 && is_absolute) {
102  			res = -errno;
103  			qb_util_perror(LOG_ERR, "couldn't open file %s", path);
104  			return res;
105  		}
106  	
107  		if (ftruncate(fd, bytes) == -1) {
108  			res = -errno;
109  			qb_util_perror(LOG_ERR, "couldn't truncate file %s", path);
110  			goto unlink_exit;
111  		}
112  	
113  		if (file_flags & O_CREAT) {
114  			long page_size = sysconf(_SC_PAGESIZE);
115  			long write_size = QB_MIN(page_size, bytes);
116  			if (page_size < 0) {
117  				res = -errno;
118  				goto unlink_exit;
119  			}
120  			buffer = calloc(1, write_size);
121  			if (buffer == NULL) {
122  				res = -ENOMEM;
123  				goto unlink_exit;
124  			}
125  			for (i = 0; i < (bytes / write_size); i++) {
126  	retry_write:
127  				written = write(fd, buffer, write_size);
128  				if (written == -1 && errno == EINTR) {
129  					goto retry_write;
130  				}
131  				if (written != write_size) {
132  					res = -ENOSPC;
133  					free(buffer);
134  					goto unlink_exit;
135  				}
136  			}
137  			free(buffer);
138  		}
139  	
140  		return fd;
141  	
142  	unlink_exit:
143  		unlink(path);
144  		if (fd > 0) {
145  			close(fd);
146  		}
147  		return res;
148  	}

151  	int32_t
152  	qb_sys_circular_mmap(int32_t fd, void **buf, size_t bytes)
153  	{
154  		void *addr_orig = NULL;
155  		void *addr;
156  		void *addr_next;
157  		int32_t res;
158  		int flags = MAP_ANONYMOUS;
159  	
160  	#ifdef QB_FORCE_SHM_ALIGN
161  	/* On a number of arches any fixed and shared mmap() mapping address
162  	 * must be aligned to 16k. If the first mmap() below is not shared then
163  	 * the first mmap() will succeed because these restrictions do not apply to
164  	 * private mappings. The second mmap() wants a shared memory mapping but
165  	 * the address returned by the first one is only page-aligned and not
166  	 * aligned to 16k.
167  	 */
168  		flags |= MAP_SHARED;
169  	#else
170  		flags |= MAP_PRIVATE;
171  	#endif /* QB_FORCE_SHM_ALIGN */
172  	
173  		addr_orig = mmap(NULL, bytes << 1, PROT_NONE, flags, -1, 0);
174  	
175  		if (addr_orig == MAP_FAILED) {
176  			return -errno;
177  		}
178  	
179  		addr = mmap(addr_orig, bytes, PROT_READ | PROT_WRITE,
180  			    MAP_FIXED | MAP_SHARED, fd, 0);
181  	
182  		if (addr != addr_orig) {
183  			res = -errno;
184  			goto cleanup_fail;
185  		}
186  	#if defined(QB_BSD) && defined(MADV_NOSYNC)
187  		madvise(addr_orig, bytes, MADV_NOSYNC);
188  	#endif
189  		addr_next = ((char *)addr_orig) + bytes;
190  		addr = mmap(addr_next,
191  			    bytes, PROT_READ | PROT_WRITE,
192  			    MAP_FIXED | MAP_SHARED, fd, 0);
193  		if (addr != addr_next) {
194  			res = -errno;
195  			goto cleanup_fail;
196  		}
197  	#if defined(QB_BSD) && defined(MADV_NOSYNC)
198  		madvise(((char *)addr_orig) + bytes, bytes, MADV_NOSYNC);
199  	#endif
200  	
201  		res = close(fd);
202  		if (res) {
203  			goto cleanup_fail;
204  		}
205  		*buf = addr_orig;
206  		return 0;
207  	
208  	cleanup_fail:
209  	
210  		if (addr_orig) {
211  			munmap(addr_orig, bytes << 1);
212  		}
213  		close(fd);
214  		return res;
215  	}

217  	int32_t
218  	qb_sys_fd_nonblock_cloexec_set(int32_t fd)
219  	{
220  		int32_t res = 0;
221  		int32_t oldflags = fcntl(fd, F_GETFD, 0);
222  	
223  		if (oldflags < 0) {
224  			oldflags = 0;
225  		}
226  		oldflags |= FD_CLOEXEC;
227  		res = fcntl(fd, F_SETFD, oldflags);
228  		if (res == -1) {
229  			res = -errno;
230  			qb_util_perror(LOG_ERR,
231  				       "Could not set close-on-exit on fd:%d", fd);
232  			return res;
233  		}
234  	
235  		res = fcntl(fd, F_SETFL, O_NONBLOCK);
236  		if (res == -1) {
237  			res = -errno;
238  			qb_util_log(LOG_ERR, "Could not set non-blocking on fd:%d", fd);
239  		}
240  	
241  		return res;
242  	}

264  	int32_t
265  	qb_atomic_int_exchange_and_add(volatile int32_t QB_GNUC_MAY_ALIAS * atomic,
266  				       int32_t val)
267  	{
268  		int32_t result;
269  	
270  		qb_thread_lock(qb_atomic_mutex);
271  		result = *atomic;
272  		*atomic += val;
273  		qb_thread_unlock(qb_atomic_mutex);
274  	
275  		return result;
276  	}

286  	int32_t
287  	qb_atomic_int_compare_and_exchange(volatile int32_t QB_GNUC_MAY_ALIAS * atomic,
288  					   int32_t oldval, int32_t newval)
289  	{
290  		int32_t result;
291  	
292  		qb_thread_lock(qb_atomic_mutex);
293  		if (*atomic == oldval) {
294  			result = QB_TRUE;
295  			*atomic = newval;
296  		} else {
297  			result = QB_FALSE;
298  		}
299  		qb_thread_unlock(qb_atomic_mutex);
300  	
301  		return result;
302  	}
