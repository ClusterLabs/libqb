33   	static void
34   	store_this_qb(const char *fmt, ...)
35   	{
36   		char buf[QB_LOG_MAX_LEN];
37   		va_list ap;
38   	
39   		va_start(ap, fmt);
40   		qb_vsnprintf_serialize(buf, QB_LOG_MAX_LEN, fmt, ap);
41   		va_end(ap);
42   	}
43   	
44   	static void
45   	store_this_snprintf(const char *fmt, ...)
46   	{
47   		char buf[QB_LOG_MAX_LEN];
48   		va_list ap;
49   	
50   		va_start(ap, fmt);
51   		vsnprintf(buf, QB_LOG_MAX_LEN, fmt, ap);
52   		va_end(ap);
53   	}
