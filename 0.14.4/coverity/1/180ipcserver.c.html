100  	static int32_t
101  	s1_msg_process_fn(qb_ipcs_connection_t * c, void *data, size_t size)
102  	{
103  		struct qb_ipc_request_header *hdr;
104  		struct my_req *req_pt;
105  		struct qb_ipc_response_header response;
106  		ssize_t res;
107  		struct iovec iov[2];
108  		char resp[100];
109  		int32_t sl;
110  		int32_t send_ten_events = QB_FALSE;
111  	
112  		hdr = (struct qb_ipc_request_header *)data;
113  		if (hdr->id == (QB_IPC_MSG_USER_START + 1)) {
114  			return 0;
115  		}
116  	
117  		req_pt = (struct my_req *)data;
118  		qb_log(LOG_DEBUG, "msg received (id:%d, size:%d, data:%s)",
119  		       req_pt->hdr.id, req_pt->hdr.size, req_pt->message);
120  	
121  		if (strcmp(req_pt->message, "kill") == 0) {
122  			exit(0);
123  		}
124  		response.size = sizeof(struct qb_ipc_response_header);
125  		response.id = 13;
126  		response.error = 0;
127  	
128  		sl = snprintf(resp, 100, "ACK %zd bytes", size) + 1;
129  		iov[0].iov_len = sizeof(response);
130  		iov[0].iov_base = &response;
131  		iov[1].iov_len = sl;
132  		iov[1].iov_base = resp;
133  		response.size += sl;
134  	
135  		send_ten_events = (strcmp(req_pt->message, "events") == 0);
136  	
137  		if (use_events && !send_ten_events) {
138  			res = qb_ipcs_event_sendv(c, iov, 2);
139  		} else {
140  			res = qb_ipcs_response_sendv(c, iov, 2);
141  		}
142  		if (res < 0) {
143  			errno = - res;
144  			qb_perror(LOG_ERR, "qb_ipcs_response_send");
145  		}
146  		if (send_ten_events) {
147  			int32_t i;
148  			qb_log(LOG_INFO, "request to send 10 events");
149  			for (i = 0; i < 10; i++) {
150  				res = qb_ipcs_event_sendv(c, iov, 2);
151  				qb_log(LOG_INFO, "sent event %d res:%d", i, res);
152  			}
153  		}
154  		return 0;
155  	}

184  		GIOChannel *channel;
185  		int32_t events;

187  		qb_ipcs_dispatch_fn_t fn;

189  	};

191  	static gboolean
192  	gio_read_socket(GIOChannel * gio, GIOCondition condition, gpointer data)
193  	{
194  		struct gio_to_qb_poll *adaptor = (struct gio_to_qb_poll *)data;
195  		gint fd = g_io_channel_unix_get_fd(gio);
196  	
197  		return (adaptor->fn(fd, condition, adaptor->data) == 0);
198  	}

200  	static int32_t
201  	my_g_dispatch_add(enum qb_loop_priority p, int32_t fd, int32_t evts,
202  			  void *data, qb_ipcs_dispatch_fn_t fn)
203  	{
204  		struct gio_to_qb_poll *adaptor;
205  		GIOChannel *channel;
206  		int32_t res = 0;
207  	
208  		res = qb_array_index(gio_map, fd, (void **)&adaptor);
209  		if (res < 0) {
210  			return res;
211  		}
212  		if (adaptor->is_used) {
213  			return -EEXIST;
214  		}
215  	
216  		channel = g_io_channel_unix_new(fd);
217  		if (!channel) {
218  			return -ENOMEM;
219  		}
220  	
221  		adaptor->channel = channel;
222  		adaptor->fn = fn;
223  		adaptor->events = evts;
224  		adaptor->data = data;
225  		adaptor->p = p;
226  		adaptor->is_used = TRUE;
227  	
228  		g_io_add_watch(channel, evts, gio_read_socket, adaptor);
229  		return 0;
230  	}

240  	my_g_dispatch_del(int32_t fd)
241  	{
242  		struct gio_to_qb_poll *adaptor;
243  		if (qb_array_index(gio_map, fd, (void **)&adaptor) == 0) {

277  	int32_t
278  	main(int32_t argc, char *argv[])
279  	{
280  		const char *options = "mpseugh";
281  		int32_t opt;
282  		int32_t rc;
283  		enum qb_ipc_type ipc_type = QB_IPC_NATIVE;
284  		struct qb_ipcs_service_handlers sh = {
285  			.connection_accept = s1_connection_accept_fn,
286  			.connection_created = s1_connection_created_fn,
287  			.msg_process = s1_msg_process_fn,
288  			.connection_destroyed = s1_connection_destroyed_fn,
289  			.connection_closed = s1_connection_closed_fn,
290  		};
291  		struct qb_ipcs_poll_handlers ph = {
292  			.job_add = my_job_add,
293  			.dispatch_add = my_dispatch_add,
294  			.dispatch_mod = my_dispatch_mod,
295  			.dispatch_del = my_dispatch_del,
296  		};
297  	#ifdef HAVE_GLIB
298  		struct qb_ipcs_poll_handlers glib_ph = {
299  			.job_add = NULL, /* FIXME */
300  			.dispatch_add = my_g_dispatch_add,
301  			.dispatch_mod = my_g_dispatch_mod,
302  			.dispatch_del = my_g_dispatch_del,
303  		};
304  	#endif /* HAVE_GLIB */
305  	
306  		while ((opt = getopt(argc, argv, options)) != -1) {
307  			switch (opt) {
308  			case 'm':
309  				ipc_type = QB_IPC_SHM;
310  				break;
311  			case 'u':
312  				ipc_type = QB_IPC_SOCKET;
313  				break;
314  			case 'g':
315  				use_glib = QB_TRUE;
316  				break;
317  			case 'e':
318  				use_events = QB_TRUE;
319  				break;
320  			case 'h':
321  			default:
322  				show_usage(argv[0]);
323  				exit(0);
324  				break;
325  			}
326  		}
327  		signal(SIGINT, sigusr1_handler);
328  	
329  		qb_log_init("ipcserver", LOG_USER, LOG_TRACE);
330  		qb_log_filter_ctl(QB_LOG_STDERR, QB_LOG_FILTER_ADD,
331  				  QB_LOG_FILTER_FILE, "*", LOG_TRACE);
332  		qb_log_format_set(QB_LOG_STDERR, "%f:%l [%p] %b");
333  		qb_log_ctl(QB_LOG_STDERR, QB_LOG_CONF_ENABLED, QB_TRUE);
334  	
335  		s1 = qb_ipcs_create("ipcserver", 0, ipc_type, &sh);
336  		if (s1 == 0) {
337  			qb_perror(LOG_ERR, "qb_ipcs_create");
338  			exit(1);
339  		}
340  		if (!use_glib) {
341  			bms_loop = qb_loop_create();
342  			qb_ipcs_poll_handlers_set(s1, &ph);
343  			rc = qb_ipcs_run(s1);
344  			if (rc != 0) {
345  				errno = -rc;
346  				qb_perror(LOG_ERR, "qb_ipcs_run");
347  				exit(1);
348  			}
349  			qb_loop_run(bms_loop);
350  		} else {
351  	#ifdef HAVE_GLIB
352  			glib_loop = g_main_loop_new(NULL, FALSE);
353  			gio_map = qb_array_create_2(16, sizeof(struct gio_to_qb_poll), 1);
354  			qb_ipcs_poll_handlers_set(s1, &glib_ph);
355  			rc = qb_ipcs_run(s1);
356  			if (rc != 0) {
357  				errno = -rc;
358  				qb_perror(LOG_ERR, "qb_ipcs_run");
359  				exit(1);
360  			}
361  			g_main_loop_run(glib_loop);
362  	#else
363  			qb_log(LOG_ERR,
364  			       "You don't seem to have glib-devel installed.\n");
365  	#endif
366  		}
367  		return EXIT_SUCCESS;
368  	}
