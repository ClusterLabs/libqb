37   	/*
38   	 * move the write pointer to the next 128 byte boundary
39   	 * write_pt goes in 4 bytes (sizeof(uint32_t))
40   	 * #define USE_CACHE_LINE_ALIGNMENT 1
41   	 */
42   	#ifdef USE_CACHE_LINE_ALIGNMENT
43   	#define QB_CACHE_LINE_SIZE 128
44   	#define QB_CACHE_LINE_WORDS (QB_CACHE_LINE_SIZE/sizeof(uint32_t))
45   	#define idx_cache_line_step(idx)	\
46   	do {					\
47   		if (idx % QB_CACHE_LINE_WORDS) {			\
48   			idx += (QB_CACHE_LINE_WORDS - (idx % QB_CACHE_LINE_WORDS));	\
49   		}				\
50   		if (idx > (rb->shared_hdr->word_size - 1)) {		\
51   			idx = ((idx) % (rb->shared_hdr->word_size));	\
52   		}						\
53   	} while (0)
54   	#else
55   	#define QB_CACHE_LINE_SIZE 0
56   	#define QB_CACHE_LINE_WORDS 0
57   	#define idx_cache_line_step(idx)			\
58   	do {							\
59   		if (idx > (rb->shared_hdr->word_size - 1)) {		\
60   			idx = ((idx) % (rb->shared_hdr->word_size));	\
61   		}						\
62   	} while (0)
63   	#endif

66   	/* the chunk header is two words
67   	 * 1) the chunk data size
68   	 * 2) the magic number
69   	 */
70   	#define QB_RB_CHUNK_HEADER_WORDS 2
71   	#define QB_RB_CHUNK_HEADER_SIZE (sizeof(uint32_t) * QB_RB_CHUNK_HEADER_WORDS)
72   	/*
73   	 * margin is the gap we leave when checking to see if we have enough
74   	 * space for a new chunk.
75   	 * So:
76   	 * qb_rb_space_free() >= QB_RB_CHUNK_MARGIN + new data chunk
77   	 * The extra word size is to allow for non word sized data chunks.
78   	 * QB_CACHE_LINE_WORDS is to make sure we have space to align the
79   	 * chunk.
80   	 */
81   	#define QB_RB_WORD_ALIGN 1
82   	#define QB_RB_CHUNK_MARGIN (sizeof(uint32_t) * (QB_RB_CHUNK_HEADER_WORDS +\
83   							QB_RB_WORD_ALIGN +\
84   							QB_CACHE_LINE_WORDS))
85   	#define QB_RB_CHUNK_MAGIC		0xA1A1A1A1
86   	#define QB_RB_CHUNK_MAGIC_DEAD		0xD0D0D0D0
87   	#define QB_RB_CHUNK_MAGIC_ALLOC		0xA110CED0
88   	#define QB_RB_CHUNK_SIZE_GET(rb, pointer) \
89   		rb->shared_data[pointer]
90   	#define QB_RB_CHUNK_MAGIC_GET(rb, pointer) \
91   		rb->shared_data[(pointer + 1) % rb->shared_hdr->word_size]
92   	#define QB_RB_CHUNK_MAGIC_SET(rb, pointer, new_val) \
93   		rb->shared_data[(pointer + 1) % rb->shared_hdr->word_size] = new_val;
94   	#define QB_RB_CHUNK_DATA_GET(rb, pointer) \
95   		&rb->shared_data[(pointer + QB_RB_CHUNK_HEADER_WORDS) % rb->shared_hdr->word_size]
96   	
97   	#define QB_MAGIC_ASSERT(_ptr_) \
98   	do {							\
99   		uint32_t chunk_magic = QB_RB_CHUNK_MAGIC_GET(rb, _ptr_); \
100  		if (chunk_magic != QB_RB_CHUNK_MAGIC) print_header(rb); \
101  		assert(chunk_magic == QB_RB_CHUNK_MAGIC); \
102  	} while (0)
103  	
104  	#define idx_step(idx)					\
105  	do {							\
106  		if (idx > (rb->shared_hdr->word_size - 1)) {		\
107  			idx = ((idx) % (rb->shared_hdr->word_size));	\
108  		}						\
109  	} while (0)

114  	qb_ringbuffer_t *
115  	qb_rb_open(const char *name, size_t size, uint32_t flags,
116  		   size_t shared_user_data_size)
117  	{
118  		struct qb_ringbuffer_s *rb;
119  		size_t real_size;
120  		size_t shared_size;
121  		char path[PATH_MAX];
122  		int32_t fd_hdr;
123  		int32_t fd_data;
124  		uint32_t file_flags = O_RDWR;
125  		char filename[PATH_MAX];
126  		int32_t error = 0;
127  		void *shm_addr;
128  		long page_size = sysconf(_SC_PAGESIZE);
129  	
130  	#ifdef QB_FORCE_SHM_ALIGN
131  		page_size = QB_MAX(page_size, 16 * 1024);
132  	#endif /* QB_FORCE_SHM_ALIGN */
133  		real_size = QB_ROUNDUP(size, page_size);
134  		shared_size =
135  		    sizeof(struct qb_ringbuffer_shared_s) + shared_user_data_size;
136  	
137  		if (flags & QB_RB_FLAG_CREATE) {
138  			file_flags |= O_CREAT | O_TRUNC;
139  		}
140  	
141  		rb = calloc(1, sizeof(struct qb_ringbuffer_s));
142  		if (rb == NULL) {
143  			return NULL;
144  		}
145  	
146  		/*
147  		 * Create a shared_hdr memory segment for the header.
148  		 */
149  		snprintf(filename, PATH_MAX, "qb-%s-header", name);
150  		fd_hdr = qb_sys_mmap_file_open(path, filename,
151  					       shared_size, file_flags);
152  		if (fd_hdr < 0) {
153  			error = fd_hdr;
154  			qb_util_log(LOG_ERR, "couldn't create file for mmap");
155  			goto cleanup_hdr;
156  		}
157  	
158  		rb->shared_hdr = mmap(0,
159  				      shared_size,
160  				      PROT_READ | PROT_WRITE, MAP_SHARED, fd_hdr, 0);
161  	
162  		if (rb->shared_hdr == MAP_FAILED) {
163  			error = -errno;
164  			qb_util_log(LOG_ERR, "couldn't create mmap for header");
165  			goto cleanup_hdr;
166  		}
167  		qb_atomic_init();
168  	
169  		rb->flags = flags;
170  	
171  		/*
172  		 * create the semaphore
173  		 */
174  		if (flags & QB_RB_FLAG_CREATE) {
175  			rb->shared_data = NULL;
176  			/* rb->shared_hdr->word_size tracks data by ints and not bytes/chars. */
177  			rb->shared_hdr->word_size = real_size / sizeof(uint32_t);
178  			rb->shared_hdr->write_pt = 0;
179  			rb->shared_hdr->read_pt = 0;
180  			(void)strlcpy(rb->shared_hdr->hdr_path, path, PATH_MAX);
181  		}
182  		error = qb_rb_sem_create(rb, flags);
183  		if (error < 0) {
184  			qb_util_perror(LOG_ERR, "couldn't get a semaphore");
185  			goto cleanup_hdr;
186  		}
187  	
188  		/* Create the shared_data memory segment for the actual ringbuffer.
189  		 * They have to be separate.
190  		 */
191  		if (flags & QB_RB_FLAG_CREATE) {
192  			snprintf(filename, PATH_MAX, "qb-%s-data", name);
193  			fd_data = qb_sys_mmap_file_open(path,
194  							filename,
195  							real_size, file_flags);
196  			(void)strlcpy(rb->shared_hdr->data_path, path, PATH_MAX);
197  		} else {
198  			fd_data = qb_sys_mmap_file_open(path,
199  							rb->shared_hdr->data_path,
200  							real_size, file_flags);
201  		}
202  		if (fd_data < 0) {
203  			error = fd_data;
204  			qb_util_log(LOG_ERR, "couldn't create file for mmap");
205  			goto cleanup_hdr;
206  		}
207  	
208  		qb_util_log(LOG_DEBUG,
209  			    "shm size:%zd; real_size:%zd; rb->word_size:%d", size,
210  			    real_size, rb->shared_hdr->word_size);
211  	
212  		error = qb_sys_circular_mmap(fd_data, &shm_addr, real_size);
213  		rb->shared_data = shm_addr;
214  		if (error != 0) {
215  			qb_util_log(LOG_ERR, "couldn't create circular mmap on %s",
216  				    rb->shared_hdr->data_path);
217  			goto cleanup_data;
218  		}
219  	
220  		if (flags & QB_RB_FLAG_CREATE) {
221  			memset(rb->shared_data, 0, real_size);
222  			rb->shared_data[rb->shared_hdr->word_size] = 5;
223  			rb->shared_hdr->ref_count = 1;
224  		} else {
225  			qb_atomic_int_inc(&rb->shared_hdr->ref_count);
226  		}
227  	
228  		close(fd_hdr);
229  		close(fd_data);
230  		return rb;
231  	
232  	cleanup_data:
233  		close(fd_data);
234  		if (flags & QB_RB_FLAG_CREATE) {
235  			unlink(rb->shared_hdr->data_path);
236  		}
237  	
238  	cleanup_hdr:
239  		if (fd_hdr >= 0) {
240  			close(fd_hdr);
241  		}
242  		if (rb && (flags & QB_RB_FLAG_CREATE)) {
243  			unlink(rb->shared_hdr->hdr_path);
244  			if (rb->sem_destroy_fn) {
245  				(void)rb->sem_destroy_fn(rb);
246  			}
247  		}
248  		if (rb && (rb->shared_hdr != MAP_FAILED && rb->shared_hdr != NULL)) {
249  			munmap(rb->shared_hdr, sizeof(struct qb_ringbuffer_shared_s));
250  		}
251  		free(rb);
252  		errno = -error;
253  		return NULL;
254  	}

256  	void
257  	qb_rb_close(struct qb_ringbuffer_s * rb)
258  	{
(1) Event cond_false: Condition "rb == NULL", taking false branch 
259  		if (rb == NULL) {
260  			return;
(2) Event if_end: End of if statement 
261  		}
262  	
263  		(void)qb_atomic_int_dec_and_test(&rb->shared_hdr->ref_count);
(3) Event read_parm_fld: Reading a parameter field. 
264  		if (rb->flags & QB_RB_FLAG_CREATE) {
265  			if (rb->sem_destroy_fn) {
266  				(void)rb->sem_destroy_fn(rb);
267  			}
268  			unlink(rb->shared_hdr->data_path);
269  			unlink(rb->shared_hdr->hdr_path);
270  			qb_util_log(LOG_DEBUG,
271  				    "Free'ing ringbuffer: %s",
272  				    rb->shared_hdr->hdr_path);
273  		} else {
274  			qb_util_log(LOG_DEBUG,
275  				    "Closing ringbuffer: %s", rb->shared_hdr->hdr_path);
276  		}
277  		munmap(rb->shared_data, (rb->shared_hdr->word_size * sizeof(uint32_t)) << 1);
278  		munmap(rb->shared_hdr, sizeof(struct qb_ringbuffer_shared_s));
279  		free(rb);
280  	}

329  	ssize_t
330  	qb_rb_space_free(struct qb_ringbuffer_s * rb)
331  	{
332  		uint32_t write_size;
333  		uint32_t read_size;
334  		size_t space_free = 0;
335  	
336  		if (rb == NULL) {
337  			return -EINVAL;
338  		}
339  		write_size = rb->shared_hdr->write_pt;
340  		read_size = rb->shared_hdr->read_pt;
341  	
342  		if (write_size > read_size) {
343  			space_free =
344  			    (read_size - write_size + rb->shared_hdr->word_size) - 1;
345  		} else if (write_size < read_size) {
346  			space_free = (read_size - write_size) - 1;
347  		} else {
348  			if (rb->sem_getvalue_fn && rb->sem_getvalue_fn(rb) > 0) {
349  				space_free = 0;
350  			} else {
351  				space_free = rb->shared_hdr->word_size;
352  			}
353  		}
354  	
355  		/* word -> bytes */
356  		return (space_free * sizeof(uint32_t));
357  	}

359  	ssize_t
360  	qb_rb_space_used(struct qb_ringbuffer_s * rb)
361  	{
362  		uint32_t write_size;
363  		uint32_t read_size;
364  		size_t space_used;
365  	
366  		if (rb == NULL) {
367  			return -EINVAL;
368  		}
369  		write_size = rb->shared_hdr->write_pt;
370  		read_size = rb->shared_hdr->read_pt;
371  	
372  		if (write_size > read_size) {
373  			space_used = write_size - read_size;
374  		} else if (write_size < read_size) {
375  			space_used =
376  			    (write_size - read_size + rb->shared_hdr->word_size) - 1;
377  		} else {
378  			space_used = 0;
379  		}
380  		/* word -> bytes */
381  		return (space_used * sizeof(uint32_t));
382  	}

397  	void *
398  	qb_rb_chunk_alloc(struct qb_ringbuffer_s * rb, size_t len)
399  	{
400  		uint32_t write_pt;
401  	
402  		if (rb == NULL) {
403  			errno = EINVAL;
404  			return NULL;
405  		}
406  		/*
407  		 * Reclaim data if we are over writing and we need space
408  		 */
409  		if (rb->flags & QB_RB_FLAG_OVERWRITE) {
410  			while (qb_rb_space_free(rb) < (len + QB_RB_CHUNK_MARGIN)) {
411  				_rb_chunk_reclaim(rb);
412  			}
413  		} else {
414  			if (qb_rb_space_free(rb) < (len + QB_RB_CHUNK_MARGIN)) {
415  				errno = EAGAIN;
416  				return NULL;
417  			}
418  		}
419  	
420  		write_pt = rb->shared_hdr->write_pt;
421  		/*
422  		 * insert the chunk header
423  		 */
424  		rb->shared_data[write_pt] = 0;
425  		QB_RB_CHUNK_MAGIC_SET(rb, write_pt, QB_RB_CHUNK_MAGIC_ALLOC);
426  	
427  		/*
428  		 * return a pointer to the beginning of the chunk data
429  		 */
430  		return (void *)QB_RB_CHUNK_DATA_GET(rb, write_pt);
431  	
432  	}

434  	static uint32_t
435  	qb_rb_chunk_step(struct qb_ringbuffer_s * rb, uint32_t pointer)
436  	{
437  		uint32_t chunk_size = QB_RB_CHUNK_SIZE_GET(rb, pointer);
438  		/*
439  		 * skip over the chunk header
440  		 */
441  		pointer += QB_RB_CHUNK_HEADER_WORDS;
442  	
443  		/*
444  		 * skip over the user's data.
445  		 */
446  		pointer += (chunk_size / sizeof(uint32_t));
447  		/* make allowance for non-word sizes */
448  		if ((chunk_size % (sizeof(uint32_t) * QB_RB_WORD_ALIGN)) != 0) {
449  			pointer++;
450  		}
451  	
452  		idx_cache_line_step(pointer);
453  		return pointer;
454  	}

456  	int32_t
457  	qb_rb_chunk_commit(struct qb_ringbuffer_s * rb, size_t len)
458  	{
459  		uint32_t old_write_pt;
460  	
461  		if (rb == NULL) {
462  			return -EINVAL;
463  		}
464  		/*
465  		 * commit the magic & chunk_size
466  		 */
467  		old_write_pt = rb->shared_hdr->write_pt;
468  		rb->shared_data[old_write_pt] = len;
469  	
470  		/*
471  		 * commit the new write pointer
472  		 */
473  		rb->shared_hdr->write_pt = qb_rb_chunk_step(rb, old_write_pt);
474  		QB_RB_CHUNK_MAGIC_SET(rb, old_write_pt, QB_RB_CHUNK_MAGIC);
475  	
476  		DEBUG_PRINTF("commit [%zd] read: %u, write: %u -> %u (%u)\n",
477  			     (rb->sem_getvalue_fn ? rb->sem_getvalue_fn(rb) : 0),
478  			     rb->shared_hdr->read_pt,
479  			     old_write_pt,
480  			     rb->shared_hdr->write_pt,
481  			     rb->shared_hdr->word_size);
482  	
483  		/*
484  		 * post the notification to the reader
485  		 */
486  		if (rb->sem_post_fn) {
487  			return rb->sem_post_fn(rb);
488  		} else {
489  			return 0;
490  		}
491  	}

493  	ssize_t
494  	qb_rb_chunk_write(struct qb_ringbuffer_s * rb, const void *data, size_t len)
495  	{
496  		char *dest = qb_rb_chunk_alloc(rb, len);
497  		int32_t res = 0;
498  	
499  		if (rb == NULL) {
500  			return -EINVAL;
501  		}
502  	
503  		if (dest == NULL) {
504  			return -errno;
505  		}
506  	
507  		memcpy(dest, data, len);
508  	
509  		res = qb_rb_chunk_commit(rb, len);
510  		if (res < 0) {
511  			return res;
512  		}
513  	
514  		return len;
515  	}

517  	static void
518  	_rb_chunk_reclaim(struct qb_ringbuffer_s * rb)
519  	{
520  		uint32_t old_read_pt;
521  		uint32_t new_read_pt;
522  		uint32_t chunk_magic;
523  	
524  		old_read_pt = rb->shared_hdr->read_pt;
525  		chunk_magic = QB_RB_CHUNK_MAGIC_GET(rb, old_read_pt);
526  		if (chunk_magic != QB_RB_CHUNK_MAGIC) {
527  			return;
528  		}
529  	
530  		new_read_pt = qb_rb_chunk_step(rb, old_read_pt);
531  	
532  		/*
533  		 * clear the header
534  		 */
535  		rb->shared_data[old_read_pt] = 0;
536  		QB_RB_CHUNK_MAGIC_SET(rb, old_read_pt, QB_RB_CHUNK_MAGIC_DEAD);
537  	
538  		/*
539  		 * set the new read pointer after clearing the header
540  		 * to prevent a situation where a fast writer will write their
541  		 * new chunk between setting the new read pointer and clearing the
542  		 * header.
543  		 */
544  		rb->shared_hdr->read_pt = new_read_pt;
545  	
546  		DEBUG_PRINTF("reclaim [%zd]: read: %u -> %u, write: %u\n",
547  			     (rb->sem_getvalue_fn ? rb->sem_getvalue_fn(rb) : 0),
548  			     old_read_pt,
549  			     rb->shared_hdr->read_pt,
550  			     rb->shared_hdr->write_pt);
551  	}

553  	void
554  	qb_rb_chunk_reclaim(struct qb_ringbuffer_s * rb)
555  	{
556  		if (rb == NULL) {
557  			return;
558  		}
559  		_rb_chunk_reclaim(rb);
560  	}

562  	ssize_t
563  	qb_rb_chunk_peek(struct qb_ringbuffer_s * rb, void **data_out, int32_t timeout)
564  	{
565  		uint32_t read_pt;
566  		uint32_t chunk_size;
567  		uint32_t chunk_magic;
568  		int32_t res = 0;
569  	
570  		if (rb == NULL) {
571  			return -EINVAL;
572  		}
573  		if (rb->sem_timedwait_fn) {
574  			res = rb->sem_timedwait_fn(rb, timeout);
575  		}
576  		if (res < 0 && res != -EIDRM) {
577  			if (res == -ETIMEDOUT) {
578  				return 0;
579  			} else {
580  				errno = -res;
581  				qb_util_perror(LOG_ERR, "sem_timedwait");
582  			}
583  			return res;
584  		}
585  		read_pt = rb->shared_hdr->read_pt;
586  		chunk_magic = QB_RB_CHUNK_MAGIC_GET(rb, read_pt);
587  		if (chunk_magic != QB_RB_CHUNK_MAGIC) {
588  			if (rb->sem_post_fn) {
589  				(void)rb->sem_post_fn(rb);
590  			}
591  			return 0;
592  		}
593  		chunk_size = QB_RB_CHUNK_SIZE_GET(rb, read_pt);
594  		*data_out = QB_RB_CHUNK_DATA_GET(rb, read_pt);
595  		return chunk_size;
596  	}

598  	ssize_t
599  	qb_rb_chunk_read(struct qb_ringbuffer_s * rb, void *data_out, size_t len,
600  			 int32_t timeout)
601  	{
602  		uint32_t read_pt;
603  		uint32_t chunk_size;
604  		uint32_t chunk_magic;
605  		int32_t res = 0;
606  	
607  		if (rb == NULL) {
608  			return -EINVAL;
609  		}
610  		if (rb->sem_timedwait_fn) {
611  			res = rb->sem_timedwait_fn(rb, timeout);
612  		}
613  		if (res < 0 && res != -EIDRM) {
614  			if (res != -ETIMEDOUT) {
615  				qb_util_perror(LOG_ERR, "sem_timedwait");
616  			}
617  			return res;
618  		}
619  	
620  		read_pt = rb->shared_hdr->read_pt;
621  		chunk_magic = QB_RB_CHUNK_MAGIC_GET(rb, read_pt);
622  	
623  		if (chunk_magic != QB_RB_CHUNK_MAGIC) {
624  			if (rb->sem_timedwait_fn == NULL) {
625  				return -ETIMEDOUT;
626  			} else {
627  				(void)rb->sem_post_fn(rb);
628  	#ifdef EBADMSG
629  				return -EBADMSG;
630  	#else
631  				return -EINVAL;
632  	#endif
633  			}
634  		}
635  	
636  		chunk_size = QB_RB_CHUNK_SIZE_GET(rb, read_pt);
637  		if (len < chunk_size) {
638  			qb_util_log(LOG_ERR,
639  				    "trying to recv chunk of size %d but %d available",
640  				    len, chunk_size);
641  			(void)rb->sem_post_fn(rb);
642  			return -ENOBUFS;
643  		}
644  		;
645  		memcpy(data_out,
646  		       QB_RB_CHUNK_DATA_GET(rb, read_pt),
647  		       chunk_size);
648  	
649  		_rb_chunk_reclaim(rb);
650  	
651  		return chunk_size;
652  	}

656  	{

668  		printf(" =>used [%zu bytes]\n", qb_rb_space_used(rb));

670  	}
