24   	static int32_t
25   	my_posix_sem_timedwait(qb_ringbuffer_t * rb, int32_t ms_timeout)
26   	{
27   		struct timespec ts_timeout;
28   		int32_t res;
29   	
30   		if (ms_timeout > 0) {
31   			qb_util_timespec_from_epoch_get(&ts_timeout);
32   			qb_timespec_add_ms(&ts_timeout, ms_timeout);
33   		}
34   	
35   	sem_wait_again:
36   		if (ms_timeout > 0) {
37   			res = rpl_sem_timedwait(&rb->shared_hdr->posix_sem, &ts_timeout);
38   		} else if (ms_timeout == 0) {
39   			res = rpl_sem_trywait(&rb->shared_hdr->posix_sem);
40   		} else {
41   			res = rpl_sem_wait(&rb->shared_hdr->posix_sem);
42   		}
43   		if (res == -1) {
44   			switch (errno) {
45   			case EINTR:
46   				goto sem_wait_again;
47   				break;
48   			case EAGAIN:
49   				res = -ETIMEDOUT;
50   				break;
51   			case ETIMEDOUT:
52   				res = -errno;
53   				break;
54   			default:
55   				res = -errno;
56   				qb_util_perror(LOG_ERR, "error waiting for semaphore");
57   				break;
58   			}
59   		}
60   		return res;
61   	}

63   	static int32_t
64   	my_posix_sem_post(qb_ringbuffer_t * rb)
65   	{
66   		if (rpl_sem_post(&rb->shared_hdr->posix_sem) < 0) {
67   			return -errno;
68   		} else {
69   			return 0;
70   		}
71   	}

74   	my_posix_getvalue_fn(struct qb_ringbuffer_s *rb)
75   	{
76   		int val;
77   		if (rpl_sem_getvalue(&rb->shared_hdr->posix_sem, &val) < 0) {
78   			return -errno;

94   	static int32_t
95   	my_posix_sem_create(struct qb_ringbuffer_s *rb, uint32_t flags)
96   	{
97   		int32_t pshared = QB_FALSE;
98   		if (flags & QB_RB_FLAG_SHARED_PROCESS) {
99   			if ((flags & QB_RB_FLAG_CREATE) == 0) {
100  				return 0;
101  			}
102  			pshared = QB_TRUE;
103  		}
104  		if (rpl_sem_init(&rb->shared_hdr->posix_sem, pshared, 0) == -1) {
105  			return -errno;
106  		} else {
107  			return 0;
108  		}
109  	}

111  	static int32_t
112  	my_sysv_sem_timedwait(qb_ringbuffer_t * rb, int32_t ms_timeout)
113  	{
114  		struct sembuf sops[1];
115  		int32_t res = 0;
116  	#ifdef HAVE_SEMTIMEDOP
117  		struct timespec ts_timeout;
118  		struct timespec *ts_pt;
119  	
120  		if (ms_timeout >= 0) {
121  			/*
122  			 * Note: sem_timedwait takes an absolute time where as semtimedop
123  			 * takes a relative time.
124  			 */
125  			ts_timeout.tv_sec = 0;
126  			ts_timeout.tv_nsec = 0;
127  			qb_timespec_add_ms(&ts_timeout, ms_timeout);
128  			ts_pt = &ts_timeout;
129  		} else {
130  			ts_pt = NULL;
131  		}
132  	#endif /* HAVE_SEMTIMEDOP */
133  	
134  		/*
135  		 * wait for sem post.
136  		 */
137  		sops[0].sem_num = 0;
138  		sops[0].sem_op = -1;
139  	#ifdef HAVE_SEMTIMEDOP
140  		sops[0].sem_flg = 0;
141  	#else
142  		sops[0].sem_flg = IPC_NOWAIT;
143  	#endif /* HAVE_SEMTIMEDOP */
144  	
145  	semop_again:
146  	#ifdef HAVE_SEMTIMEDOP
147  		if (semtimedop(rb->sem_id, sops, 1, ts_pt) == -1)
148  	#else
149  		if (semop(rb->sem_id, sops, 1) == -1)
150  	#endif /* HAVE_SEMTIMEDOP */
151  		{
152  			if (errno == EINTR) {
153  				goto semop_again;
154  			} else if (errno == EAGAIN) {
155  				/* make consistent with sem_timedwait */
156  				res = -ETIMEDOUT;
157  			} else {
158  				res = -errno;
159  				qb_util_perror(LOG_ERR, "error waiting for semaphore");
160  			}
161  			return res;
162  		}
163  		return 0;
164  	}

166  	static int32_t
167  	my_sysv_sem_post(qb_ringbuffer_t * rb)
168  	{
169  		struct sembuf sops[1];
170  	
171  		if ((rb->flags & QB_RB_FLAG_SHARED_PROCESS) == 0) {
172  			return 0;
173  		}
174  	
175  		sops[0].sem_num = 0;
176  		sops[0].sem_op = 1;
177  		sops[0].sem_flg = 0;
178  	
179  	semop_again:
180  		if (semop(rb->sem_id, sops, 1) == -1) {
181  			if (errno == EINTR) {
182  				goto semop_again;
183  			} else {
(1) Event tainted_string_warning: Using a non string-literal such as "descriptor.format" as a format-string argument can lead to a vulnerability if a user can control its content. [details] 
184  				qb_util_perror(LOG_ERR,
185  					       "could not increment semaphore");
186  			}
187  	
188  			return -errno;
189  		}
190  		return 0;
191  	}

193  	static ssize_t
194  	my_sysv_getvalue_fn(struct qb_ringbuffer_s *rb)
195  	{
196  		ssize_t res = semctl(rb->sem_id, 0, GETVAL, 0);
197  		if (res == -1) {
198  			return -errno;
199  		}
200  		return res;
201  	}

204  	my_sysv_sem_destroy(qb_ringbuffer_t * rb)
205  	{
206  		if (semctl(rb->sem_id, 0, IPC_RMID, 0) == -1) {
207  			return -errno;
208  		} else {
209  			return 0;
210  		}
211  	}

213  	static int32_t
214  	my_sysv_sem_create(qb_ringbuffer_t * rb, uint32_t flags)
215  	{
216  		union semun options;
217  		int32_t res;
218  		key_t sem_key;
219  	
220  		sem_key = ftok(rb->shared_hdr->hdr_path, (rb->shared_hdr->word_size + 1));
221  	
222  		if (sem_key == -1) {
223  			res = -errno;
224  			qb_util_perror(LOG_ERR, "couldn't get a sem id");
225  			return res;
226  		}
227  	
228  		if (flags & QB_RB_FLAG_CREATE) {
229  			rb->sem_id = semget(sem_key, 1, IPC_CREAT | IPC_EXCL | 0600);
230  			if (rb->sem_id == -1) {
231  				res = -errno;
232  				qb_util_perror(LOG_ERR, "couldn't create a semaphore");
233  				return res;
234  			}
235  			options.val = 0;
236  			res = semctl(rb->sem_id, 0, SETVAL, options);
237  		} else {
238  			rb->sem_id = semget(sem_key, 0, 0600);
239  			if (rb->sem_id == -1) {
240  				res = -errno;
241  				qb_util_perror(LOG_ERR, "couldn't get a sem id");
242  				return res;
243  			}
244  			res = 0;
245  		}
246  		qb_util_log(LOG_DEBUG, "sem key:%d, id:%d, value:%d",
247  			    (int)sem_key, rb->sem_id, semctl(rb->sem_id, 0, GETVAL, 0));
248  	
249  		return res;
250  	}

252  	int32_t
253  	qb_rb_sem_create(struct qb_ringbuffer_s * rb, uint32_t flags)
254  	{
255  		int32_t rc;
256  		int32_t use_posix = QB_TRUE;
257  	
258  		if ((flags & QB_RB_FLAG_SHARED_PROCESS) &&
259  		    !(flags & QB_RB_FLAG_NO_SEMAPHORE)) {
260  	#if defined(HAVE_POSIX_PSHARED_SEMAPHORE) || \
261  	    defined(HAVE_RPL_PSHARED_SEMAPHORE)
262  			use_posix = QB_TRUE;
263  	#else
264  		#ifdef HAVE_SYSV_PSHARED_SEMAPHORE
265  			use_posix = QB_FALSE;
266  		#else
267  			return -ENOTSUP;
268  		#endif /* HAVE_SYSV_PSHARED_SEMAPHORE */
269  	#endif /* HAVE_POSIX_PSHARED_SEMAPHORE */
270  		}
271  		if (flags & QB_RB_FLAG_NO_SEMAPHORE) {
272  			rc = 0;
273  			rb->sem_timedwait_fn = NULL;
274  			rb->sem_post_fn = NULL;
275  			rb->sem_getvalue_fn = NULL;
276  			rb->sem_destroy_fn = NULL;
277  		} else if (use_posix) {
278  			rc = my_posix_sem_create(rb, flags);
279  			rb->sem_timedwait_fn = my_posix_sem_timedwait;
280  			rb->sem_post_fn = my_posix_sem_post;
281  			rb->sem_getvalue_fn = my_posix_getvalue_fn;
282  			rb->sem_destroy_fn = my_posix_sem_destroy;
283  		} else {
284  			rc = my_sysv_sem_create(rb, flags);
285  			rb->sem_timedwait_fn = my_sysv_sem_timedwait;
286  			rb->sem_post_fn = my_sysv_sem_post;
287  			rb->sem_getvalue_fn = my_sysv_getvalue_fn;
288  			rb->sem_destroy_fn = my_sysv_sem_destroy;
289  		}
290  		return rc;
291  	}
